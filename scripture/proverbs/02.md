function proverbs_2_wisdomsValueAndProtection() {
  // === 2:1-8 - The Pursuit of Wisdom ===
  parent.exhortSon({
    condition: [
      "if_you_receive_my_words",
      "treasure_up_my_commandments_with_you",
      "make_your_ear_attentive_to_wisdom",
      "incline_your_heart_to_understanding",
      "if_you_call_out_for_insight",
      "raise_your_voice_for_understanding",
      "if_you_seek_it_like_silver",
      "search_for_it_as_for_hidden_treasures"
    ],
    promise: [
      {
        outcome: "you_will_understand_fear_of_the_Lord",
        detail: "find_knowledge_of_God"
      },
      {
        source: "for_the_Lord_gives_wisdom",
        method: [
          "from_His_mouth_come_knowledge_and_understanding",
          "He_stores_up_sound_wisdom_for_the_upright",
          "He_is_a_shield_to_those_who_walk_in_integrity"
        ],
        purpose: [
          "guarding_paths_of_justice",
          "watching_over_way_of_His_saints"
        ]
      }
    ]
  });

  // === 2:9-15 - Benefits of Wisdom ===
  wisdom.deliverBenefits({
    outcome: [
      "you_will_understand_righteousness_justice_and_equity",
      "every_good_path"
    ],
    mechanism: [
      {
        agent: "wisdom_will_come_into_your_heart",
        effect: "knowledge_will_be_pleasant_to_your_soul"
      },
      {
        agent: "discretion_will_watch_over_you",
        effect: "understanding_will_guard_you"
      }
    ],
    protection: {
      from: "way_of_evil",
      description: [
        {
          type: "men_of_perverted_speech",
          behavior: [
            "forsake_paths_of_uprightness",
            "walk_in_ways_of_darkness",
            "rejoice_in_doing_evil",
            "delight_in_perverseness_of_evil"
          ],
          character: "crooked_in_their_paths",
          intent: "devious_in_their_ways"
        }
      ]
    }
  });

  // === 2:16-19 - Protection from Immorality ===
  wisdom.protectFromImmoralWoman({
    description: {
      type: "adulteress_with_seductive_words",
      behavior: [
        "forsaken_partner_of_her_youth",
        "forgotten_covenant_of_her_God"
      ]
    },
    consequence: [
      "her_house_sinks_down_to_death",
      "her_paths_to_the_shades",
      "none_who_go_to_her_come_back",
      "nor_do_they_regain_paths_of_life"
    ]
  });

  // === 2:20-22 - The Path of the Righteous vs. the Wicked ===
  wisdom.concludeWithContrast({
    righteous: {
      instruction: "walk_in_way_of_the_good",
      path: "keep_to_paths_of_the_righteous",
      outcome: [
        "the_upright_will_inhabit_the_land",
        "those_with_integrity_will_remain_in_it"
      ]
    },
    wicked: {
      outcome: [
        "the_wicked_will_be_cut_off_from_the_land",
        "the_treacherous_will_be_rooted_out_of_it"
      ]
    }
  });
}
