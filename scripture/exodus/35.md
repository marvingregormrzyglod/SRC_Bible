# Exodus 35
```
function exodus_35_tabernaclePreparations() {
  // === 35:1-3 - Sabbath Command Reiterated ===
  moses.addressIsrael({
    context: {
      audience: "all_congregation_of_Israel",
      setting: "after_descending_from_Mount_Sinai"
    },
    command: {
      sabbath: {
        instruction: [
          "work_shall_be_done_six_days",
          "seventh_day_shall_be_holy_Sabbath_of_rest_to_Yahweh"
        ],
        prohibition: [
          "whoever_does_work_on_it_shall_be_put_to_death",
          "kindle_no_fire_in_any_dwelling_on_Sabbath_day"
        ]
      }
    }
  });

  // === 35:4-19 - Call for Tabernacle Offerings ===
  moses.requestContributions({
    audience: "all_congregation_of_Israel",
    instruction: {
      source: "Yahweh_has_commanded",
      offering: {
        type: "freewill_offering_to_Yahweh",
        contributors: "everyone_whose_heart_stirs_him",
        materials: [
          "gold_silver_bronze",
          "blue_purple_scarlet_yarn_fine_linen_goats_hair",
          "tanned_rams_skins_badgers_skins_acacia_wood",
          "oil_for_light_spices_for_anointing_oil_and_incense",
          "onyx_stones_and_stones_for_setting_in_ephod_and_breastpiece"
        ]
      },
      purpose: {
        tabernacle: [
          "make_all_Yahweh_has_commanded",
          "tabernacle_and_its_tent_and_coverings",
          "clasps_frames_bars_pillars_sockets",
          "ark_and_its_poles_mercy_seat_and_veil_of_screen",
          "table_its_poles_all_its_utensils_and_bread_of_Presence",
          "lampstand_its_utensils_lamps_and_oil_for_light",
          "altar_of_incense_its_poles_anointing_oil_sweet_incense",
          "screen_for_door_of_tabernacle",
          "altar_of_burnt_offering_its_grate_poles_utensils",
          "bronze_basin_and_its_stand",
          "hangings_of_court_its_pillars_sockets_screen_for_court_gate",
          "pegs_of_tabernacle_and_court_and_their_cords",
          "service_garments_for_ministering_in_holy_place",
          "holy_garments_for_Aaron_and_his_sons"
        ]
      }
    }
  });

  // === 35:20-29 - Israel's Generous Response ===
  israelites.bringOfferings({
    context: {
      trigger: "departed_from_Moses'_presence"
    },
    response: {
      contributors: [
        "every_man_and_woman_whose_heart_moved_them",
        "all_who_were_willing"
      ],
      materialsBrought: [
        "gold_jewelry_brooches_earrings_rings_ornaments",
        "blue_purple_scarlet_yarn_fine_linen_goats_hair",
        "tanned_rams_skins_badgers_skins",
        "acacia_wood",
        "oil_for_light_spices_for_anointing_oil_and_incense",
        "onyx_stones_and_stones_for_ephod_and_breastpiece"
      ],
      additionalContributions: {
        skilledPersons: [
          "men_and_women_with_skill",
          "brought_work_of_spinning_yarn_and_linen",
          "brought_freewill_offerings_for_all_work_Yahweh_commanded"
        ],
        leaders: [
          "brought_onyx_stones_and_stones_for_setting",
          "spices_and_oil_for_light_anointing_and_incense"
        ]
      },
      summary: "Israelites_brought_freewill_offering_to_Yahweh"
    }
  });

  // === 35:30-35 - Appointment of Craftsmen ===
  moses.appointCraftsmen({
    audience: "Israelites",
    announcement: {
      yahwehSelection: {
        bezalel: {
          tribe: "Judah",
          sonOf: "Uri",
          grandsonOf: "Hur",
          endowment: [
            "called_by_name",
            "filled_with_Spirit_of_God",
            "wisdom_understanding_knowledge_in_all_crafts"
          ],
          skills: [
            "work_in_gold_silver_bronze",
            "cutting_stones_for_setting",
            "carving_wood",
            "all_manner_of_craftsmanship"
          ]
        },
        oholiab: {
          tribe: "Dan",
          sonOf: "Ahisamach",
          endowment: [
            "called_by_name",
            "given_ability_to_teach_others"
          ],
          skills: [
            "engraver_designer_embroiderer_in_yarn_and_linen",
            "weaver_for_all_work"
          ]
        }
      },
      generalCraftsmen: {
        endowment: "Yahweh_has_put_skill_in_hearts_of_all_who_are_gifted",
        task: "carry_out_all_Yahweh_has_commanded"
      }
    },
    assignment: {
      recipients: "Bezalel_Oholiab_and_every_gifted_craftsman",
      task: "make_all_Yahweh_has_commanded",
      authority: "Yahweh_has_given_ability"
    }
  });
}
```
