function exodus_4_mosesDivineSignsAndReturn() {
  // === 4:1-9 - Moses’ Third Objection and God’s Signs ===
  moses.raiseObjection({
    concern: [
      "what_if_they_do_not_believe_me",
      "or_listen_to_me",
      "say_Lord_did_not_appear_to_you"
    ],
    divineResponse: {
      sign1: {
        object: "staff_in_your_hand",
        instruction: "throw_it_on_ground",
        result: [
          "staff_became_serpent",
          "Moses_ran_from_it"
        ],
        reversal: [
          "Lord_said_take_it_by_tail",
          "Moses_took_it_and_it_became_staff_again"
        ],
        purpose: "that_they_may_believe_Lord_God_of_their_fathers_has_appeared_to_you"
      },
      sign2: {
        instruction: [
          "put_your_hand_inside_your_cloak",
          "take_it_out"
        ],
        result: "hand_was_leprous_like_snow",
        reversal: [
          "put_hand_back_inside_cloak",
          "take_it_out",
          "hand_was_restored_like_rest_of_flesh"
        ],
        purpose: "if_they_do_not_believe_first_sign_they_may_believe_latter_sign"
      },
      sign3: {
        condition: "if_they_do_not_believe_even_these_two_signs_or_listen_to_you",
        instruction: [
          "take_water_from_Nile",
          "pour_it_on_dry_ground"
        ],
        result: "water_will_become_blood_on_dry_ground",
        purpose: "final_sign_to_confirm_your_authority"
      }
    }
  });

  // === 4:10-17 - Moses’ Fourth Objection and Aaron’s Appointment ===
  moses.raiseObjection({
    concern: [
      "I_am_not_eloquent",
      "slow_of_speech_and_tongue"
    ],
    divineResponse: {
      reassurance: [
        "who_has_made_mans_mouth",
        "who_makes_him_mute_or_deaf_seeing_or_blind",
        "is_it_not_I_the_Lord"
      ],
      promise: [
        "go_and_I_will_be_with_your_mouth",
        "teach_you_what_you_shall_speak"
      ]
    }
  });

  moses.requestExemption({
    plea: "please_send_someone_else",
    divineReaction: [
      "anger_of_Lord_was_kindled_against_Moses",
      "solution: is_there_not_Aaron_your_brother_the_Levite"
    ],
    appointment: {
      aaron: {
        role: [
          "he_can_speak_well",
          "he_is_coming_to_meet_you",
          "he_will_be_glad_in_his_heart"
        ],
        function: [
          "you_shall_speak_to_him_and_put_words_in_his_mouth",
          "I_will_be_with_your_mouth_and_his_mouth",
          "teach_you_both_what_to_do"
        ],
        analogy: [
          "he_shall_be_your_mouth",
          "you_shall_be_to_him_as_God"
        ]
      },
      instruction: "take_staff_in_your_hand_with_which_you_shall_do_signs"
    }
  });

  // === 4:18-23 - Moses Returns to Egypt ===
  moses.prepareDeparture({
    context: {
      request: [
        "Moses_said_to_Jethro_his_father-in-law",
        "let_me_go_back_to_my_people_in_Egypt",
        "see_whether_they_are_still_alive"
      ],
      permission: "Jethro_said_go_in_peace"
    },
    divineInstruction: {
      assurance: "Lord_said_to_Moses_in_Midian_go_back_to_Egypt",
      reason: "all_men_who_sought_your_life_are_dead",
      prophecy: [
        "I_will_harden_Pharaohs_heart",
        "he_will_not_let_people_go"
      ],
      messageToPharaoh: [
        "say_to_Pharaoh_thus_says_Lord",
        "Israel_is_my_firstborn_son",
        "let_my_son_go_that_he_may_serve_me",
        "if_you_refuse_I_will_kill_your_firstborn_son"
      ]
    }
  });

  // === 4:24-26 - The Incident at the Lodging Place ===
  moses.divineEncounter({
    context: "at_lodging_place_on_the_way",
    incident: [
      "Lord_met_him_and_sought_to_put_him_to_death",
      "Zipporah_took_flint_and_cut_off_her_sons_foreskin",
      "touched_Moses_feet_with_it",
      "said_surely_you_are_bridegroom_of_blood_to_me"
    ],
    outcome: [
      "Lord_let_him_alone",
      "Zipporah_said_bridegroom_of_blood_because_of_circumcision"
    ]
  });

  // === 4:27-31 - Moses and Aaron Unite and Convince Israel ===
  aaron.meetMoses({
    divineInstruction: [
      "Lord_said_to_Aaron_go_into_wilderness_to_meet_Moses",
      "he_went_and_met_him_at_mountain_of_God",
      "kissed_him"
    ],
    communication: [
      "Moses_told_Aaron_all_words_of_Lord",
      "all_signs_He_commanded_him_to_do"
    ]
  });

  mosesAndAaron.addressIsrael({
    actions: [
      "gathered_all_elders_of_people_of_Israel",
      "Aaron_spoke_all_words_Lord_had_spoken_to_Moses",
      "performed_signs_in_sight_of_people"
    ],
    response: {
      people: [
        "believed",
        "heard_Lord_had_visited_people_of_Israel",
        "seen_their_affliction",
        "bowed_their_heads_and_worshiped"
      ]
    }
  });
}
