# Exodus 23
```
function exodus_23_laws() {
  // === 23:1-9 - Laws About Justice and Mercy ===
  God.declareLaws({
    context: {
      audience: "Israelites",
      deliveredBy: "Moses"
    },
    laws: {
      justice: [
        {
          command: [
            "shall_not_spread_false_report",
            "shall_not_join_hand_with_wicked_to_be_malicious_witness"
          ]
        },
        {
          command: [
            "shall_not_follow_crowd_to_do_evil",
            "shall_not_pervert_justice_by_siding_with_crowd_in_dispute"
          ]
        },
        {
          command: "shall_not_show_partiality_to_poor_man_in_his_lawsuit"
        },
        {
          case: "if_you_meet_your_enemy's_ox_or_donkey_going_astray",
          command: "bring_it_back_to_him"
        },
        {
          case: "if_you_see_donkey_of_one_who_hates_you_lying_under_its_burden",
          command: [
            "shall_not_leave_it",
            "help_him_to_lift_it_up"
          ]
        },
        {
          command: [
            "shall_not_pervert_justice_due_to_your_poor_in_his_lawsuit",
            "keep_far_from_false_charge",
            "do_not_kill_innocent_and_righteous",
            "for_I_will_not_acquit_the_wicked"
          ]
        },
        {
          command: [
            "shall_take_no_bribe",
            "for_bribe_blinds_clear-sighted_and_subverts_cause_of_righteous"
          ]
        },
        {
          command: [
            "shall_not_oppress_sojourner",
            "for_you_know_heart_of_sojourner_having_been_sojourners_in_Egypt"
          ]
        }
      ]
    }
  });

  // === 23:10-13 - Laws About Sabbath and Rest ===
  God.declareLaws({
    sabbath: [
      {
        case: "six_years_you_shall_sow_your_land_and_gather_its_yield",
        command: [
          "in_seventh_year_let_it_rest_and_lie_fallow",
          "so_poor_of_your_people_may_eat",
          "what_they_leave_beasts_of_field_may_eat"
        ],
        extension: "do_likewise_with_your_vineyard_and_olive_orchard"
      },
      {
        case: "six_days_you_shall_do_your_work",
        command: [
          "on_seventh_day_you_shall_rest",
          "so_your_ox_and_donkey_may_rest",
          "son_of_your_servant_and_sojourner_may_be_refreshed"
        ]
      },
      {
        command: [
          "be_careful_to_do_all_I_have_said_to_you",
          "make_no_mention_of_names_of_other_gods",
          "nor_let_it_be_heard_from_your_mouth"
        ]
      }
    ]
  });

  // === 23:14-19 - Laws About Festivals ===
  God.declareLaws({
    festivals: [
      {
        command: "three_times_a_year_you_shall_keep_feast_to_Me"
      },
      {
        feast: "Feast_of_Unleavened_Bread",
        command: [
          "keep_it_in_month_of_Abib",
          "for_in_it_you_came_out_of_Egypt",
          "none_shall_appear_before_Me_empty-handed"
        ]
      },
      {
        feast: "Feast_of_Harvest",
        command: "observe_with_firstfruits_of_your_labor_from_field"
      },
      {
        feast: "Feast_of_Ingathering",
        command: "observe_at_end_of_year_when_you_gather_in_your_labor_from_field"
      },
      {
        command: [
          "three_times_in_year_all_your_males_shall_appear_before_Lord_God",
          "shall_not_offer_blood_of_My_sacrifice_with_leavened_bread",
          "fat_of_My_feast_shall_not_remain_until_morning"
        ]
      },
      {
        command: [
          "best_of_firstfruits_of_your_ground_you_shall_bring_to_house_of_Lord_your_God",
          "shall_not_boil_kid_in_its_mother's_milk"
        ]
      }
    ]
  });

  // === 23:20-33 - Promises and Instructions for Conquest ===
  God.declarePromises({
    guidance: {
      agent: "angel",
      purpose: [
        "to_guard_you_on_the_way",
        "to_bring_you_to_place_I_have_prepared"
      ],
      command: [
        "pay_careful_attention_to_him",
        "obey_his_voice",
        "do_not_rebel_against_him",
        "for_he_will_not_pardon_your_transgression",
        "for_My_name_is_in_him"
      ],
      promise: {
        if: "you_obey_his_voice_and_do_all_I_say",
        then: [
          "I_will_be_enemy_to_your_enemies",
          "oppose_those_who_oppose_you"
        ]
      }
    },
    conquest: {
      action: [
        "My_angel_will_go_before_you",
        "bring_you_to_Amorites_Hittites_Perizzites_Canaanites_Hivites_Jebusites",
        "I_will_wipe_them_out"
      ],
      command: [
        "shall_not_bow_down_to_their_gods",
        "nor_serve_them",
        "nor_do_as_they_do",
        "overthrow_them_and_break_their_pillars"
      ],
      promise: [
        "you_shall_serve_Lord_your_God",
        "He_will_bless_your_bread_and_water",
        "I_will_take_sickness_away_from_among_you",
        "none_shall_miscarry_or_be_barren_in_your_land",
        "I_will_fill_number_of_your_days"
      ],
      strategy: [
        "I_will_send_fear_before_you",
        "cause_confusion_among_people_you_come_against",
        "make_all_your_enemies_turn_their_backs_to_you",
        "send_hornet_before_you_to_drive_out_Hivites_Canaanites_Hittites"
      ],
      method: [
        "will_not_drive_them_out_in_single_year",
        "lest_land_become_desolate_and_wild_beasts_multiply_against_you",
        "drive_them_out_little_by_little",
        "until_you_increase_and_possess_the_land"
      ],
      boundaries: [
        "set_your_border_from_Red_Sea_to_sea_of_Philistines",
        "from_wilderness_to_River",
        "give_inhabitants_of_land_into_your_hand",
        "drive_them_out_before_you"
      ],
      finalCommand: [
        "make_no_covenant_with_them_or_their_gods",
        "they_shall_not_dwell_in_your_land",
        "lest_they_make_you_sin_against_Me",
        "for_if_you_serve_their_gods_it_will_be_snare_to_you"
      ]
    }
  });
}
```
