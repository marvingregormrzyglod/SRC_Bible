# Amos 4

function amos_4_rebukeAndCall() {
  // === 4:1-3 - Rebuke of the Women of Samaria ===
  lord.address({
    audience: "cows_of_Bashan_on_mount_of_Samaria",
    critique: [
      "oppress_the_poor",
      "crush_the_needy",
      "say_to_your_husbands_bring_that_we_may_drink"
    ],
    judgment: {
      declaration: "Lord_God_has_sworn_by_His_holiness",
      consequence: [
        "days_are_coming_upon_you",
        "they_shall_take_you_away_with_hooks",
        "your_posterity_with_fishhooks",
        "you_shall_go_out_through_breaches_each_one_straight_ahead",
        "you_shall_be_cast_out_toward_Harmon"
      ]
    }
  });

  // === 4:4-5 - Satirical Call to False Worship ===
  lord.speak({
    audience: "Israel",
    tone: "sarcasm",
    instruction: [
      "come_to_Bethel_and_transgress",
      "at_Gilgal_multiply_transgression",
      "bring_your_sacrifices_every_morning",
      "your_tithes_every_three_days",
      "offer_sacrifice_of_thanksgiving_with_leaven",
      "proclaim_freewill_offerings_publish_them"
    ],
    reason: "for_so_you_love_to_do_O_people_of_Israel",
    authority: "declares_Lord_God"
  });

  // === 4:6-11 - Recounting of Unheeded Disciplines===
  lord.recallDisciplines({
    audience: "Israel",
    disciplines: [
      {
        action: "gave_you_cleanness_of_teeth_in_all_your_cities",
        detail: "lack_of_bread_in_all_your_places",
        response: "yet_you_did_not_return_to_Me"
      },
      {
        action: "withheld_rain_from_you_when_there_were_yet_three_months_to_harvest",
        detail: [
          "sent_rain_on_one_city_and_not_on_another",
          "one_field_would_have_rain_and_another_without_would_dry_up",
          "two_or_three_cities_would_stagger_to_another_to_drink_water_and_would_not_be_satisfied"
        ],
        response: "yet_you_did_not_return_to_Me"
      },
      {
        action: "struck_you_with_blight_and_mildew",
        detail: "locusts_devoured_your_fig_and_olive_trees",
        response: "yet_you_did_not_return_to_Me"
      },
      {
        action: "sent_among_you_pestilence_after_manner_of_Egypt",
        detail: [
          "killed_your_young_men_with_sword",
          "carried_away_your_horses",
          "made_stench_of_your_camp_go_up_into_your_nostrils"
        ],
        response: "yet_you_did_not_return_to_Me"
      },
      {
        action: "overthrew_some_of_you_as_God_overthrew_Sodom_and_Gomorrah",
        detail: "you_were_as_brand_plucked_out_of_burning",
        response: "yet_you_did_not_return_to_Me"
      }
    ],
    authority: "declares_Lord"
  });

  // === 4:12-13 - Call to Prepare for Judgment ===
  lord.proclaim({
    audience: "Israel",
    warning: {
      statement: "therefore_thus_I_will_do_to_you_O_Israel",
      preparation: "because_I_will_do_this_to_you_prepare_to_meet_your_God_O_Israel"
    },
    descriptionOfGod: {
      attributes: [
        "He_who_forms_mountains",
        "creates_the_wind",
        "declares_to_man_what_is_his_thought",
        "makes_dawn_into_darkness",
        "treads_on_high_places_of_earth"
      ],
      name: "Lord_God_of_hosts"
    }
  });
}
