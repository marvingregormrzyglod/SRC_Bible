  });

  amos pronounceJudgment({

      "beautiful_virgins_and_young_men_shall_faint_for_thirst",

      "those_who_say_as_way_of_Beersheba_lives"
    ],
    fate: [
      "they_shall_fall",
      "never_rise_again"
    ]
  });
}
