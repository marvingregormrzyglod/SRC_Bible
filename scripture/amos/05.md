# Amos 5
```
function amos_5_lamentAndCallToRepentance() {
  // === 5:1-3 - Lament Over Israel's Fall ===
  lord.deliverLament({
    audience: "house_of_Israel",
    message: {
      type: "funeral_dirge",
      content: [
        "fallen_no_more_to_rise_is_virgin_Israel",
        "forsaken_on_her_land_with_none_to_raise_her_up"
      ]
    },
    prophecy: {
      authority: "thus_says_Lord_God",
      outcome: [
        "city_that_went_out_thousand_shall_have_hundred",
        "that_went_out_hundred_shall_have_ten"
      ]
    }
  });

  // === 5:4-7 - Call to Seek God and Live ===
  lord.exhort({
    audience: "house_of_Israel",
    command: [
      "seek_Me_and_live",
      "do_not_seek_Bethel",
      "do_not_enter_Gilgal",
      "do_not_pass_to_Beersheba"
    ],
    reason: [
      "Gilgal_shall_surely_go_into_exile",
      "Bethel_shall_come_to_nothing"
    ],
    repeatedCall: {
      command: "seek_Lord_and_live",
      warning: "lest_He_break_out_like_fire_in_house_of_Joseph_and_devour_with_none_to_quench_it"
    },
    accusation: {
      target: "you_who_turn_justice_to_wormwood",
      action: "cast_down_righteousness_to_earth"
    }
  });

  // === 5:8-9 - Hymn to God's Power ===
  lord.describeDivinePower({
    attributes: [
      "He_who_made_Pleiades_and_Orion",
      "turns_deep_darkness_into_morning",
      "darkens_day_into_night",
      "calls_for_waters_of_sea_and_pours_them_out_on_surface_of_earth"
    ],
    name: "Lord_is_His_name",
    actions: [
      "flashes_forth_destruction_on_strong",
      "brings_ruin_on_fortified_city"
    ]
  });

  // === 5:10-13 - Rebuke for Injustice ===
  lord.rebuke({
    audience: "Israel",
    charges: [
      "hate_him_who_reproves_in_gate",
      "abhor_him_who_speaks_truth",
      "trample_on_poor",
      "exact_taxes_of_grain_from_him"
    ],
    consequences: [
      "you_have_built_houses_of_hewn_stone_but_shall_not_dwell_in_them",
      "planted_pleasant_vineyards_but_shall_not_drink_their_wine"
    ],
    reason: {
      knowledge: "I_know_how_many_are_your_transgressions",
      sins: [
        "afflict_righteous",
        "take_bribe",
        "turn_aside_needy_in_gate"
      ]
    },
    counsel: "therefore_prudent_will_keep_silent_in_such_a_time_for_it_is_an_evil_time"
  });

  // === 5:14-15 - Call to Righteousness ===
  lord.command({
    audience: "Israel",
    instruction: [
      "seek_good_and_not_evil_that_you_may_live",
      "hate_evil_and_love_good",
      "establish_justice_in_gate"
    ],
    promise: [
      "Lord_God_of_hosts_will_be_with_you_as_you_have_said",
      "perhaps_Lord_God_of_hosts_will_be_gracious_to_remnant_of_Joseph"
    ]
  });

  // === 5:16-17 - Announcement of Mourning ===
  lord.announce({
    authority: "thus_says_Lord_God_of_hosts",
    scene: {
      locations: ["in_all_squares", "in_all_streets"],
      actions: [
        "wailing_shall_be_in_all_squares",
        "they_shall_say_alas_alas",
        "call_farmers_to_mourning",
        "those_skilled_in_lamentation_to_wailing"
      ]
    },
    extent: "in_all_vineyards_there_shall_be_wailing",
    reason: "I_will_pass_through_your_midst",
    authority: "says_Lord"
  });

  // === 5:18-20 - Woe to Those Expecting the Day of the Lord ===
  lord.pronounceWoe({
    audience: "you_who_desire_day_of_Lord",
    question: "why_would_you_have_day_of_Lord",
    description: {
      nature: [
        "darkness_and_not_light",
        "gloom_with_no_brightness"
      ],
      metaphor: [
        "as_if_man_fled_from_lion_and_bear_met_him",
        "went_into_house_and_rested_hand_on_wall_and_serpent_bit_him"
      ]
    },
    reaffirmation: "day_of_Lord_is_darkness_and_not_light"
  });

  // === 5:21-24 - Rejection of Hypocritical Worship ===
  lord.rejectWorship({
    audience: "Israel",
    declaration: [
      "I_hate_I_despise_your_feasts",
      "I_take_no_delight_in_your_solemn_assemblies",
      "I_will_not_accept_your_burnt_offerings_and_grain_offerings",
      "I_will_not_look_upon_your_peace_offerings_of_fatted_animals",
      "take_away_from_Me_noise_of_your_songs",
      "I_will_not_listen_to_melody_of_your_harps"
    ],
    command: "let_justice_roll_down_like_waters_and_righteousness_like_ever_flowing_stream"
  });

  // === 5:25-27 - Historical Rebuke and Exile Prophecy ===
  lord.questionAndProphesy({
    audience: "house_of_Israel",
    question: [
      "did_you_bring_Me_sacrifices_and_offerings_during_forty_years_in_wilderness"
    ],
    accusation: [
      "you_shall_take_up_Sikkuth_your_king_and_Kiyyun_your_star_god",
      "images_you_made_for_yourselves"
    ],
    judgment: {
      action: "I_will_send_you_into_exile_beyond_Damascus",
      authority: "says_Lord_whose_name_is_God_of_hosts"
    }
  });
}
```
