# Amos 6
```
function amos_6_woeAndJudgment() {
  // === 6:1-7 - Woe to the Complacent in Zion and Samaria ===
  lord.pronounceWoe({
    audience: [
      "those_who_are_at_ease_in_Zion",
      "those_who_feel_secure_on_mountain_of_Samaria",
      "notable_men_of_first_of_nations_to_whom_house_of_Israel_comes"
    ],
    challenge: {
      command: [
        "pass_over_to_Calnah_and_see",
        "go_from_there_to_Hamath_the_great",
        "go_down_to_Gath_of_Philistines"
      ],
      question: [
        "are_they_better_than_these_kingdoms",
        "is_their_territory_greater_than_your_territory"
      ]
    },
    accusation: [
      "put_far_away_day_of_disaster",
      "bring_near_reign_of_violence",
      "lie_on_beds_of_ivory",
      "lounge_on_their_couches",
      "eat_lambs_from_flock_and_calves_from_stall",
      "sing_idle_songs_to_sound_of_harp",
      "like_David_invent_for_themselves_instruments_of_music",
      "drink_wine_in_bowls",
      "anoint_themselves_with_finest_oils",
      "are_not_grieved_over_ruin_of_Joseph"
    ],
    judgment: {
      declaration: "therefore_they_shall_now_be_first_of_those_who_go_into_exile",
      consequence: [
        "revelry_of_those_who_stretch_themselves_out_shall_pass_away"
      ],
      authority: "sworn_by_Himself_declares_Lord_God_of_hosts"
    }
  });

  // === 6:8-11 - Divine Oath and Destruction ===
  lord.declareOath({
    statement: [
      "I_abhor_pride_of_Jacob",
      "I_hate_his_strongholds"
    ],
    action: [
      "I_will_deliver_up_city_and_all_it_contains"
    ],
    scenario: {
      condition: "if_ten_men_remain_in_one_house_they_shall_die",
      detail: [
        "if_relative_who_burns_him_comes_to_carry_bones_out_of_house",
        "says_to_one_in_innermost_parts_is_anyone_else_with_you",
        "he_says_no",
        "then_he_says_hush_for_name_of_Lord_must_not_be_mentioned"
      ]
    },
    confirmation: {
      action: "Lord_commands_and_great_house_shall_be_struck_into_fragments",
      extent: "small_house_into_bits"
    }
  });

  // === 6:12-14 - Rebuke for Perversity and Prophecy of Oppression ===
  lord.questionAndProphesy({
    audience: "Israel",
    rhetoricalQuestions: [
      "do_horses_run_on_rocks",
      "does_one_plow_there_with_oxen"
    ],
    accusation: [
      "you_have_turned_justice_into_poison",
      "fruit_of_righteousness_into_wormwood",
      "rejoice_in_Lo-debar",
      "say_have_we_not_by_our_own_strength_captured_Karnaim"
    ],
    prophecy: {
      declaration: "behold_I_am_raising_up_against_you_a_nation_O_house_of_Israel",
      consequence: [
        "they_shall_oppress_you_from_Lebo-hamath_to_Brook_of_Arabah"
      ],
      authority: "declares_Lord_God_of_hosts"
    }
  });
}
```
