# The Gospel of John
## Chapter 1 - Jesus in the Beginning
```javascript
function describe_the_word() {
    word = {
        existed_since: "the_beginning",
        relationship: "with_God",
        essence: "divine"
    }

    creation = word.makes_everything()
    life = word.brings_life()
    light = word.shines_brightly()

    if (darkness.meets(light)) {
        darkness.can_overcome = false
    }

    world.reaction = {
        some_reject: true,
        some_accept: true
    }

    return word.takes_human_form()
}

function describe_johns_witness() {
    john = {
        role: "witness",
        task: "speak_of_the_light",
        clarification: "not_the_light"
    }

    if (people.ask("who_are_you")) {
        john.says("a_voice_in_the_wilderness")
    }

    return john.performs_baptism()
}

function describe_word_incarnate() {
    word.takes_human_form = {
        lives: "among_us",
        filled_with: "grace_and_truth"
    }

    grace = {
        given: "grace_upon_grace",
        comes_through: "Christ"
    }

    if (law.is_given()) {
        law.is_fulfilled = true
    }

    return truth.shows("Father_to_Son")
}

function describe_johns_witness() {
    john = {
        sent_by: "God",
        role: "witness",
        purpose: "point_to_the_light",
        identity: "not_the_light"
    }

    light = "true_light_for_all"
    world.receives = john.testifies_about(light)

    if (people.ask("who_are_you")) {
        john.says("not_the_Messiah")
        john.clarifies("voice_in_wilderness")
        john.points_to("one_coming_after")
    }

    return john.prepares_the_way()
}

function describe_the_words_arrival() {
    word = {
        enters: "world",
        made: "everything",
        belongs_to: "world"
    }

    world.reaction = {
        knows_him: false,
        accepts_him: false
    }

    if (people.accept(word)) {
        gift = word.gives("right_to_be_Gods_children")
        birth = "not_by_flesh_but_by_God"
    }

    word.takes_human_form = {
        lives: "among_us",
        glory: "full_of_grace_and_truth",
        seen_by: "us"
    }

    return word.reveals_God()
}

function describe_johns_testimony() {
    john.sees = "Lamb_of_God"
    john.declares = {
        message: "takes_away_worlds_sin",
        rank: "greater_than_me",
        existed: "before_me"
    }

    baptism = john.performs_with("water")
    spirit = john.witnesses("descend_like_dove")
    confirmation = john.receives("this_is_Gods_Son")

    return john.points_followers_to(word)
}

function describe_the_first_followers() {
    disciples = {
        first_two: "from_john",
        action: "follow_Jesus",
        question: "where_are_you_staying"
    }

    jesus.says("come_and_see")
    time_spent = disciples.stay_with(jesus)

    andrew = {
        finds: "brother_simon",
        declares: "we_found_the_Messiah"
    }

    jesus.meets(simon) {
        new_name = "Peter"
        meaning = "rock"
    }

    philip = jesus.calls("follow_me")
    nathanael = philip.tells("Jesus_from_Nazareth")

    if (nathanael.doubts("can_good_come_from_Nazareth")) {
        jesus.knows(nathanael) {
            character: "true_israelite",
            seen: "under_fig_tree"
        }
        nathanael.believes("Son_of_God")
    }

    jesus.promises = {
        vision: "heaven_open",
        sight: "angels_ascending_and_descending",
        on: "Son_of_Man"
    }

    return disciples.follow(jesus)
}
```
