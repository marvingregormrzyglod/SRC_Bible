# The Gospel of John
## Chapter 1 - Jesus in the Beginning
```javascript
function describe_the_words_beginning() {
    // Verses 1-2
    word = {
        existed_since: "the_beginning",
        relationship: "with_God",
        essence: "divine"
    }

    // Verse 3
    creation = word.makes_everything()
    nothing_made = "without_word"

    // Verses 4-5
    life = word.brings_life()
    light = life.shines_for("all_people")
    if (darkness.meets(light)) {
        darkness.can_overcome = false
    }

    return word.exists_as_light()
}

function describe_johns_witness() {
    // Verses 6-8
    john = {
        sent_by: "God",
        role: "witness",
        purpose: "point_to_the_light",
        identity: "not_the_light"
    }

    // Verses 9-10
    light = "true_light_for_all"
    word.enters = "world"
    world.made_by = "word"
    world.knows_word = false

    return john.prepares_for(word)
}

function describe_the_words_reception() {
    // Verses 11-13
    word.comes_to = "his_own"
    if (people.accept(word)) {
        gift = word.gives("right_to_be_Gods_children")
        birth = "not_by_flesh_but_by_God"
    } else {
        his_own.accept = false
    }

    // Verse 14
    word.takes_human_form = {
        lives: "among_us",
        glory: "full_of_grace_and_truth"
    }

    // Verses 15-18
    john.testifies = "one_greater_than_me"
    fullness = word.gives("grace_upon_grace")
    grace_and_truth = "come_through_Jesus_Christ"
    Son.reveals = "God_the_Father"

    return word.brings_grace()
}

function describe_johns_questioning() {
    // Verses 19-23
    if (priests_and_levites.ask("who_are_you")) {
        john.says("not_the_Messiah")
        john.clarifies("voice_in_wilderness")
        john.points_to("one_coming_after")
    }

    // Verses 24-28
    pharisees.question = "why_baptize"
    john.replies = {
        method: "with_water",
        hint: "one_among_you",
        humility: "not_worthy_to_untie_sandals"
    }
    location = "Bethany_beyond_Jordan"

    return john.prepares_the_way()
}

function describe_johns_testimony() {
    // Verses 29-34
    john.sees = "Jesus"
    john.proclaims = {
        title: "Lamb_of_God",
        purpose: "takes_away_worlds_sin",
        rank: "greater_and_before_me"
    }
    baptism = john.performs_with("water")
    spirit = john.witnesses("descend_like_dove")
    john.confirms = "this_is_Gods_Son"

    return john.points_followers_to(word)
}

function describe_the_first_followers() {
    // Verses 35-39
    disciples = {
        first_two: "from_john",
        action: "follow_Jesus",
        question: "where_are_you_staying"
    }
    jesus.says("come_and_see")
    time_spent = "with_Jesus"

    // Verses 40-42
    andrew = {
        finds: "brother_Simon",
        declares: "we_found_the_Messiah"
    }
    jesus.meets(simon) {
        new_name = "Peter",
        meaning = "rock"
    }

    // Verses 43-51
    philip = jesus.calls("follow_me")
    nathanael = philip.tells("Jesus_from_Nazareth")
    if (nathanael.doubts("can_good_come_from_Nazareth")) {
        jesus.knows(nathanael) {
            character: "true_israelite",
            seen: "under_fig_tree"
        }
        nathanael.believes("Son_of_God")
    }
    jesus.promises = {
        vision: "heaven_open",
        sight: "angels_ascending_and_descending",
        on: "Son_of_Man"
    }

    return disciples.follow(jesus)
}
```
