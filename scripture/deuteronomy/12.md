# Deuteronomy 12
```
function deuteronomy_12_worshipRegulations() {
  // === 12:1-4 - Destroy Pagan Worship Sites ===
  moses.instructIsrael({
    context: {
      audience: "Israel",
      setting: "preparing_to_possess_land"
    },
    command: {
      scope: "statutes_and_rules_to_observe_in_land_Lord_gives_you",
      duration: "all_days_you_live_on_earth",
      actions: [
        "destroy_all_places_where_nations_you_dispossess_worshiped_their_gods",
        "break_down_their_altars",
        "dash_in_pieces_their_pillars",
        "burn_their_Asherim_with_fire",
        "chop_down_engraved_images_of_their_gods",
        "destroy_their_names_from_that_place"
      ],
      prohibition: "do_not_do_so_to_Lord_your_God"
    }
  });

  // === 12:5-14 - Worship at the Chosen Place ===
  moses.regulateWorship({
    instruction: {
      location: {
        chosenPlace: "place_Lord_your_God_will_choose_from_all_your_tribes",
        purpose: "put_His_name_and_make_His_habitation",
        command: [
          "seek_and_go_there",
          "bring_burnt_offerings_sacrifices_tithes_contributions_votive_gifts_freewill_offerings_firstborn_of_herd_and_flock"
        ]
      },
      activities: [
        "eat_there_before_Lord_your_God",
        "rejoice_in_all_you_put_your_hand_to_you_and_your_households",
        "do_not_offer_burnt_offerings_at_any_place_you_see",
        "offer_them_only_at_place_Lord_chooses_in_one_of_your_tribes"
      ]
    },
    provision: [
      "eat_slaughter_within_your_towns_as_much_as_you_desire",
      "eat_as_if_it_were_gazelle_or_deer_both_clean_and_unclean_may_eat"
    ],
    prohibition: "do_not_eat_blood_pour_it_on_ground_like_water"
  });

  // === 12:15-19 - Instructions for Non-Sacrificial Slaughter ===
  moses.permitNonSacrificialSlaughter({
    allowance: {
      context: "within_all_your_towns",
      action: [
        "slaughter_and_eat_meat_according_to_blessing_of_Lord",
        "both_unclean_and_clean_may_eat",
        "as_if_gazelle_or_deer"
      ]
    },
    restriction: [
      "do_not_eat_blood",
      "pour_it_on_ground_like_water"
    ],
    additionalCommand: [
      "eat_tithes_of_grain_new_wine_oil_firstborn_of_herd_and_flock_only_at_place_Lord_chooses",
      "eat_them_before_Lord_to_learn_to_fear_Him_always"
    ],
    careForLevites: [
      "do_not_neglect_Levite_within_your_towns",
      "he_has_no_portion_or_inheritance_with_you"
    ]
  });

  // === 12:20-28 - Eating Meat in the Land ===
  moses.regulateMeatConsumption({
    context: "when_Lord_enlarges_your_territory_and_you_crave_meat",
    permission: [
      "slaughter_and_eat_meat_within_your_towns_as_Lord_has_blessed_you",
      "both_clean_and_unclean_may_eat_as_with_gazelle_or_deer"
    ],
    restriction: [
      "do_not_eat_blood_for_blood_is_life",
      "pour_it_on_ground_like_water",
      "do_not_eat_it_so_it_may_go_well_with_you_and_your_children"
    ],
    sacrificialRequirement: [
      "offerings_dedicated_to_Lord_must_be_brought_to_chosen_place",
      "offer_burnt_offerings_blood_and_flesh_on_altar_of_Lord",
      "blood_of_sacrifices_poured_on_altar_flesh_you_may_eat"
    ],
    exhortation: [
      "be_careful_to_obey_all_these_words",
      "do_what_is_right_and_good_in_sight_of_Lord"
    ]
  });

  // === 12:29-32 - Warning Against Idolatry ===
  moses.warnAgainstIdolatry({
    context: "when_Lord_cuts_off_nations_and_you_dispossess_them",
    warning: [
      "take_care_not_to_be_ensnared_to_follow_them",
      "do_not_inquire_about_their_gods_or_how_they_served_them"
    ],
    prohibition: [
      "do_not_do_so_to_Lord_your_God",
      "they_performed_every_abominable_act_Lord_hates",
      "they_burned_their_sons_and_daughters_in_fire_to_their_gods"
    ],
    finalCommand: [
      "be_careful_to_do_all_I_command_you",
      "do_not_add_to_it_or_take_from_it"
    ]
  });
}
```
