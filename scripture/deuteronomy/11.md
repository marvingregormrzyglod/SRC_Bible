# Deuteronomy 11
```
function deuteronomy_11_exhortAndWarn() {
  // === 11:1-7 - Call to Obedience Based on God’s Acts ===
  moses.exhortIsrael({
    context: {
      audience: "Israel",
      purpose: "motivate_obedience_through_past_acts"
    },
    command: [
      "love_Lord_your_God",
      "keep_His_charge_statutes_rules_and_commandments_always"
    ],
    reminder: {
      audience: "you_who_know_God’s_works_not_your_children",
      divineActs: [
        "discipline_you_saw",
        "great_work_against_Egypt_Pharaoh_and_his_house",
        "what_He_did_in_Red_Sea_to_Egypt’s_army",
        "what_He_did_to_Dathan_and_Abiram_swallowed_by_earth",
        "all_works_Lord_did_before_your_eyes_in_wilderness"
      ],
      purpose: "know_I_speak_not_to_your_children_but_to_you_who_saw"
    }
  });

  // === 11:8-15 - Blessings of Obedience ===
  moses.describeBlessings({
    condition: "keep_all_commandments_I_command_today",
    outcomes: [
      {
        strength: "be_strong_to_possess_land_you_are_crossing_to",
        longevity: "live_long_in_land_Lord_swore_to_your_fathers"
      },
      {
        landDescription: {
          quality: [
            "land_with_streams_fountains_springs_flowing_in_valleys_and_hills",
            "land_of_wheat_barley_vines_figs_pomegranates_olive_trees_honey",
            "land_where_you_eat_bread_without_scarcity",
            "land_with_stones_of_iron_and_hills_to_dig_copper"
          ],
          divineCare: [
            "land_Lord_your_God_cares_for",
            "eyes_of_Lord_always_on_it_from_beginning_to_end_of_year"
          ]
        }
      },
      {
        conditionalPromise: {
          if: [
            "diligently_obey_my_commandments",
            "love_Lord_your_God",
            "serve_Him_with_all_your_heart_and_soul"
          ],
          then: [
            "Lord_will_give_rain_in_its_season_early_and_latter",
            "you_will_gather_grain_new_wine_and_oil",
            "He_will_give_grass_in_fields_for_your_cattle",
            "you_will_eat_and_be_full"
          ]
        }
      }
    ]
  });

  // === 11:16-21 - Warning Against Idolatry ===
  moses.warnAgainstIdolatry({
    danger: [
      "take_care_lest_your_heart_be_deceived",
      "turn_aside_to_serve_and_worship_other_gods"
    ],
    consequence: [
      "Lord’s_anger_kindled_against_you",
      "He_shuts_up_heavens_no_rain",
      "land_yields_no_fruit",
      "you_perish_quickly_from_good_land"
    ],
    preventiveAction: [
      "lay_up_these_words_in_your_heart_and_soul",
      "bind_them_as_sign_on_your_hand_and_frontlets_between_your_eyes",
      "teach_them_to_your_children",
      "talk_of_them_when_sitting_in_house_walking_by_way_lying_down_rising",
      "write_them_on_doorposts_of_house_and_gates"
    ],
    promise: {
      condition: "obey_to_love_Lord_and_keep_His_commandments",
      result: [
        "your_days_and_days_of_your_children_multiplied",
        "like_days_of_heavens_above_earth"
      ]
    }
  });

  // === 11:22-25 - Promise of Victory ===
  moses.promiseVictory({
    condition: [
      "diligently_keep_all_commandments",
      "love_Lord_your_God",
      "walk_in_all_His_ways",
      "hold_fast_to_Him"
    ],
    divineAction: [
      "Lord_will_drive_out_all_nations_before_you",
      "you_will_dispossess_nations_greater_and_mightier_than_you"
    ],
    territorialPromise: [
      "every_place_sole_of_your_foot_treads_shall_be_yours",
      "from_wilderness_to_Lebanon_from_river_Euphrates_to_western_sea"
    ],
    assurance: "no_one_will_stand_against_you_Lord_will_put_dread_and_fear_of_you_on_all_land"
  });

  // === 11:26-32 - Blessing and Curse Set Before Israel ===
  moses.presentChoice({
    declaration: [
      "I_set_before_you_today_blessing_and_curse"
    ],
    options: [
      {
        blessing: {
          condition: "obey_commandments_of_Lord_your_God"
        }
      },
      {
        curse: {
          condition: [
            "do_not_obey_commandments",
            "turn_aside_from_way_to_follow_other_gods"
          ]
        }
      }
    ],
    instruction: {
      futureAction: [
        "when_Lord_brings_you_into_land",
        "set_blessing_on_Mount_Gerizim",
        "set_curse_on_Mount_Ebal"
      ],
      location: "beyond_Jordan_near_oaks_of_Moreh_opposite_Gilgal"
    },
    finalExhortation: [
      "you_are_crossing_Jordan_to_possess_land",
      "be_careful_to_do_all_statutes_and_rules_set_before_you"
    ]
  });
}
```
