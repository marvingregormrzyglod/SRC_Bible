# Deuteronomy 6
```
function deuteronomy_6_greatCommandmentAndTeaching() {
  // === 6:1-3 - Introduction to Commandments ===
  moses.proclaimCommandments({
    context: {
      audience: "Israel",
      setting: "land_beyond_Jordan",
      purpose: "to_do_them_in_the_land_you_are_entering_to_possess"
    },
    instruction: {
      source: "LORD_your_God_commanded_me_to_teach_you",
      scope: ["statutes", "rules"],
      outcome: [
        "you_and_your_son_and_your_sons_son_may_fear_LORD_your_God",
        "keep_all_His_statutes_and_commandments",
        "prolong_your_days"
      ],
      promise: {
        condition: "hear_and_be_careful_to_do",
        result: [
          "it_may_be_well_with_you",
          "you_may_multiply_greatly",
          "in_land_flowing_with_milk_and_honey"
        ],
        source: "as_LORD_God_of_your_fathers_has_promised_you"
      }
    }
  });

  // === 6:4-9 - The Shema and Teaching Diligently ===
  moses.declareShema({
    coreBelief: {
      declaration: "Hear_O_Israel_LORD_our_God_LORD_is_one",
      command: [
        "love_LORD_your_God_with_all_your_heart",
        "with_all_your_soul",
        "with_all_your_might"
      ]
    },
    transmission: {
      instruction: [
        "these_words_shall_be_on_your_heart",
        "teach_them_diligently_to_your_children",
        "talk_of_them_when_you_sit_in_your_house",
        "when_you_walk_by_the_way",
        "when_you_lie_down",
        "when_you_rise"
      ],
      reminders: [
        "bind_them_as_sign_on_your_hand",
        "they_shall_be_as_frontlets_between_your_eyes",
        "write_them_on_doorposts_of_your_house_and_on_your_gates"
      ]
    }
  });

  // === 6:10-15 - Warning Against Forgetting God ===
  moses.warnAgainstForgetting({
    context: {
      futureBlessing: [
        "LORD_your_God_brings_you_into_land_swore_to_Abraham_Isaac_Jacob",
        "great_and_good_cities_you_did_not_build",
        "houses_full_of_good_things_you_did_not_fill",
        "cisterns_you_did_not_dig",
        "vineyards_and_olive_trees_you_did_not_plant"
      ],
      condition: "when_you_eat_and_are_full"
    },
    warning: {
      instruction: [
        "take_care_lest_you_forget_LORD_who_brought_you_out_of_Egypt",
        "fear_LORD_your_God_and_serve_Him_only",
        "swear_by_His_name"
      ],
      prohibition: [
        "do_not_go_after_other_gods_of_peoples_around_you"
      ],
      consequence: [
        "anger_of_LORD_will_be_kindled_against_you",
        "He_will_destroy_you_from_face_of_land"
      ],
      reason: "LORD_your_God_is_jealous_God_among_you"
    }
  });

  // === 6:16-19 - Obedience and Conquest ===
  moses.exhortObedience({
    prohibition: {
      command: "do_not_put_LORD_your_God_to_test",
      example: "as_you_tested_Him_at_Massah"
    },
    instruction: [
      "diligently_keep_commandments_of_LORD_your_God",
      "His_testimonies_and_statutes",
      "do_what_is_right_and_good_in_sight_of_LORD"
    ],
    promise: [
      "it_may_be_well_with_you",
      "you_may_go_in_and_take_possession_of_good_land",
      "LORD_swore_to_your_fathers",
      "thrust_out_all_your_enemies_as_LORD_has_promised"
    ]
  });

  // === 6:20-25 - Teaching Future Generations ===
  moses.instructForFutureGenerations({
    scenario: {
      question: "when_your_son_asks_in_time_to_come",
      inquiry: "what_are_testimonies_statutes_and_rules_LORD_our_God_commanded"
    },
    response: {
      history: [
        "we_were_slaves_of_Pharaoh_in_Egypt",
        "LORD_brought_us_out_with_mighty_hand",
        "LORD_showed_signs_and_wonders_great_and_grievous_against_Egypt",
        "He_brought_us_out_to_give_us_land_swore_to_our_fathers"
      ],
      purpose: [
        "LORD_commanded_us_to_do_all_these_statutes",
        "to_fear_LORD_our_God_for_our_good_always",
        "to_keep_us_alive_as_at_this_day"
      ],
      outcome: [
        "it_will_be_righteousness_for_us",
        "if_we_are_careful_to_do_all_this_commandment_before_LORD_our_God"
      ]
    }
  });
}
```
