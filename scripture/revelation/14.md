      ]
    }
  });

  // === 14:8 - Second Angel’s Proclamation ===

    message: {

      volume: "loud_voice",

        condition: "if_anyone_worships_beast_and_its_image_or_receives_mark",

      endurance: "here_is_call_for_endurance_of_saints",

    }
  });

        "angel_thrust_his_sickle_into_earth",

        }
      }
    }
  });
