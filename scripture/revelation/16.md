# Revelation 16
```
function revelation_16_the_seven_bowls_of_wrath() {
  // === 16:1 - Command to Pour the Bowls ===
  temple.command({
    source: "loud_voice_from_sanctuary",
    recipients: "seven_angels",
    instruction: "Go_and_pour_out_seven_bowls_of_God’s_wrath_on_earth"
  });

  // === 16:2 - First Bowl: Harmful Sores ===
  angel.pourBowl({
    bowl: 1,
    target: "earth",
    result: {
      affliction: {
        description: "harmful_and_painful_sores",
        victims: [
          "people_who_bore_mark_of_beast",
          "those_who_worshiped_its_image"
        ]
      }
    }
  });

  // === 16:3 - Second Bowl: Sea to Blood ===
  angel.pourBowl({
    bowl: 2,
    target: "sea",
    result: {
      transformation: {
        effect: "became_like_blood_of_dead_man",
        consequence: "every_living_thing_in_sea_died"
      }
    }
  });

  // === 16:4-7 - Third Bowl: Rivers and Springs to Blood ===
  angel.pourBowl({
    bowl: 3,
    target: "rivers_and_springs_of_water",
    result: {
      transformation: {
        effect: "they_became_blood"
      }
    }
  });

  angel.ofWaters.declare({
    proclamation: [
      "Just_are_you_O_Holy_One_who_is_and_who_was",
      "for_you_brought_these_judgments",
      "they_shed_blood_of_saints_and_prophets",
      "you_gave_them_blood_to_drink",
      "they_deserve_it"
    ]
  });

  altar.respond({
    affirmation: [
      "Yes_Lord_God_Almighty",
      "true_and_just_are_your_judgments"
    ]
  });

  // === 16:8-9 - Fourth Bowl: Scorching Sun ===
  angel.pourBowl({
    bowl: 4,
    target: "sun",
    result: {
      effect: {
        permission: "given_power_to_scorch_people_with_fire",
        affliction: "people_seared_by_intense_heat"
      },
      response: {
        action: "cursed_name_of_God_who_had_power_over_plagues",
        outcome: "did_not_repent_or_give_Him_glory"
      }
    }
  });

  // === 16:10-11 - Fifth Bowl: Darkness on Beast’s Kingdom ===
  angel.pourBowl({
    bowl: 5,
    target: "throne_of_beast",
    result: {
      effect: {
        kingdom: "plunged_into_darkness",
        affliction: "people_gnawed_their_tongues_in_anguish"
      },
      response: {
        action: [
          "cursed_God_of_heaven_for_their_pain_and_sores",
          "did_not_repent_of_their_deeds"
        ]
      }
    }
  });

  // === 16:12-16 - Sixth Bowl: Euphrates and Armageddon ===
  angel.pourBowl({
    bowl: 6,
    target: "great_river_Euphrates",
    result: {
      effect: {
        river: "its_water_dried_up",
        purpose: "prepare_way_for_kings_from_east"
      },
      deception: {
        source: [
          "mouth_of_dragon",
          "mouth_of_beast",
          "mouth_of_false_prophet"
        ],
        manifestation: "three_unclean_spirits_like_frogs",
        role: "spirits_of_demons_performing_signs",
        mission: {
          action: "go_abroad_to_kings_of_whole_world",
          purpose: "assemble_them_for_battle_on_great_day_of_God_Almighty"
        }
      },
      interjection: {
        speaker: "Christ",
        warning: [
          "Behold_I_am_coming_like_thief",
          "Blessed_is_one_who_stays_awake_keeps_garments"
        ]
      },
      gathering: {
        location: {
          name: "Armageddon",
          language: "Hebrew"
        },
        action: "assembled_kings_there"
      }
    }
  });

  // === 16:17-21 - Seventh Bowl: Final Judgment ===
  angel.pourBowl({
    bowl: 7,
    target: "air",
    result: {
      proclamation: {
        source: "loud_voice_from_throne_in_sanctuary",
        content: "It_is_done"
      },
      phenomena: [
        {
          signs: [
            "flashes_of_lightning",
            "rumblings",
            "peals_of_thunder"
          ]
        },
        {
          earthquake: {
            description: "great_earthquake_unprecedented",
            impact: "such_as_had_never_been_since_man_on_earth"
          }
        }
      ],
      city: {
        target: "great_city",
        effect: "split_into_three_parts",
        otherCities: "cities_of_nations_fell"
      },
      judgment: {
        target: "Babylon_the_great",
        action: "God_remembered_her",
        punishment: "gave_her_cup_of_wine_of_fury_of_His_wrath"
      },
      geography: {
        effect: [
          "every_island_fled",
          "mountains_not_found"
        ]
      },
      hail: {
        description: "great_hailstones_about_100_pounds_each",
        action: "fell_from_heaven_on_people",
        response: {
          action: "people_cursed_God_for_plague_of_hail",
          reason: "plague_was_exceedingly_severe"
        }
      }
    }
  });
}
```
