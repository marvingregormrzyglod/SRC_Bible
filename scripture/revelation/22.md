function revelation_22_the_river_of_life_and_final_words() {
  // === 22:1-5 - The River and Tree of Life ===
  angel.show({
    scene: {
      river: {
        description: "river_of_water_of_life",
        appearance: "bright_as_crystal",
        source: "flowing_from_throne_of_God_and_Lamb",
        path: "through_middle_of_street_of_city"
      },
      tree: {
        identity: "tree_of_life",
        position: "on_either_side_of_river",
        yield: {
          fruit: "bearing_12_kinds_of_fruit",
          frequency: "yielding_fruit_each_month"
        },
        leaves: {
          purpose: "for_healing_of_nations"
        }
      }
    },
    conditions: {
      curse: "no_more_anything_accursed",
      throne: {
        presence: "throne_of_God_and_Lamb_will_be_in_it"
      },
      servants: {
        actions: [
          "worship_Him",
          "see_His_face"
        ],
        mark: "His_name_will_be_on_their_foreheads"
      },
      light: {
        night: "no_more_night",
        sources: {
          lamp: "no_need_of_light_of_lamp",
          sun: "no_need_of_light_of_sun",
          reason: "Lord_God_will_be_their_light"
        }
      },
      reign: {
        duration: "they_will_reign_forever_and_ever"
      }
    }
  });

  // === 22:6-11 - Affirmation of Truth and Urgency ===
  angel.affirm({
    message: {
      content: [
        "These_words_are_trustworthy_and_true",
        "Lord_God_of_spirits_of_prophets_has_sent_His_angel",
        "to_show_His_servants_what_must_soon_take_place"
      ]
    }
  });

  Christ.declare({
    statement: [
      "Behold_I_am_coming_soon",
      "My_reward_is_with_Me",
      "to_repay_each_one_for_what_he_has_done"
    ],
    identity: [
      "Alpha_and_Omega",
      "First_and_Last",
      "Beginning_and_End"
    ]
  });

  blessing.invoke({
    recipients: "those_who_wash_their_robes",
    privileges: [
      "right_to_tree_of_life",
      "enter_city_by_gates"
    ],
    exclusions: {
      outside: [
        "dogs",
        "sorcerers",
        "sexually_immoral",
        "murderers",
        "idolaters",
        "everyone_who_loves_and_practices_falsehood"
      ]
    }
  });

  Christ.testify({
    statement: [
      "I_Jesus_have_sent_my_angel_to_testify_to_you",
      "for_churches",
      "I_am_Root_and_Descendant_of_David",
      "Bright_Morning_Star"
    ]
  });

  // === 22:17-21 - Final Invitation and Warning ===
  invitation.extend({
    sources: [
      "Spirit",
      "Bride"
    ],
    call: "Come",
    recipients: [
      "one_who_hears",
      "one_who_is_thirsty"
    ],
    promises: [
      "let_him_come",
      "let_one_who_desires_take_water_of_life_without_price"
    ]
  });

  warning.issue({
    speaker: "He_who_testifies",
    content: [
      "I_warn_everyone_who_hears_words_of_prophecy_of_this_book",
      {
        condition: "if_anyone_adds_to_them",
        consequence: "God_will_add_to_him_plagues_described_in_book"
      },
      {
        condition: "if_anyone_takes_away_from_words_of_book",
        consequence: [
          "God_will_take_away_his_share_in_tree_of_life",
          "and_in_holy_city_described_in_book"
        ]
      }
    ]
  });

  Christ.conclude({
    affirmation: [
      "I_am_coming_soon"
    ]
  });

  john.respond({
    prayer: [
      "Amen",
      "Come_Lord_Jesus"
    ]
  });

  benediction.offer({
    content: [
      "Grace_of_Lord_Jesus_be_with_all",
      "Amen"
    ]
  });
}
