function revelation_5_the_scroll_and_the_lamb() {
  // === 5:1-4 - The Scroll and the Search for the Worthy ===
  throne.observe({
    rightHand: {
      holder: "One_seated_on_the_throne",
      item: {
        description: "scroll",
        features: {
          writing: "within_and_on_the_back",
          seals: "sealed_with_seven_seals"
        }
      }
    }
  });

  angel.proclaim({
    role: "mighty_angel",
    question: {
      volume: "loud_voice",
      content: "Who_is_worthy_to_open_scroll_and_break_its_seals"
    },
    response: {
      result: "no_one_found",
      scope: ["in_heaven", "on_earth", "under_earth"],
      capability: "able_to_open_scroll_or_look_into_it"
    }
  });

  john.react({
    emotion: "wept_loudly",
    reason: "no_one_found_worthy_to_open_scroll_or_look_into_it"
  });

  // === 5:5-7 - The Lion and the Lamb ===
  elder.intervene({
    messageToJohn: {
      comfort: "Do_not_weep",
      revelation: {
        identity: "Lion_of_tribe_of_Judah_Root_of_David",
        achievement: "has_conquered",
        capability: "able_to_open_scroll_and_its_seven_seals"
      }
    }
  });

  throne.center({
    figure: {
      identity: "Lamb",
      appearance: {
        condition: "as_though_slain",
        features: {
          horns: 7,
          eyes: 7,
          symbolism: "seven_spirits_of_God_sent_out_into_all_earth"
        }
      },
      position: "standing_between_throne_and_four_living_creatures_amid_elders"
    }
  });

  lamb.action({
    approach: "went_to_One_seated_on_throne",
    task: "took_scroll_from_right_hand"
  });

  // === 5:8-10 - Worship of the Lamb ===
  worship.initialResponse({
    trigger: "Lamb_took_the_scroll",
    participants: {
      fourLivingCreatures: {
        action: "fell_down_before_Lamb",
        items: {
          harps: true,
          bowls: {
            description: "golden_bowls_full_of_incense",
            symbolism: "prayers_of_saints"
          }
        }
      },
      twentyFourElders: {
        action: "fell_down_before_Lamb",
        items: {
          harps: true,
          bowls: {
            description: "golden_bowls_full_of_incense",
            symbolism: "prayers_of_saints"
          }
        }
      }
    },
    song: {
      title: "new_song",
      lyrics: [
        "Worthy_are_you_to_take_scroll_and_open_its_seals",
        "for_you_were_slain",
        "by_your_blood_ransomed_people_for_God",
        "from_every_tribe_language_people_nation",
        "made_them_kingdom_and_priests_to_our_God",
        "they_shall_reign_on_earth"
      ]
    }
  });

  // === 5:11-14 - Universal Worship ===
  worship.expanded({
    participants: {
      angels: {
        count: "myriads_of_myriads_thousands_of_thousands",
        position: "around_throne_living_creatures_elders"
      }
    },
    proclamation: {
      volume: "loud_voice",
      content: [
        "Worthy_is_the_Lamb_who_was_slain",
        "to_receive_power_wealth_wisdom_might_honor_glory_blessing"
      ]
    }
  });

  worship.universal({
    scope: {
      participants: [
        "every_creature_in_heaven",
        "on_earth",
        "under_earth",
        "in_sea",
        "all_that_is_in_them"
      ]
    },
    declaration: [
      "To_Him_who_sits_on_throne_and_to_Lamb",
      "be_blessing_honor_glory_might_forever_and_ever"
    ]
  });

  worship.conclude({
    livingCreatures: {
      affirmation: "Amen"
    },
    elders: {
      action: "fell_down_and_worshiped"
    }
  });
}
