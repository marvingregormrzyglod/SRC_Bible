function job_38_godQuestionsJob() {

        ]
      },
      {
        topic: "morning_and_dawn",
        queries: [
          "Have_you_commanded_the_morning_since_your_days_began",

          "earth_takes_shape_like_clay_under_a_seal",

          "and_their_raised_arm_is_broken"

      },
      {
        topic: "depths_of_sea_and_earth",

          "Or_have_you_seen_the_gates_of_deep_darkness",

      },
      {
        topic: "light_and_darkness",
        queries: [
          "Where_is_the_way_to_the_dwelling_of_light",

        ]
      },
      {
        topic: "snow_hail_and_rain",
        queries: [

      {
        topic: "lions",

        ]
      },
      {
        topic: "ravens",

          "Who_provides_for_the_raven_its_prey",
          "When_its_young_ones_cry_to_God",
          "Wandering_for_lack_of_food"
        ]
      }
    ]
  });
}
