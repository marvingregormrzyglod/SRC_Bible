      "from_my_mothers_womb_I_guided_widow",
      "my_heart_never_lifted_against_orphan"

  });

  });

    ],

  job.oathOfOpenness({

    ]
  });

          "weeds_instead_of_barley"
        ]
      }
    ],

  });
