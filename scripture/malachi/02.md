          "I_will_send_curse_upon_you",

        ],
        furtherJudgment: [
          "I_will_rebuke_your_offspring",
          "spread_dung_on_your_faces_dung_of_your_feasts",
          "you_shall_be_taken_away_with_it"
        ]
      }
    },
    reason: {
      failure: [
        "you_have_turned_aside_from_way",
        "caused_many_to_stumble_by_your_instruction",
        "corrupted_covenant_of_Levi"
      ],
      contrast: {
        ideal: {

            "true_instruction_in_his_mouth",
            "no_wrong_on_his_lips",
            "walked_with_me_in_peace_and_uprightness",
            "turned_many_from_iniquity"
          ],
          priestlyRole: [
            "lips_of_priest_should_guard_knowledge",
            "people_should_seek_instruction_from_his_mouth",
            "he_is_messenger_of_Lord_of_hosts"
          ]
        },
        reality: [
          "you_have_not_kept_my_ways",
          "show_partiality_in_instruction"
        ]
      }
    }
  });

  // === 2:10-16 - Rebuke to Judah for Covenant Unfaithfulness ===
  prophet.rebukeJudah({
    audience: "Judah",
    charge: {
      unity: "have_we_not_all_one_Father_did_not_one_God_create_us",
      violation: [
        "profaning_covenant_of_our_fathers_by_dealing_treacherously",
        "committed_abomination_in_Israel_and_Jerusalem",
        "profaned_sanctuary_of_Lord_by_marrying_daughter_of_foreign_god"
      ],
      consequence: {
        judgment: "Lord_will_cut_off_man_who_does_this",
        detail: "waker_and_answerer_from_tents_of_Jacob",
        condition: "even_though_he_brings_offering_to_Lord_of_hosts"
      }
    },
    furtherCharge: {
      divorce: {
        accusation: [
          "you_cover_Lords_altar_with_tears_weeping_and_groaning",
          "because_he_no_longer_regards_your_offering"
        ],
        question: "why_does_he_not",
        answer: [
          "Lord_was_witness_between_you_and_wife_of_your_youth",
          "you_have_been_faithless_to_her_though_she_is_your_companion_and_wife_by_covenant",
          "did_not_one_God_make_her_both_flesh_and_spirit_are_his",
          "he_seeks_godly_offspring"
        ],
        exhortation: [
          "take_care_not_to_be_faithless_to_wife_of_your_youth",
          "Lord_God_of_Israel_hates_divorce",
          "covering_ones_garment_with_violence"
        ]
      }
    }
  });

  // === 2:17 - Rebuke for Questioning God's Justice ===
  prophet.rebukePeople({
    accusation: [
      "you_have_wearied_Lord_with_your_words",
      "you_say_how_have_we_wearied_him",
      "by_saying_everyone_who_does_evil_is_good_in_sight_of_Lord",
      "by_saying_he_delights_in_them",
      "by_asking_where_is_God_of_justice"
    ]
  });
}
