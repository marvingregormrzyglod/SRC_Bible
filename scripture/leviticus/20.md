          "defiling_my_sanctuary",

      },
      {

        "consecrate_yourselves_and_be_holy_for_I_am_the_Lord_your_God",

      reason: "I_am_the_Lord_who_sanctifies_you"
    }
  });

  yahweh.command({
    holiness: [
      {
        instruction: "consecrate_yourselves_and_be_holy",
        reason: "I_am_the_Lord_your_God"
      },
      {
        instruction: "keep_my_statutes_and_do_them",
        reason: "I_am_the_Lord_who_sanctifies_you"
      }
    ]
  });

    familyOffenses: [
      {
        offense: "cursing_father_or_mother",

          "shall_be_put_to_death",
          "his_blood_is_upon_him"
        ]
      }
    ],
    sexualSins: [
      {
        offense: "adultery_with_another_mans_wife",
        penalty: [
          "both_adulterer_and_adulteress_shall_be_put_to_death"
        ]
      },
      {
        offense: "lying_with_fathers_wife",
        penalty: [
          "both_shall_be_put_to_death",
          "their_blood_is_upon_them"
        ]
      },
      {
        offense: "lying_with_daughter_in_law",
        penalty: [
          "both_shall_be_put_to_death",
          "they_have_committed_perversion",
          "their_blood_is_upon_them"
        ]
      },
      {
        offense: "lying_with_a_male_as_with_a_woman",
        description: "they_have_committed_an_abomination",
        penalty: [
          "both_shall_be_put_to_death",
          "their_blood_is_upon_them"
        ]
      },
      {
        offense: "taking_a_wife_and_her_mother_also",
        penalty: [
          "they_shall_be_burned_with_fire",
          "that_no_depravity_may_be_among_you"
        ]
      },
      {
        offense: "lying_with_an_animal",
        penalty: [
          "shall_be_put_to_death",
          "animal_shall_be_killed"
        ]
      },
      {
        offense: "woman_approaching_an_animal_to_lie_with_it",
        penalty: [
          "woman_shall_be_put_to_death",
          "animal_shall_be_killed",
          "their_blood_is_upon_them"
        ]
      },
      {
        offense: "lying_with_sister_or_half_sister",
        description: "they_have_seen_each_others_nakedness",

          "they_shall_be_cut_off_in_sight_of_their_people",
          "he_has_uncovered_his_sisters_nakedness",
          "he_shall_bear_his_iniquity"
        ]
      },
      {
        offense: "lying_with_woman_during_her_menstrual_impurity",
        penalty: [
          "both_shall_be_cut_off_from_among_their_people"

    },

      promise: "I_have_given_you_their_land",

      reason: [
        "I_am_the_Lord_who_separated_you_from_the_peoples",
        "you_shall_be_holy_to_me_for_I_the_Lord_am_holy"
      ]
    }
  });

  // === 20:27 - Penalty for Occult Practices ===
  yahweh.decree({
    occult: {
      offense: "being_a_medium_or_necromancer",

      penalty: [
        "shall_be_put_to_death",
        "they_shall_be_stoned_with_stones",
        "their_blood_is_upon_them"
      ]
    }
  });
}
