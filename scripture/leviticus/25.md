    },
    rules: [
      "every_man_return_to_his_property",

      "no_sowing_reaping_or_gathering_grapes"
    ],
    landTransactions: {

          "fewer_years_reduce_price"
        ]
      },
      ethics: [
        "do_not_wrong_one_another",
        "fear_your_God"
      ]
    }

        usage: [
          "sow_in_eighth_year",
          "eat_old_produce_until_ninth_year_harvest"
        ]
      }
    }
  });

    },

        case: "man_becomes_poor_and_sells_property",

          "nearest_kinsman_shall_redeem_what_brother_sold",
          "if_no_redeemer_and_he_prospers_he_may_redeem",

          "return_remainder_to_buyer_and_reclaim_property"
        ]
      },
      {
        case: "no_means_to_redeem",
        outcome: [
          "property_remains_with_buyer_until_Jubilee",
          "released_in_Jubilee_and_returns_to_original_owner"
        ]
      },
      {
        case: "houses_in_walled_cities",
        redemptionPeriod: "within_one_year",

    ],
    authority: "I_am_the_Lord_your_God_who_brought_you_out_of_Egypt"

        "buy_slaves_from_sojourners_among_you_and_their_families",
        "hold_as_property_and_bequeath_to_children"
      ],
      distinction: "not_to_rule_ruthlessly_over_Israelites_but_may_over_slaves"
    },
    redemptionOfServants: {
      scenario: "Israelite_sold_to_sojourner_or_resident",
      rules: [
        "he_may_be_redeemed",
        "kinsman_may_redeem_him",
        "calculate_price_based_on_years_to_Jubilee",
        "treat_as_hired_worker_not_slave"
      ],
      outcome: [
        "if_not_redeemed_serves_until_Jubilee",
        "then_he_and_his_children_go_free"
      ]
    },
    conclusion: {
      ownership: "Israelites_are_My_servants_not_to_be_enslaved",
      authority: "I_am_the_Lord_your_God"
    }
  });
}
