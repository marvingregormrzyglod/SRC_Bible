# Leviticus 6
```
function leviticus_6_lawsOnOfferingsAndRestitution() {
  // === 6:1-7 - Laws for Sin Offering for Deceit ===
  yahweh.instructMoses({
    context: {
      audience: "Aaron_and_his_sons",
      purpose: "laws_for_sinning_against_the_Lord"
    },
    sinOffering: {
      conditions: [
        "deceives_neighbor_in_matter_of_deposit_or_security",
        "robs_neighbor",
        "oppresses_neighbor",
        "finds_lost_property_and_lies_about_it",
        "swears_falsely_in_any_matter"
      ],
      consequences: [
        "sins_and_becomes_guilty"
      ],
      restitution: {
        requirements: [
          "restore_in_full_what_was_taken",
          "add_one_fifth_to_it",
          "give_it_to_owner_on_day_of_guilt_offering"
        ]
      },
      offering: {
        type: "ram_without_defect_from_flock",
        valuation: "according_to_sanctuary_shekel",
        purpose: "atonement_before_the_Lord",
        result: "forgiveness_for_sins_committed"
      }
    }
  });

  // === 6:8-13 - Laws for Burnt Offering ===
  yahweh.instructAaronAndSons({
    context: {
      offering: "burnt_offering",
      location: "altar"
    },
    procedure: {
      fire: [
        "keep_burning_on_altar_all_night_until_morning",
        "fire_must_not_go_out"
      ],
      priestDuties: [
        "put_on_linen_garments_and_linen_undergarments",
        "remove_ashes_from_burnt_offering",
        "place_ashes_beside_altar",
        "change_garments_to_take_ashes_outside_camp_to_clean_place"
      ],
      maintenance: [
        "fire_shall_be_kept_burning_continually",
        "priest_shall_burn_wood_on_it_every_morning",
        "arrange_burnt_offering_on_it",
        "burn_fat_of_peace_offerings_on_it"
      ],
      perpetualStatute: "fire_must_not_go_out"
    }
  });

  // === 6:14-18 - Laws for Grain Offering ===
  yahweh.specifyGrainOffering({
    context: {
      offering: "grain_offering",
      performers: "sons_of_Aaron"
    },
    procedure: {
      presentation: [
        "offer_before_the_Lord_in_front_of_altar"
      ],
      portion: {
        priest: [
          "take_handful_of_fine_flour_and_oil_with_all_frankincense",
          "burn_as_memorial_portion_on_altar",
          "pleasing_aroma_to_the_Lord"
        ],
        remainder: [
          "for_Aaron_and_his_sons",
          "eaten_without_leaven_in_holy_place",
          "eaten_in_court_of_tent_of_meeting"
        ]
      },
      holiness: [
        "most_holy_of_Lord’s_food_offerings",
        "only_males_among_Aaron’s_descendants_may_eat",
        "perpetual_statute_for_generations"
      ],
      warning: [
        "whatever_touches_it_shall_become_holy"
      ]
    }
  });

  // === 6:19-23 - Grain Offering for Priest’s Anointing ===
  yahweh.instructPriestlyOffering({
    context: {
      occasion: "anointing_of_priest",
      performer: "priest_anointed_in_Aaron’s_place"
    },
    offering: {
      type: "grain_offering",
      quantity: "one_tenth_of_ephah_of_fine_flour",
      frequency: "regularly_half_in_morning_half_in_evening",
      preparation: [
        "prepared_with_oil_on_griddle",
        "well_mixed_and_broken_in_pieces",
        "offered_as_pleasing_aroma_to_the_Lord"
      ],
      performer: "priest_from_among_Aaron’s_sons_who_is_anointed",
      requirement: [
        "wholly_burned",
        "not_to_be_eaten"
      ],
      statute: "perpetual_offering_for_generations"
    }
  });

  // === 6:24-30 - Laws for Sin Offering ===
  yahweh.specifySinOffering({
    context: {
      offering: "sin_offering",
      location: "before_the_Lord"
    },
    procedure: {
      slaughter: [
        "kill_in_place_where_burnt_offering_is_killed",
        "most_holy"
      ],
      priestDuties: [
        "priest_who_offers_it_shall_eat_it",
        "eaten_in_holy_place_in_court_of_tent_of_meeting"
      ],
      holiness: [
        "whatever_touches_its_flesh_shall_be_holy",
        "if_blood_splatters_on_garment_wash_it_in_holy_place",
        "earthenware_vessel_used_must_be_broken",
        "bronze_vessel_must_be_scoured_and_rinsed"
      ],
      eligibility: [
        "all_males_among_priests_may_eat_it",
        "most_holy"
      ],
      exception: [
        "if_blood_brought_into_tent_of_meeting_for_atonement",
        "offering_shall_be_burned_with_fire",
        "not_to_be_eaten"
      ]
    }
  });
}
```
