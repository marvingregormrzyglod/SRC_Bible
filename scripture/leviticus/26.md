function leviticus_26_blessingsAndCurses() {

  lord.commandObedience({
    context: {
      audience: "Israelites",
      speaker: "Moses"
    },
    prohibitions: [
      "make_no_idols_or_carved_images",

    ],
    affirmations: [
      "keep_My_Sabbaths",
      "reverence_My_sanctuary"
    ],
    authority: "I_am_the_Lord"
  });

  // === 26:3-13 - Blessings for Obedience ===
  lord.promiseBlessings({
    condition: "walk_in_My_statutes_and_keep_My_commandments",
    blessings: [
      {
        category: "agricultural",
        details: [
          "rain_in_its_season",
          "land_yield_its_produce",
          "trees_of_field_bear_fruit",
          "threshing_reach_vintage",
          "vintage_reach_sowing_time",
          "eat_bread_to_the_full"
        ]
      },
      {
        category: "security",
        details: [
          "dwell_in_your_land_securely",
          "peace_in_the_land",
          "lie_down_with_none_making_you_afraid",
          "remove_harmful_beasts_from_land",
          "no_sword_passing_through_land"
        ]
      },
      {
        category: "military",
        details: [
          "chase_your_enemies_and_they_fall_before_you_by_sword",
          "five_of_you_chase_a_hundred",
          "hundred_of_you_put_ten_thousand_to_flight"
        ]
      },
      {
        category: "divine_favor",
        details: [
          "turn_to_you_and_make_you_fruitful",
          "multiply_you_and_confirm_My_covenant_with_you",
          "old_stores_cleared_for_new",
          "set_My_dwelling_among_you",
          "not_abhor_you",
          "walk_among_you_and_be_your_God",
          "you_shall_be_My_people"
        ]
      }
    ],
    authority: "I_am_the_Lord_your_God_who_brought_you_out_of_Egypt"
  });

  // === 26:14-39 - Curses for Disobedience ===
  lord.warnCurses({
    condition: "do_not_listen_to_Me_or_do_all_these_commandments",
    curses: [
      {
        stage: 1,
        consequences: [
          "appoint_over_you_sudden_terror",
          "wasting_diseases_and_fever",
          "eyes_fail_and_soul_languish",
          "sow_seed_in_vain_enemies_eat_it",
          "set_My_face_against_you",
          "struck_down_before_enemies",
          "ruled_by_those_who_hate_you",
          "flee_when_none_pursues"
        ]
      },
      {
        stage: 2,
        condition: "still_do_not_listen_to_Me",
        consequences: [
          "discipline_you_sevenfold_for_your_sins",
          "break_pride_of_your_power",
          "heavens_like_iron_and_earth_like_bronze",
          "strength_spent_in_vain",
          "land_not_yield_produce",
          "trees_not_bear_fruit"
        ]
      },
      {
        stage: 3,
        condition: "walk_contrary_to_Me",
        consequences: [
          "punish_you_sevenfold_for_your_sins",
          "send_wild_beasts_to_bereave_you_of_children",
          "destroy_your_cattle",
          "reduce_your_number",
          "highways_desolate"
        ]
      },
      {
        stage: 4,
        condition: "still_walk_contrary_to_Me",

          "act_with_hostility_against_you",
          "strike_you_sevenfold_for_your_sins",
          "bring_sword_upon_you_to_execute_vengeance",
          "send_pestilence_among_you",
          "deliver_you_into_enemy_hands",
          "bread_supply_cut_off",
          "eat_and_not_be_satisfied"
        ]
      },
      {
        stage: 5,
        condition: "still_do_not_obey_Me",
        consequences: [
          "punish_you_sevenfold_for_your_sins",
          "eat_flesh_of_your_sons_and_daughters",
          "destroy_high_places_and_altars",
          "cast_your_carcasses_on_idol_ruins",
          "abhor_you",
          "lay_your_cities_waste",
          "sanctuaries_desolate",
          "not_smell_your_pleasing_aromas",
          "land_desolate_and_enemies_dwell_in_it"
        ]
      },
      {
        stage: 6,
        consequences: [
          "scatter_you_among_nations",
          "unsheathe_sword_after_you",
          "land_become_desolate",
          "cities_become_waste",
          "land_enjoy_its_Sabbaths_while_desolate",
          "pay_for_your_iniquities"
        ]
      },
      {
        stage: 7,
        consequences: [
          "languish_in_enemies_lands",
          "hearts_of_survivors_melt",
          "sound_of_driven_leaf_causes_fear",

          "land_of_enemies_consume_you",
          "survivors_waste_away_in_iniquity"
        ]
      }
    ]
  });

  // === 26:40-45 - Promise of Restoration ===
  lord.promiseRestoration({
    condition: [
      "confess_their_iniquity_and_iniquity_of_fathers",
      "acknowledge_treachery_against_Me",
      "admit_walking_contrary_to_Me",
      "heart_is_humbled_and_they_make_amends"
    ],
    restoration: [
      "remember_My_covenant_with_Jacob_Isaac_and_Abraham",
      "remember_the_land",
      "not_reject_them_or_abhor_them_to_destroy_utterly",
      "not_break_My_covenant_with_them"
    ],
    authority: "I_am_the_Lord_their_God",
    purpose: "for_their_sake_remember_covenant_with_ancestors_brought_out_of_Egypt"
  });

  // === 26:46 - Conclusion ===
  lord.conclude({
    summary: [
      "these_are_statutes_ordinances_and_laws",
      "given_by_Lord_between_Himself_and_Israelites"
    ],
    location: "Mount_Sinai",
    mediator: "Moses"
  });
}
