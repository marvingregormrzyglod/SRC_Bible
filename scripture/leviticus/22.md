# Leviticus 22
```
function leviticus_22_sacredOfferings() {
  // === 22:1-9 - Priestly Conduct with Holy Things ===
  yahweh.speakToMoses({
    audience: "Aaron_and_his_sons",
    instructions: [
      {
        subject: "handling_holy_things",
        command: "deal_carefully_with_holy_things_of_people_of_Israel_consecrated_to_me",
        prohibition: "do_not_profane_my_holy_name",
        reason: "I_am_the_Lord"
      },
      {
        subject: "uncleanness",
        prohibition: "no_descendant_of_Aaron_who_is_unclean_may_approach_to_offer_holy_things",
        conditionsOfUncleanness: [
          "leprosy",
          "discharge",
          "contact_with_dead_body",
          "emission_of_semen",
          "contact_with_unclean_creature",
          "contact_with_any_unclean_person_or_thing"
        ],
        procedure: [
          "if_unclean_he_shall_not_eat_of_holy_things_until_clean",
          "bathe_his_body_in_water_and_wait_until_evening_to_become_clean"
        ],
        penalty: [
          "if_he_eats_holy_things_while_unclean",
          "he_shall_be_cut_off_from_his_people"
        ],
        reason: "I_am_the_Lord"
      },
      {
        subject: "compliance",
        command: "sons_of_Aaron_shall_keep_my_charge",
        consequence: "lest_they_bear_sin_and_die_for_profaning_holy_things",
        affirmation: "I_am_the_Lord_who_sanctifies_them"
      }
    ]
  });

  // === 22:10-16 - Who May Eat Holy Things ===
  yahweh.decree({
    eatingHolyThings: [
      {
        prohibition: "no_outsider_shall_eat_of_holy_things",
        exceptions: [
          "priests_household",
          "slave_born_in_his_house",
          "slave_bought_with_his_money"
        ]
      },
      {
        condition: "if_priests_daughter_marries_an_outsider",
        prohibition: "she_shall_not_eat_of_holy_things",
        exception: [
          "if_widowed_or_divorced_with_no_child_and_returns_to_fathers_house",
          "she_may_eat_of_her_fathers_food"
        ]
      },
      {
        offense: "if_anyone_eats_holy_thing_unintentionally",
        penalty: [
          "he_shall_restore_it_to_priest_with_one_fifth_added",
          "priests_shall_not_profane_holy_things_of_people_of_Israel"
        ]
      },
      {
        warning: "priests_shall_not_let_people_eat_holy_things_and_bring_guilt_requiring_restitution",
        reason: "I_am_the_Lord_who_sanctifies_them"
      }
    ]
  });

  // === 22:17-25 - Acceptable Offerings ===
  yahweh.speakToMoses({
    audience: "Aaron_his_sons_and_all_people_of_Israel",
    regulations: [
      {
        subject: "offerings",
        types: [
          "burnt_offering",
          "vow",
          "freewill_offering"
        ],
        requirements: [
          "male_without_ble mish",
          "from_cattle_sheep_or_goats",
          "acceptable_to_be_accepted_on_your_behalf"
        ],
        prohibitions: [
          "do_not_offer_anything_with_defect_for_it_will_not_be_accepted",
          "do_not_offer_blind_lame_injured_maimed_or_with_discharge_sore_or_scabs"
        ]
      },
      {
        subject: "peace_offering",
        requirements: [
          "must_be_without_blemish_to_be_accepted",
          "no_blind_broken_maimed_with_sore_scabs_or_crushed_testicles"
        ],
        prohibition: "do_not_offer_to_Lord_or_give_them_on_altar_as_food_offering"
      },
      {
        subject: "animals_from_foreigners",
        prohibition: "do_not_accept_such_animals_with_defects_for_offering",
        reason: "their_deformity_makes_them_unacceptable"
      }
    ]
  });

  // === 22:26-30 - Rules for Young Animals ===
  yahweh.decree({
    youngAnimals: [
      {
        subject: "newborn_ox_sheep_or_goat",
        rule: [
          "shall_remain_with_its_mother_for_seven_days",
          "from_eighth_day_onward_it_may_be_accepted_as_offering"
        ]
      },
      {
        prohibition: "do_not_slaughter_ox_or_sheep_and_its_young_on_same_day"
      },
      {
        subject: "thank_offering",
        rule: [
          "offer_it_so_it_may_be_accepted",
          "eat_it_on_same_day_leave_none_until_morning"
        ],
        reason: "I_am_the_Lord"
      }
    ]
  });

  // === 22:31-33 - Conclusion and Exhortation ===
  yahweh.command({
    obedience: {
      instruction: "keep_my_commandments_and_do_them",
      reason: "I_am_the_Lord"
    },
    sanctity: {
      prohibition: "do_not_profane_my_holy_name",
      command: "I_must_be_sanctified_among_people_of_Israel",
      reason: [
        "I_am_the_Lord_who_sanctifies_you",
        "I_brought_you_out_of_Egypt_to_be_your_God"
      ],
      affirmation: "I_am_the_Lord"
    }
  });
}
```
