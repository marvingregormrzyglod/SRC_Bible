function leviticus_19_holinessAndJustice() {
  // === 19:1-2 - Call to Holiness ===
  yahweh.speakToMoses({
    audience: "all_congregation_of_people_of_Israel",
    command: {
      holiness: "be_holy_for_I_the_Lord_your_God_am_holy"
    }
  });

  // === 19:3-8 - Reverence and Obedience ===
  yahweh.decree({
    reverence: [
      {
        subject: "each_of_you",
        action: "revere_mother_and_father",
        reason: "I_am_the_Lord_your_God"
      },
      {
        subject: "each_of_you",
        action: "keep_my_Sabbaths",
        reason: "I_am_the_Lord_your_God"
      }
    ],
    idolatry: {
      prohibition: "do_not_turn_to_idols_or_make_molten_gods",
      reason: "I_am_the_Lord_your_God"
    },
    peaceOffering: {
      procedure: [
        "offer_it_so_it_may_be_accepted_on_your_behalf",
        "eat_it_on_day_of_offering_or_next_day",
        "anything_left_until_third_day_shall_be_burned_with_fire"
      ],
      violation: {
        condition: "if_eaten_on_third_day",
        consequence: [
          "it_is_unacceptable",
          "it_will_not_be_accepted",
          "eater_bears_his_iniquity_for_profaning_what_is_holy_to_the_Lord",
          "that_person_shall_be_cut_off_from_his_people"
        ]
      }
    }
  });

  // === 19:9-10 - Provision for the Poor ===
  yahweh.command({
    harvest: {
      fields: [
        "do_not_reap_to_very_edges_of_your_field",
        "do_not_gather_gleanings_of_your_harvest"
      ],
      vineyards: [
        "do_not_strip_your_vineyard_bare",
        "do_not_gather_fallen_grapes_of_your_vineyard"
      ],
      purpose: "leave_them_for_the_poor_and_sojourner",
      reason: "I_am_the_Lord_your_God"
    }
  });

  // === 19:11-18 - Ethical Conduct ===
  yahweh.instruct({
    integrity: [
      {
        prohibition: [
          "do_not_steal",
          "do_not_deal_falsely",
          "do_not_lie_to_one_another"
        ]
      },
      {
        prohibition: "do_not_swear_by_my_name_falsely",
        consequence: "profane_the_name_of_your_God",
        reason: "I_am_the_Lord"
      },
      {
        prohibition: [
          "do_not_oppress_or_rob_your_neighbor",
          "do_not_withhold_wages_of_laborer_until_morning"
        ]
      },
      {
        prohibition: "do_not_curse_the_deaf_or_put_stumbling_block_before_the_blind",
        command: "fear_your_God",
        reason: "I_am_the_Lord"
      }
    ],
    justice: [
      {
        prohibition: "do_not_render_unjust_judgments",
        instruction: [
          "do_not_show_partiality_to_poor_or_deference_to_great",
          "judge_your_neighbor_with_justice"
        ]
      },
      {
        prohibition: [
          "do_not_go_around_as_slanderer_among_your_people",
          "do_not_stand_against_life_of_your_neighbor"
        ],
        reason: "I_am_the_Lord"
      },
      {
        prohibition: [
          "do_not_hate_your_brother_in_your_heart",
          "do_not_take_vengeance_or_bear_grudge_against_sons_of_your_people"
        ],
        instruction: [
          "reason_frankly_with_your_neighbor_lest_you_incur_sin",
          "love_your_neighbor_as_yourself"
        ],
        reason: "I_am_the_Lord"
      }
    ]
  });

  // === 19:19-29 - Miscellaneous Laws ===
  yahweh.establish({
    purity: [
      {
        subject: "livestock",
        prohibition: "do_not_let_different_kinds_breed_together"
      },
      {
        subject: "fields",
        prohibition: "do_not_sow_with_two_kinds_of_seed"
      },
      {
        subject: "garments",
        prohibition: "do_not_wear_clothing_woven_of_two_kinds_of_material"
      }
    ],
    slavery: {
      condition: "if_man_lies_with_female_slave_betrothed_to_another",
      consequence: [
        "there_shall_be_punishment",
        "they_shall_not_be_put_to_death_because_she_was_not_free",
        "he_shall_bring_guilt_offering_to_Lord_at_entrance_of_tent_of_meeting",
        "priest_shall_make_atonement_for_him_with_ram_of_guilt_offering"
      ]
    },
    fruitTrees: {
      rule: [
        "when_you_plant_any_fruit_tree",
        "count_its_fruit_as_uncircumcised_for_three_years",
        "in_fourth_year_its_fruit_is_holy_an_offering_of_praise_to_the_Lord",
        "in_fifth_year_you_may_eat_its_fruit_to_increase_its_yield"
      ],
      reason: "I_am_the_Lord_your_God"
    },
    occult: {
      prohibition: [
        "do_not_eat_anything_with_blood",
        "do_not_practice_divination_or_soothsaying",
        "do_not_round_off_hair_on_temples_or_mar_beard",
        "do_not_make_cuts_on_body_for_dead_or_tattoo_yourselves"
      ],
      reason: "I_am_the_Lord"
    },
    familyAndSociety: [
      {
        prohibition: "do_not_profane_your_daughter_by_making_her_prostitute",
        consequence: "lest_land_fall_into_prostitution_and_become_full_of_depravity"
      },
      {
        command: [
          "keep_my_Sabbaths",
          "revere_my_sanctuary"
        ],
        reason: "I_am_the_Lord"
      },
      {
        prohibition: "do_not_turn_to_mediums_or_necromancers",
        consequence: "do_not_seek_them_out_to_be_defiled_by_them",
        reason: "I_am_the_Lord_your_God"
      }
    ]
  });

  // === 19:30-37 - Respect and Fairness ===
  yahweh.command({
    worship: [
      {
        command: [
          "keep_my_Sabbaths",
          "revere_my_sanctuary"
        ],
        reason: "I_am_the_Lord"
      }
    ],
    respect: [
      {
        command: "rise_before_the_aged_and_honor_the_elderly",
        instruction: "fear_your_God",
        reason: "I_am_the_Lord"
      }
    ],
    sojourner: [
      {
        instruction: [
          "do_not_wrong_sojourner_in_your_land",
          "treat_sojourner_as_native_among_you",
          "love_him_as_yourself_for_you_were_sojourners_in_Egypt"
        ],
        reason: "I_am_the_Lord_your_God"
      }
    ],
    commerce: [
      {
        prohibition: "do_not_use_dishonest_standards_in_judging_length_weight_or_quantity",
        command: "use_just_balances_just_weights_just_ephah_and_just_hin",
        reason: "I_am_the_Lord_your_God_who_brought_you_out_of_Egypt"
      }
    ],
    conclusion: {
      instruction: "observe_all_my_statutes_and_judgments_and_do_them",
      reason: "I_am_the_Lord"
    }
  });
}
