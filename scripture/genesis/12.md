function genesis_12_abramsCallAndJourney() {
  // === 12:1-3 - The Call of Abram ===
  lord.instructAbram({
    recipient: "Abram",
    command: [
      "go_from_your_country",
      "go_from_your_kindred",
      "go_from_your_fathers_house",
      "go_to_the_land_I_will_show_you"
    ],
    promise: {
      blessing: [
        "I_will_make_you_a_great_nation",
        "I_will_bless_you",
        "I_will_make_your_name_great",
        "you_shall_be_a_blessing"
      ],
      favor: [
        "I_will_bless_those_who_bless_you",
        "him_who_dishonors_you_I_will_curse"
      ],
      globalImpact: "in_you_all_families_of_earth_shall_be_blessed"
    }
  });

  // === 12:4-6 - Abram’s Departure and Arrival in Canaan ===
  abram.obeyCommand({
    context: {
      age: 75,
      departure: "from_Haran"
    },
    companions: [
      "Sarai_his_wife",
      "Lot_his_brothers_son",
      "all_their_possessions",
      "people_they_had_acquired_in_Haran"
    ],
    destination: "land_of_Canaan",
    journey: [
      {
        location: "Shechem",
        landmark: "oak_of_Moreh",
        note: "Canaanites_were_in_the_land"
      }
    ]
  });

  // === 12:7-9 - God’s Promise and Abram’s Worship ===
  lord.appearToAbram({
    location: "Shechem",
    promise: "to_your_offspring_I_will_give_this_land"
  });

  abram.buildAltar({
    location: "Shechem",
    purpose: "to_the_Lord_who_appeared_to_him"
  });

  abram.journeyFurther({
    locations: [
      {
        place: "hill_country_east_of_Bethel",
        action: [
          "pitched_his_tent",
          "Bethel_on_west_Ai_on_east",
          "built_altar_to_the_Lord",
          "called_upon_name_of_Lord"
        ]
      },
      {
        place: "Negeb",
        action: "journeyed_on_by_stages"
      }
    ]
  });

  // === 12:10-20 - Abram in Egypt ===
  abram.enterEgypt({
    reason: "famine_in_the_land",
    action: {
      instructionToSarai: {
        context: "you_are_a_woman_beautiful_in_appearance",
        fear: "Egyptians_will_kill_me_but_let_you_live",
        request: [
          "say_you_are_my_sister",
          "that_it_may_go_well_with_me",
          "my_life_may_be_spared"
        ]
      }
    }
  });

  egyptians.respondToSarai({
    observation: "she_was_very_beautiful",
    action: [
      "Pharaohs_princes_saw_her",
      "commended_her_to_Pharaoh",
      "woman_was_taken_into_Pharaohs_house"
    ]
  });

  abram.receiveFavor({
    source: "Pharaoh",
    gifts: [
      "sheep",
      "oxen",
      "male_donkeys",
      "male_servants",
      "female_servants",
      "female_donkeys",
      "camels"
    ]
  });

  lord.intervene({
    target: "Pharaoh_and_his_house",
    action: "struck_with_great_plagues",
    reason: "because_of_Sarai_Abrams_wife"
  });

  pharaoh.confrontAbram({
    discovery: "Sarai_was_Abrams_wife",
    rebuke: [
      "why_did_you_not_tell_me_she_was_your_wife",
      "why_did_you_say_she_is_my_sister",
      "I_took_her_for_my_wife"
    ],
    command: [
      "here_is_your_wife_take_her",
      "go"
    ]
  });

  pharaoh.escortAbram({
    action: [
      "sent_him_away",
      "with_his_wife_and_all_he_had"
    ]
  });
}
