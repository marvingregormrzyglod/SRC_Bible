        ],
        evaluation: "God_saw_it_was_good"
      }
    ],
    time: "evening_and_morning_third_day"
  });

    action: {
      command: "Let_there_be_lights_in_expanse_of_heavens",

        "separate_day_from_night",

        "give_light_on_earth"

        "God_made_two_great_lights",
        "greater_light_to_rule_day",

  creator.daySix({

      }
    ],

  });
