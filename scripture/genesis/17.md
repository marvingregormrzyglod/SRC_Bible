function genesis_covenant_of_circumcision() {  
    // Genesis 17:1-2 (God’s Covenant Renewal)  
    abram.age = 99;  
    god.appearAs("El_Shaddai", {  
        command: "walk_blameless",  
        covenantTerms: {  
            descendants: "exceedingly_numerous",  
            relationship: "God_to_you_and_offspring"  
        }  
    });  

    // Genesis 17:3-8 (Name Change & Land Promise)  
    abram.rename("abraham", { meaning: "father_of_multitude" });  
    sarai.rename("sarah", { meaning: "princess_of_nations" });  
    god.pledgeLand("everlasting_possession", {  
        boundaries: "entire_Canaan",  
        recipients: "descendants_after_you"  
    });  

    // Genesis 17:9-14 (Circumcision Ordinance)  
    circumcision = god.establishSign("covenant_seal", {  
        parameters: {  
            subjects: "every_male",  
            age: "8_days_old",  
            exceptions: "none",  
            penalty: "cut_off_from_people"  
        }  
    });  
    abraham.household.applyCircumcision({  
        immediate: true,  
        participants: [ishmael, "all_servants_born_or_bought"]  
    });  

    // Genesis 17:15-21 (Isaac’s Promise)  
    god.revealPlan(sarah, {  
        action: "bless_her",  
        outcome: "son_by_next_year",  
        name: "isaac",  
        covenantStatus: "established_with_him"  
    });  
    abraham.react("laugh", {  
        thought: "100_year_old_father?",  
        plea: "may_ishmael_live_before_you"  
    });  
    god.override("no_isaac_will_be_covenant_heir", {  
        ishmaelBlessing: "father_of_12_princes",  
        isaacTiming: "appointed_season_next_year"  
    });  

    // Genesis 17:23-27 (Immediate Obedience)  
    abraham.obey("same_day", {  
        circumcised: [  
            self,  
            ishmael.age = 13,  
            "all_household_males"  
        ]  
    });  

    return {  
        covenant: {  
            sign: circumcision,  
            heir: "isaac",  
            status: "everlasting"  
        },  
        irony: abraham.laugh.log("doubts_before_miracle"),  
        foreshadowing: isaac.waitFor("birth_and_sacrifice_test")  
    };  
}
