function joshua_4_memorialStones() {
  // === 4:1-9 - Setting Up Memorial Stones ===
  lord.instructJoshua({
    context: {
      timing: "when_all_nation_had_finished_crossing_Jordan"
    },
    command: {
      toJoshua: {
        action: "choose_twelve_men_from_people_one_from_each_tribe",
        instruction: [
          "take_twelve_stones_from_middle_of_Jordan_where_priests_feet_stood_firm",
          "carry_them_over_with_you",
          "set_them_down_in_place_where_you_lodge_tonight"
        ]
      }
    },
    execution: {
      joshuaToTwelveMen: {
        selection: "called_twelve_men_chosen_from_Israel_one_from_each_tribe",
        command: [
          "pass_on_before_ark_of_Lord_your_God_into_middle_of_Jordan",
          "each_take_up_a_stone_on_his_shoulder_according_to_number_of_tribes",
          "that_this_may_be_a_sign_among_you"
        ],
        purpose: [
          "when_your_children_ask_in_time_to_come_what_do_these_stones_mean",
          "you_shall_tell_them_waters_of_Jordan_were_cut_off_before_ark_of_covenant",
          "these_stones_shall_be_memorial_to_people_of_Israel_forever"
        ]
      },
      action: {
        result: [
          "people_of_Israel_did_as_Joshua_commanded",
          "took_up_twelve_stones_from_middle_of_Jordan",
          "carried_them_to_place_where_they_lodged_and_set_them_down"
        ]
      },
      additionalMemorial: {
        joshuaAction: [
          "set_up_twelve_stones_in_middle_of_Jordan",
          "in_place_where_feet_of_priests_carrying_ark_stood",
          "they_are_there_to_this_day"
        ]
      }
    }
  });

  // === 4:10-14 - Completion of the Crossing ===
  crossing.complete({
    priests: {
      action: [
        "stood_in_middle_of_Jordan_until_everything_Lord_commanded_Joshua_was_done",
        "according_to_all_Moses_had_commanded_Joshua"
      ]
    },
    people: {
      action: "hurried_and_crossed",
      completion: "when_all_people_had_finished_crossing",
      priestsMovement: "ark_of_Lord_and_priests_passed_over_before_people"
    },
    transjordanTribes: {
      contribution: [
        "armed_men_of_Reubenites_Gadites_and_half_tribe_of_Manasseh",
        "about_forty_thousand_ready_for_war_passed_before_Lord_to_plains_of_Jericho"
      ]
    },
    lordExaltation: {
      result: [
        "that_day_Lord_exalted_Joshua_in_sight_of_all_Israel",
        "they_stood_in_awe_of_him_as_they_had_of_Moses_all_days_of_his_life"
      ]
    }
  });

  // === 4:15-18 - Priests Exit the Jordan ===
  lord.commandJoshua({
    instruction: {
      toPriests: "come_up_out_of_Jordan"
    },
    execution: {
      joshuaToPriests: {
        command: "come_up_out_of_Jordan"
      },
      result: [
        "when_priests_carrying_ark_of_covenant_came_up_from_middle_of_Jordan",
        "soles_of_priests_feet_were_lifted_up_on_dry_ground",
        "waters_of_Jordan_returned_to_their_place_and_overflowed_all_its_banks_as_before"
      ]
    }
  });

  // === 4:19-24 - Encampment and Purpose of Stones ===
  people.encamp({
    location: "Gilgal_on_east_border_of_Jericho",
    date: "tenth_day_of_first_month",
    action: {
      joshua: {
        setup: "set_up_twelve_stones_taken_from_Jordan_at_Gilgal"
      },
      explanationToIsrael: {
        message: [
          "when_your_children_ask_their_fathers_in_times_to_come_what_do_these_stones_mean",
          "let_your_children_know_Israel_passed_over_Jordan_on_dry_ground",
          "Lord_your_God_dried_up_waters_of_Jordan_before_you_until_you_passed_over",
          "as_Lord_your_God_did_to_Red_Sea_which_He_dried_up_before_us_until_we_passed_over"
        ],
        purpose: [
          "that_all_peoples_of_earth_may_know_hand_of_Lord_is_mighty",
          "that_you_may_fear_Lord_your_God_forever"
        ]
      }
    }
  });
}
