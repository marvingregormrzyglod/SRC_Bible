# Joshua 1
```
function joshua_1_commissionAndEncouragement() {
  // === 1:1-9 - God's Commission to Joshua ===
  lord.addressJoshua({
    context: {
      after: "death_of_Moses_servant_of_Lord",
      recipient: "Joshua_son_of_Nun_Moses_aide"
    },
    commission: {
      charge: [
        "Moses_my_servant_is_dead",
        "now_arise_go_over_this_Jordan_you_and_all_this_people",
        "to_land_I_am_giving_to_them_to_people_of_Israel"
      ],
      promise: [
        "every_place_sole_of_your_foot_will_tread_I_have_given_you",
        "from_wilderness_and_Lebanon_to_great_river_Euphrates_and_great_sea",
        "no_one_will_stand_before_you_all_days_of_your_life",
        "as_I_was_with_Moses_so_I_will_be_with_you",
        "I_will_not_leave_you_or_forsake_you"
      ],
      exhortation: [
        {
          command: "be_strong_and_courageous",
          reason: "you_shall_cause_this_people_to_inherit_land_I_swore_to_their_fathers"
        },
        {
          command: "be_strong_and_very_courageous",
          instruction: [
            "be_careful_to_do_according_to_all_law_Moses_commanded",
            "do_not_turn_from_it_to_right_or_left",
            "that_you_may_have_good_success_wherever_you_go"
          ]
        },
        {
          instruction: [
            "book_of_law_shall_not_depart_from_your_mouth",
            "meditate_on_it_day_and_night",
            "be_careful_to_do_all_written_in_it",
            "then_you_will_make_your_way_prosperous_and_have_good_success"
          ]
        },
        {
          command: "be_strong_and_courageous",
          assurance: "do_not_be_afraid_or_dismayed",
          reason: "Lord_your_God_is_with_you_wherever_you_go"
        }
      ]
    }
  });

  // === 1:10-11 - Joshua's Instructions to the People ===
  joshua.commandOfficers({
    audience: "officers_of_the_people",
    instruction: {
      action: "pass_through_camp_and_command_people",
      message: [
        "prepare_your_provisions",
        "within_three_days_you_are_to_pass_over_this_Jordan",
        "to_go_in_to_take_possession_of_land_Lord_your_God_is_giving_you"
      ]
    }
  });

  // === 1:12-15 - Joshua Addresses the Transjordan Tribes ===
  joshua.addressTransjordanTribes({
    audience: [
      "Reubenites",
      "Gadites",
      "half_tribe_of_Manasseh"
    ],
    reminder: {
      past: [
        "Moses_servant_of_Lord_gave_you_this_land_beyond_Jordan",
        "Lord_your_God_is_giving_you_rest_and_has_given_you_this_land"
      ],
      instruction: [
        "your_wives_children_and_livestock_may_remain_in_land_Moses_gave_you",
        "but_all_your_mighty_men_of_valor_shall_pass_over_armed_before_your_brothers",
        "help_them_until_Lord_gives_rest_to_your_brothers_and_they_take_possession"
      ],
      condition: "then_you_shall_return_to_your_own_land_and_take_possession"
    }
  });

  // === 1:16-18 - The People's Response ===
  people.respondToJoshua({
    commitment: [
      "all_you_have_commanded_we_will_do",
      "wherever_you_send_us_we_will_go"
    ],
    loyalty: [
      "just_as_we_obeyed_Moses_in_all_things_so_we_will_obey_you",
      "may_Lord_your_God_be_with_you_as_He_was_with_Moses"
    ],
    warning: [
      "whoever_rebels_against_your_command_and_does_not_obey_shall_be_put_to_death"
    ],
    encouragement: "only_be_strong_and_courageous"
  });
}
```
