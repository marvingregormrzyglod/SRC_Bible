function joshua_7_sinOfAchanAndItsConsequences() {
  // === 7:1 - Israel’s Sin and God’s Anger ===
  lord.expressAnger({
    context: {
      sin: "Israelites_broke_faith_regarding_devoted_things",
      perpetrator: "Achan_son_of_Carmi_son_of_Zabdi_son_of_Zerah_of_tribe_of_Judah",
      action: "took_some_devoted_things"
    },
    consequence: "Lord’s_anger_burned_against_Israelites"
  });

  // === 7:2-5 - Defeat at Ai ===
  joshua.sendSpies({
    from: "Jericho",
    to: "Ai_near_Beth-aven_east_of_Bethel",
    action: "spy_out_the_land",
    report: {
      assessment: "people_of_Ai_are_few",
      recommendation: "send_two_or_three_thousand_men_to_take_it",
      note: "do_not_make_all_people_toil_there"
    }
  });

  israel.attackAi({
    force: "about_three_thousand_men",
    outcome: {
      result: "fled_before_men_of_Ai",
      casualties: "about_thirty-six_men_killed",
      pursuit: "struck_down_from_gate_to_Shebarim",
      impact: "hearts_of_people_melted_and_became_like_water"
    }
  });

  // === 7:6-9 - Joshua’s Lament ===
  joshua.lamentBeforeLord({
    actions: [
      "tore_his_clothes",
      "fell_to_ground_before_ark_of_Lord_until_evening",
      "elders_of_Israel_with_him",
      "put_dust_on_their_heads"
    ],
    prayer: {
      address: "O_Lord_God",
      questions: [
        "why_have_you_brought_this_people_over_Jordan_to_give_us_into_hands_of_Amorites_to_destroy_us",
        "would_it_not_have_been_better_to_stay_beyond_Jordan"
      ],
      plea: "what_can_I_say_when_Israel_has_turned_their_backs_before_enemies",
      concern: [
        "Canaanites_and_all_inhabitants_of_land_will_hear_and_surround_us",
        "cut_off_our_name_from_earth"
      ],
      finalQuestion: "what_will_you_do_for_your_great_name"
    }
  });

  // === 7:10-15 - God’s Response and Instructions ===
  lord.respondToJoshua({
    command: "Get_up_why_have_you_fallen_on_your_face",
    explanation: {
      sin: "Israel_has_sinned",
      violations: [
        "transgressed_my_covenant",
        "taken_devoted_things",
        "stolen_and_lied",
        "put_them_among_their_own_belongings"
      ],
      consequence: "Israel_could_not_stand_before_enemies_but_turned_their_backs",
      status: "devoted_to_destruction_until_you_remove_devoted_things"
    },
    instructions: {
      action: "consecrate_the_people",
      announcement: "in_morning_present_yourselves_by_tribes",
      process: [
        "tribe_Lord_takes_come_by_clans",
        "clan_Lord_takes_come_by_households",
        "household_Lord_takes_come_man_by_man"
      ],
      outcome: "he_who_is_taken_with_devoted_things_shall_be_burned_with_fire_he_and_all_he_has",
      reason: "he_has_transgressed_covenant_of_Lord_and_done_outrageous_thing_in_Israel"
    }
  });

  // === 7:16-21 - Discovery of Achan ===
  joshua.executeGodsInstructions({
    process: {
      selection: [
        "brought_Israel_by_tribes_and_tribe_of_Judah_was_taken",
        "brought_clans_of_Judah_and_clan_of_Zerahites_was_taken",
        "brought_household_of_Zerahites_man_by_man_and_Zabdi_was_taken",
        "brought_household_of_Zabdi_man_by_man_and_Achan_son_of_Carmi_son_of_Zabdi_son_of_Zerah_was_taken"
      ]
    },
    confrontation: joshua.addressAchan({
      command: "My_son_give_glory_to_Lord_God_of_Israel_and_give_praise_to_Him_tell_me_what_you_have_done",
      plea: "do_not_hide_it_from_me"
    }),
    confession: achan.respond({
      admission: "truly_I_have_sinned_against_Lord_God_of_Israel",
      details: [
        "saw_among_spoil_beautiful_cloak_from_Shinar",
        "two_hundred_shekels_of_silver",
        "bar_of_gold_weighing_fifty_shekels",
        "coveted_and_took_them",
        "they_are_hidden_in_earth_inside_my_tent_with_silver_underneath"
      ]
    })
  });

  // === 7:22-26 - Punishment and Restoration ===
  joshua.punishAchan({
    verification: {
      action: "sent_messengers_to_tent",
      finding: "found_devoted_things_hidden_with_silver_underneath",
      result: "took_them_from_tent_and_brought_them_to_Joshua_and_all_Israelites",
      display: "poured_them_out_before_Lord"
    },
    execution: {
      target: "Achan_and_his_sons_daughters_cattle_donkeys_sheep_tent_and_all_he_had",
      location: "brought_to_Valley_of_Achor",
      process: [
        joshua.declare({
          statement: "Why_did_you_bring_trouble_on_us_Lord_brings_trouble_on_you_today"
        }),
        "all_Israel_stoned_him_with_stones",
        "burned_them_with_fire_and_stoned_them_with_stones",
        "raised_over_him_great_heap_of_stones_to_this_day"
      ]
    },
    outcome: {
      lord: "turned_from_his_burning_anger",
      locationName: "Valley_of_Achor_meaning_trouble"
    }
  });
}
