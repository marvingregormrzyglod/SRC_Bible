function jeremiah_14_droughtAndJudgment() {
  // === 14:1-6 - Description of the Drought ===
  jeremiah.describeCalamity({
    context: {
      event: "drought_over_Judah",
      source: "word_of_Lord_to_Jeremiah"
    },
    imagery: {
      land: [
        "Judah_mourns_and_its_gates_languish",
        "people_in_black_to_the_ground",
        "cry_of_Jerusalem_goes_up"
      ],
      people: [
        "nobles_send_servants_for_water",
        "they_find_no_water_and_return_with_empty_jars",
        "ashamed_and_confounded_they_cover_their_heads"
      ],
      ground: [
        "because_ground_is_dried_up_no_rain",
        "farmers_ashamed_they_cover_their_heads"
      ],
      wildlife: [
        "doe_in_field_abandons_newborn_because_no_grass",
        "wild_donkeys_stand_on_bare_heights_panting_like_jackals",
        "their_eyes_fail_because_no_vegetation"
      ]
    }
  });

  // === 14:7-9 - Jeremiah’s Plea for Mercy ===
  jeremiah.intercedeForPeople({
    confession: [
      "our_iniquities_testify_against_us",
      "our_apostasies_are_many",
      "we_have_sinned_against_you"
    ],
    appeal: {
      basis: [
        "O_Hope_of_Israel_its_Savior_in_time_of_trouble",
        "why_should_you_be_like_stranger_in_land",
        "like_traveler_turning_aside_to_tarry_for_night",
        "why_should_you_be_like_man_confused",
        "like_warrior_who_cannot_save"
      ],
      plea: [
        "you_are_in_our_midst_O_Lord",
        "we_are_called_by_your_name",
        "do_not_leave_us"
      ]
    }
  });

  // === 14:10-12 - Lord’s Rejection of Plea ===
  lord.respondToPeople({
    accusation: [
      "they_loved_to_wander_thus",
      "they_did_not_restrain_their_feet"
    ],
    judgment: [
      "Lord_does_not_accept_them",
      "I_will_remember_their_iniquity",
      "I_will_punish_their_sins"
    ],
    instructionToJeremiah: [
      "do_not_pray_for_welfare_of_this_people",
      "though_they_fast_I_will_not_hear_their_cry",
      "though_they_offer_burnt_offerings_and_grain_offerings_I_will_not_accept_them",
      "I_will_consume_them_by_sword_famine_and_pestilence"
    ]
  });

  // === 14:13-16 - Condemnation of False Prophets ===
  jeremiah.complain({
    defense: [
      "Lord_God_the_prophets_say_to_them",
      "you_shall_not_see_sword_nor_have_famine",
      "I_will_give_you_assured_peace_in_this_place"
    ]
  });

  lord.respond({
    verdict: [
      "prophets_prophesy_lies_in_my_name",
      "I_did_not_send_them_nor_command_them_nor_speak_to_them",
      "they_prophesy_false_vision_divination_worthlessness_and_deceit_of_their_own_minds"
    ],
    judgmentOnProphets: [
      "prophets_who_prophesy_to_you_will_die_by_famine_and_sword",
      "no_one_to_bury_them_their_wives_sons_and_daughters",
      "I_will_pour_out_their_evil_upon_them"
    ],
    judgmentOnPeople: [
      "people_to_whom_they_prophesy_will_be_cast_into_streets_of_Jerusalem",
      "dead_by_famine_and_sword_with_none_to_bury_them",
      "I_will_pour_out_their_evil_upon_them"
    ]
  });

  // === 14:17-18 - Jeremiah’s Lament ===
  jeremiah.mournForPeople({
    commandFromLord: [
      "let_my_eyes_run_down_with_tears_night_and_day",
      "let_them_not_cease"
    ],
    reason: [
      "virgin_daughter_of_my_people_is_struck_down_with_crushing_blow",
      "with_very_grievous_wound"
    ],
    observation: [
      "if_I_go_into_field_those_slain_by_sword",
      "if_I_enter_city_those_sick_with_famine",
      "prophet_and_priest_wander_without_knowledge"
    ]
  });

  // === 14:19-22 - Plea for Healing and Covenant Mercy ===
  jeremiah.intercedeAgain({
    questions: [
      "have_you_utterly_rejected_Judah",
      "does_your_soul_loathe_Zion",
      "why_have_you_struck_us_down_so_no_healing_for_us",
      "we_hoped_for_peace_but_no_good_came",
      "for_time_of_healing_but_there_is_terror"
    ],
    confession: [
      "we_acknowledge_our_wickedness_O_Lord",
      "wickedness_of_our_fathers_for_we_have_sinned_against_you"
    ],
    plea: [
      "do_not_spurn_us_for_your_name’s_sake",
      "do_not_dishonor_your_glorious_throne",
      "remember_and_do_not_break_your_covenant_with_us"
    ],
    argument: [
      "are_there_any_among_vain_idols_of_nations_who_can_bring_rain",
      "can_heavens_give_showers",
      "are_you_not_He_O_Lord_our_God",
      "we_set_our_hope_in_you_for_you_do_all_these_things"
    ]
  });
}
