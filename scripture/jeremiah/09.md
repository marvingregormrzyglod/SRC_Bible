      desire: [

      ],

        "their_tongue_is_a_deadly_arrow",

        "shall_I_not_punish_them_for_these_things",

        "sound_of_cattle_is_not_heard",
        "birds_of_air_and_beasts_have_fled_and_are_gone"

        "I_will_make_Jerusalem_a_heap_of_ruins",
        "a_lair_of_jackals",
        "cities_of_Judah_a_desolation_without_inhabitant"
      ]

  });

  jeremiah.poseDivineInquiry({

      "who_is_the_man_so_wise_to_understand_this",

    divineAnswer: {
      reason: [
        "because_they_have_forsaken_My_law_I_set_before_them",

        "followed_Baals_as_their_fathers_taught_them"
      ],

        "I_will_feed_this_people_with_bitter_food",
        "give_them_poisoned_water_to_drink",

        "send_sword_after_them_until_I_have_consumed_them"

    }
  });

  // === 9:17-22 - Call to Mourn ===
  jeremiah.issueDivineCommand({
    mourning: {
      instruction: [
        "call_for_wailing_women_to_come",
        "send_for_skillful_women_to_come",
        "let_them_make_haste_and_raise_wail_over_us"

        "that_our_eyes_may_run_with_tears",

      ]
    },
    instruction: {
      trueBoast: [
        "let_him_who_boasts_boast_in_this",
        "that_he_understands_and_knows_Me",
        "that_I_am_Lord_who_practices_steadfast_love_justice_and_righteousness_in_earth"
      ],
      reason: [
        "in_these_things_I_delight"
      ],

    }
  });

        "I_will_punish_all_circumcised_only_in_flesh",
        "Egypt_Judah_Edom_Ammon_Moab_and_all_in_desert_who_dwell_with_cut_hair"
      ],
      reason: [
        "all_these_nations_are_uncircumcised",
        "house_of_Israel_is_uncircumcised_in_heart"
      ]
    }
  });
}
