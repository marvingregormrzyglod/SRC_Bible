        "so_each_may_turn_from_evil_way",
        "I_may_forgive_their_iniquity_and_sin"
      ]
    }
  });

        "you_go_and_read_words_of_Lord_from_scroll",

      ],
      hope: "perhaps_they_will_present_supplication_before_Lord_and_turn_from_evil_way",
      reasonForHope: "great_anger_and_wrath_Lord_pronounced_against_this_people"
    }
  });

  // === 36:9-10 - Baruch Reads the Scroll ===
  baruch.performReading({
    context: {
      time: "fifth_year_of_Jehoiakim_son_of_Josiah_king_of_Judah_ninth_month",
      event: "fast_proclaimed_before_Lord_for_all_people_in_Jerusalem_and_those_from_cities_of_Judah"
    },
    location: "house_of_Lord_in_chamber_of_Gemariah_son_of_Shaphan_the_secretary",
    action: "read_words_of_Jeremiah_from_scroll_to_all_people"
  });

  // === 36:11-19 - Officials Hear and Respond ===
  officials.processReading({
    initiator: "Micaiah_son_of_Gemariah_son_of_Shaphan",

      "heard_all_words_of_Lord_from_scroll",

    ],

        "Elishama_the_secretary",
        "Delaiah_son_of_Shemaiah",

        "Gemariah_son_of_Shaphan",
        "Zedekiah_son_of_Hananiah",
        "all_other_officials"
      ],
      actions: [
        "sent_Jehudi_to_summon_Baruch",
        "Baruch_read_scroll_to_them",
        "they_felt_fear_and_said_we_must_report_to_king",
        "inquired_how_Baruch_wrote_words",
        "advised_Baruch_and_Jeremiah_to_hide"
      ]
    }
  });

  kingJehoiakim.reactToScroll({
    context: {
      location: "kings_house_in_ninth_month",
      setting: "in_winter_quarters_by_firepot"

    action: [
      "officials_reported_words_to_king",
      "king_sent_Jehudi_to_get_scroll",
      "Jehudi_read_scroll_to_king_and_princes",
      "king_cut_scroll_with_knife_and_burned_it_in_firepot",
      "king_and_servants_showed_no_fear_nor_tore_clothes"

    defiance: {
      command: "seize_Baruch_and_Jeremiah",
      outcome: "Lord_hid_them"

  });

        "write_all_words_of_first_scroll_burned_by_Jehoiakim"

      condemnationOfJehoiakim: [
        "he_burned_scroll_with_words_of_Lord",
        "did_not_hear_nor_turn_from_evil",
        "therefore_Lord_says_regarding_Jehoiakim_king_of_Judah",
        {
          judgments: [
            "he_shall_have_no_one_to_sit_on_throne_of_David",
            "his_dead_body_cast_out_to_heat_by_day_and_frost_by_night",

        }
      ]
    }
  });

  // === 36:32 - Jeremiah Rewrites the Scroll ===
  jeremiah.fulfillCommand({
    scribe: "Baruch_son_of_Neriah",
    action: [
      "took_another_scroll",
      "wrote_all_words_of_scroll_burned_by_Jehoiakim",
      "added_many_similar_words"
    ]
  });
}
