    ],
    verdict: {

    }
  });

  lord.lamentCondition({
    observation: [
      "appalling_and_horrible_thing_in_the_land",
      "prophets_prophesy_falsely",
      "priests_rule_at_their_direction"
    ],
    critique: {
      people: "my_people_love_to_have_it_so"
    },
    warning: {
      question: "what_will_you_do_when_the_end_comes"
    }
  });
}
