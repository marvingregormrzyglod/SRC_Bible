function jeremiah_35_obedienceOfRechabites() {
  // === 35:1-11 - Command to Test the Rechabites ===
  jeremiah.receiveInstruction({
    context: {
      time: "days_of_Jehoiakim_son_of_Josiah_king_of_Judah",
      source: "LORD"
    },
    instruction: {
      target: "house_of_Rechabites",
      action: [
        "bring_them_to_house_of_LORD_into_one_of_chambers",
        "offer_them_wine_to_drink"
      ]
    }
  });

  jeremiah.executeInstruction({
    actions: [
      {
        step: "went_to_sons_of_Hanan_son_of_Igdaliah_man_of_God",
        target: "Rechabites",
        individuals: [
          "Jaazaniah_son_of_Jeremiah_son_of_Habazziniah",
          "his_brothers",
          "all_his_sons",
          "whole_house_of_Rechabites"
        ]
      },
      {
        step: "brought_them_to_house_of_LORD",
        location: "chamber_of_sons_of_Hanan_son_of_Igdaliah_man_of_God",
        position: "beside_chamber_of_princes_above_chamber_of_Maaseiah_son_of_Shallum_keeper_of_threshold"
      },
      {
        step: "set_before_Rechabites",
        items: [
          "pitchers_full_of_wine",
          "cups"
        ],
        invitation: "Drink_wine"
      }
    ]
  });

  rechabites.respond({
    refusal: {
      reason: "we_will_drink_no_wine",
      authority: "Jonadab_son_of_Rechab_our_father",
      commands: [
        "drink_no_wine_neither_you_nor_your_sons_forever",
        "build_no_house",
        "sow_no_seed",
        "plant_no_vineyard_nor_have_any",
        "dwell_in_tents_all_your_days"
      ],
      purpose: "that_you_may_live_many_days_in_land_where_you_sojourn"
    },
    obedience: [
      "we_have_obeyed_voice_of_Jonadab_son_of_Rechab_our_father_in_all_he_commanded",
      "we_our_wives_our_sons_our_daughters_drink_no_wine",
      "we_have_built_no_houses_to_dwell_in",
      "we_have_no_vineyard_field_or_seed",
      "we_have_lived_in_tents",
      "we_have_obeyed_and_done_all_Jonadab_our_father_commanded"
    ],
    exception: {
      reason: "when_Nebuchadnezzar_king_of_Babylon_came_up_against_land",
      action: [
        "said_Come_let_us_go_to_Jerusalem_for_fear_of_army_of_Chaldeans_and_Syrians",
        "dwell_in_Jerusalem"
      ]
    }
  });

  // === 35:12-17 - Rebuke to Judah for Disobedience ===
  lord.speakThroughJeremiah({
    audience: "men_of_Judah_and_inhabitants_of_Jerusalem",
    message: {
      contrast: {
        rechabites: [
          "have_obeyed_command_of_Jonadab_their_father",
          "kept_all_his_commands",
          "done_according_to_all_he_commanded"
        ],
        judah: [
          "have_not_obeyed_Me",
          "I_have_spoken_to_you_persistently_but_you_have_not_listened",
          "I_have_called_to_you_but_you_have_not_answered"
        ]
      },
      example: {
        subject: "sons_of_Jonadab_son_of_Rechab",
        action: "performed_command_of_their_father",
        implication: "but_this_people_has_not_obeyed_Me"
      },
      judgment: {
        target: "house_of_Judah_and_inhabitants_of_Jerusalem",
        reason: "they_have_not_listened_or_answered_Me",
        consequence: [
          "I_will_bring_upon_Judah_and_inhabitants_of_Jerusalem",
          "all_the_disaster_I_pronounced_against_them"
        ]
      }
    }
  });

  // === 35:18-19 - Blessing on the Rechabites ===
  jeremiah.deliverBlessing({
    target: "house_of_Rechabites",
    message: {
      commendation: [
        "you_have_obeyed_command_of_Jonadab_your_father",
        "kept_all_his_commands",
        "done_according_to_all_he_commanded"
      ],
      promise: {
        source: "LORD_of_hosts_God_of_Israel",
        declaration: [
          "Jonadab_son_of_Rechab_shall_never_lack_man_to_stand_before_Me"
        ]
      }
    }
  });
}
