function jeremiah_22_divineJudgment() {
  // === 22:1-5 - Call to Justice for Judah’s Kings ===
  lord.commandJeremiah({
    audience: "king_of_Judah_sitting_on_throne_of_David",
    location: "house_of_king",
    message: {
      directive: [
        "do_justice_and_righteousness",
        "deliver_robber_from_hand_of_oppressor",
        "do_no_wrong_or_violence_to_resident_alien_fatherless_and_widow",
        "do_not_shed_innocent_blood_in_this_place"
      ],
      promise: {
        condition: "if_you_obey_these_words",
        outcome: [
          "kings_sitting_on_David’s_throne_shall_ride_in_chariots_and_horses",
          "with_servants_and_people"
        ]
      },
      warning: {
        condition: "if_you_do_not_obey_these_words",
        outcome: [
          "this_house_shall_become_a_desolation",
          "sworn_by_Myself_says_the_Lord"
        ]
      }
    }
  });

  // === 22:6-9 - Judgment on Judah’s Palace ===
  lord.declareJudgment({
    target: "house_of_king_of_Judah",
    metaphor: [
      "like_Gilead_to_Me_like_summit_of_Lebanon"
    ],
    verdict: {
      desolation: "I_will_make_you_a_desert_uninhabited_cities",
      agents: [
        "I_will_prepare_destroyers_against_you",
        "each_with_his_weapons",
        "they_shall_cut_down_your_choicest_cedars_and_cast_them_into_fire"
      ]
    },
    nationsResponse: {
      question: "Why_has_the_Lord_done_thus_to_this_great_city",
      answer: "Because_they_have_forsaken_the_covenant_of_the_Lord_their_God_and_worshiped_and_served_other_gods"
    }
  });

  // === 22:10-12 - Oracle Against Shallum (Jehoahaz) ===
  lord.pronounceOracle({
    target: "Shallum_son_of_Josiah_king_of_Judah",
    instruction: [
      "do_not_weep_for_the_dead",
      "but_weep_bitterly_for_him_who_goes_away"
    ],
    prophecy: [
      "he_shall_never_return_nor_see_his_native_land",
      "Shallum_shall_die_in_place_where_they_led_him_captive"
    ]
  });

  // === 22:13-17 - Woe to Jehoiakim ===
  lord.denounceJehoiakim({
    target: "Jehoiakim_son_of_Josiah_king_of_Judah",
    charges: [
      "builds_his_house_by_unrighteousness_and_upper_rooms_by_injustice",
      "makes_neighbor_serve_without_wages",
      "focuses_on_dishonest_gain_and_shedding_innocent_blood_and_oppression_and_extortion"
    ],
    contrast: {
      fatherJosiah: [
        "did_justice_and_righteousness",
        "judged_cause_of_poor_and_needy",
        "knew_Me_says_the_Lord"
      ],
      Jehoiakim: [
        "eyes_and_heart_only_for_dishonest_gain",
        "shedding_innocent_blood",
        "oppression_and_extortion"
      ]
    }
  });

  // === 22:18-19 - Judgment on Jehoiakim’s Death ===
  lord.predictJehoiakimFate({
    lament: {
      absence: [
        "no_lament_of_Alas_my_brother_or_Alas_sister",
        "no_lament_of_Alas_lord_or_Alas_his_majesty"
      ],
      burial: "buried_with_burial_of_donkey_dragged_and_cast_out_beyond_gates_of_Jerusalem"
    }
  });

  // === 22:20-23 - Lament for Jerusalem ===
  lord.callToJerusalem({
    action: [
      "go_up_to_Lebanon_and_cry_out",
      "lift_up_your_voice_in_Bashan",
      "cry_from_Abarim"
    ],
    reason: "all_your_lovers_are_destroyed",
    rebuke: [
      "I_spoke_to_you_in_your_prosperity_but_you_said_I_will_not_listen",
      "this_has_been_your_way_from_youth_you_have_not_obeyed_My_voice"
    ],
    judgment: [
      "wind_shall_shepherd_all_your_shepherds",
      "your_lovers_shall_go_into_captivity",
      "you_shall_be_ashamed_and_confounded_for_all_your_evil"
    ],
    metaphor: {
      Jerusalem: "inhabitant_of_Lebanon_nested_in_cedars",
      pain: "pangs_come_upon_you_like_woman_in_labor"
    }
  });

  // === 22:24-30 - Oracle Against Jehoiachin (Coniah) ===
  lord.pronounceOracle({
    target: "Coniah_son_of_Jehoiakim_king_of_Judah",
    declaration: [
      "though_you_were_signet_ring_on_My_right_hand_I_would_tear_you_off",
      "I_will_give_you_into_hand_of_those_who_seek_your_life",
      "you_and_your_mother_shall_be_cast_into_another_country"
    ],
    prophecy: [
      "you_shall_die_there_and_not_see_this_land_again",
      "write_this_man_childless_a_man_who_shall_not_succeed_in_his_days",
      "none_of_his_offspring_shall_succeed_in_sitting_on_throne_of_David"
    ],
    rhetoricalQuestion: [
      "Is_this_man_Coniah_a_despised_broken_pot",
      "Why_are_he_and_his_offspring_hurled_into_land_they_do_not_know"
    ],
    lament: "O_land_land_land_hear_the_word_of_the_Lord"
  });
}
