function jeremiah_15_judgmentAndLament() {
  // === 15:1-4 - Irrevocable Judgment on Judah ===
  lord.declareToJeremiah({
    verdict: {
      intercessionRejected: [
        "though_Moses_and_Samuel_stood_before_me",
        "my_heart_would_not_turn_toward_this_people"
      ],
      command: "send_them_out_of_my_sight_and_let_them_go"
    },
    destinies: [
      {
        group: "those_for_death",
        fate: "to_death"
      },
      {
        group: "those_for_sword",
        fate: "to_sword"
      },
      {
        group: "those_for_famine",
        fate: "to_famine"
      },
      {
        group: "those_for_captivity",
        fate: "to_captivity"
      }
    ],
    cause: [
      "because_of_what_Manasseh_son_of_Hezekiah_king_of_Judah_did_in_Jerusalem"
    ],
    judgment: [
      "I_will_make_them_horror_to_all_kingdoms_of_earth",
      "I_will_appoint_over_them_four_kinds_of_destroyers",
      {
        destroyers: [
          "sword_to_kill",
          "dogs_to_tear",
          "birds_of_air_and_beasts_of_earth_to_devour_and_destroy"
        ]
      }
    ]
  });

  // === 15:5-9 - Lament and Destruction of Jerusalem ===
  lord.pronounceWoe({
    audience: "Jerusalem",
    rhetoricalQuestions: [
      "who_will_have_pity_on_you",
      "who_will_mourn_for_you",
      "who_will_turn_to_ask_of_your_welfare"
    ],
    accusation: [
      "you_have_rejected_me",
      "you_keep_going_backward"
    ],
    judgment: [
      "I_have_stretched_out_my_hand_against_you_and_destroyed_you",
      "I_am_weary_of_relenting",
      "I_have_winnowed_them_with_winnowing_fork_in_gates_of_land",
      "I_have_bereaved_them_I_have_destroyed_my_people",
      "they_did_not_turn_from_their_ways"
    ],
    consequences: [
      {
        group: "widows",
        fate: "more_numerous_than_sand_of_seas"
      },
      {
        group: "mother_of_young_men",
        fate: [
          "I_have_brought_against_her_destroyer_at_noonday",
          "I_have_made_anguish_and_terror_fall_upon_her_suddenly"
        ]
      },
      {
        group: "she_who_bore_seven",
        fate: [
          "languishes",
          "gasps_for_breath",
          "her_sun_went_down_while_it_was_yet_day",
          "she_has_been_shamed_and_disgraced"
        ]
      },
      {
        group: "survivors",
        fate: "I_will_give_to_sword_before_their_enemies"
      }
    ]
  });

  // === 15:10-14 - Jeremiah’s Complaint and God’s Response ===
  jeremiah.lament({
    expression: [
      "woe_is_me_my_mother_that_you_bore_me",
      "man_of_strife_and_contention_to_whole_land",
      "I_have_not_lent_nor_borrowed_yet_all_curse_me"
    ]
  });

  lord.respondToJeremiah({
    assurance: [
      "surely_your_enemies_will_go_into_captivity",
      "I_will_make_your_enemies_plead_with_you_in_time_of_trouble"
    ],
    rhetoricalQuestions: [
      "can_iron_break_iron_from_north_and_bronze"
    ],
    judgmentOnJudah: [
      "your_wealth_and_treasures_I_will_give_as_plunder_without_price",
      "for_all_your_sins_throughout_your_borders",
      "I_will_make_you_serve_your_enemies_in_land_you_do_not_know",
      "I_have_kindled_fire_in_my_anger_that_will_burn_forever"
    ]
  });

  // === 15:15-18 - Jeremiah’s Plea and Complaint ===
  jeremiah.appealToLord({
    plea: [
      "O_Lord_you_know",
      "remember_me_and_visit_me",
      "take_vengeance_for_me_on_my_persecutors",
      "do_not_take_me_away_in_your_patience"
    ],
    basis: [
      "for_your_sake_I_bear_reproach",
      "your_words_were_found_and_I_ate_them",
      "your_words_became_joy_and_delight_of_my_heart",
      "I_am_called_by_your_name_O_Lord_God_of_hosts"
    ],
    commitment: [
      "I_did_not_sit_in_company_of_revelers",
      "I_sat_alone_because_your_hand_was_upon_me",
      "I_was_filled_with_indignation"
    ],
    complaint: [
      "why_is_my_pain_unceasing",
      "why_is_my_wound_incurable_refusing_to_be_healed",
      "will_you_be_to_me_like_deceptive_brook_like_waters_that_fail"
    ]
  });

  // === 15:19-21 - Lord’s Response and Promise ===
  lord.answerJeremiah({
    condition: [
      "if_you_return_I_will_restore_you",
      "you_shall_stand_before_me",
      "if_you_utter_what_is_precious_and_not_what_is_worthless",
      "you_shall_be_as_my_mouth"
    ],
    instruction: [
      "let_them_return_to_you_but_you_must_not_return_to_them"
    ],
    promise: [
      "I_will_make_you_to_this_people_a_fortified_wall_of_bronze",
      "they_will_fight_against_you_but_not_prevail",
      "I_am_with_you_to_save_and_deliver_you",
      "I_will_deliver_you_out_of_hand_of_wicked",
      "I_will_redeem_you_from_grasp_of_ruthless"
    ]
  });
}
