      "Edom_Moab_and_Ammonites",
      "kings_of_Tyre_Sidon_and_coastland_across_sea",
      "Dedan_Tema_Buz_and_all_who_cut_hair",

      "all_kingdoms_on_face_of_earth",
      "Sheshach_after_them"
    ],
    divineDeclaration: {

      reason: "sword_I_am_sending_among_you",
      inevitability: [

    },
    lament: {

      reason: [

        "no_flight_for_shepherds",

        "peaceful_folds_are_brought_to_silence",

        "He_has_left_His_lair_like_lion",
        "land_is_desolate_because_of_fierce_oppressor_and_His_anger"
      ]
    }
  });
}
