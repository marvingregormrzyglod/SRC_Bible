function jeremiah_4_warningAndLament() {
  // === 4:1-2 - Call to Repentance ===
  lord.declareToIsrael({
    context: {
      audience: "Israel"
    },
    condition: [
      "if_you_return_to_me",
      "if_you_remove_your_abominations_from_my_presence",
      "if_you_do_not_waver"
    ],
    oath: {
      requirement: "swear_the_Lord_lives_in_truth_justice_and_righteousness",
      outcome: [
        "nations_shall_bless_themselves_in_Him",
        "in_Him_they_shall_glory"
      ]
    }
  });

  // === 4:3-4 - Exhortation to Genuine Reform ===
  lord.addressJudahAndJerusalem({
    metaphor: [
      "break_up_your_fallow_ground",
      "do_not_sow_among_thorns"
    ],
    instruction: [
      {
        action: "circumcise_yourselves_to_the_Lord",
        target: "remove_foreskins_of_your_hearts",
        audience: "men_of_Judah_and_inhabitants_of_Jerusalem"
      }
    ],
    warning: {
      consequence: [
        "my_wrath_go_forth_like_fire",
        "burn_with_none_to_quench_it"
      ],
      reason: "because_of_evil_of_your_deeds"
    }
  });

  // === 4:5-9 - Announcement of Coming Judgment ===
  lord.proclaimJudgment({
    context: {
      region: "Judah_and_Jerusalem"
    },
    instruction: [
      "sound_the_trumpet_throughout_the_land",
      "cry_aloud_and_say_assemble",
      "let_us_go_into_fortified_cities"
    ],
    imagery: {
      threat: "lion_has_gone_up_from_his_thicket",
      agent: "destroyer_of_nations_has_set_out",
      impact: [
        "your_cities_will_be_ruined",
        "without_inhabitant"
      ]
    },
    reason: [
      "your_ways_and_deeds_have_brought_this_upon_you",
      "your_evil_is_so_bitter_it_reaches_your_heart"
    ],
    outcome: {
      leaders: [
        "king_and_princes_shall_lose_heart",
        "priests_shall_be_horrified",
        "prophets_appalled"
      ]
    }
  });

  // === 4:10-12 - Jeremiah’s Complaint and God’s Response ===
  jeremiah.complain({
    statement: "you_have_utterly_deceived_this_people_and_Jerusalem",
    claim: "saying_you_shall_have_peace_while_sword_reaches_their_soul"
  });

  lord.respond({
    imagery: {
      judgment: [
        "a_wind_too_strong_for_winnowing",
        "a_full_wind_from_bare_heights_in_the_desert"
      ],
      target: "daughter_of_my_people",
      purpose: "not_to_winnow_or_cleanse"
    }
  });

  // === 4:13-18 - Description of Invader’s Approach ===
  lord.describeInvasion({
    imagery: {
      invader: [
        "he_comes_up_like_clouds",
        "his_chariots_like_whirlwind",
        "his_horses_swifter_than_eagles"
      ],
      cry: "woe_to_us_for_we_are_ruined"
    },
    exhortation: {
      audience: "Jerusalem",
      action: "wash_your_heart_from_evil_that_you_may_be_saved",
      warning: "how_long_shall_evil_thoughts_lodge_within_you"
    },
    announcement: {
      source: "voice_declares_from_Dan",
      report: "proclaims_trouble_from_hills_of_Ephraim",
      scope: [
        "tell_it_to_the_nations",
        "say_against_Jerusalem_besiegers_come_from_far_country"
      ],
      reason: [
        "your_ways_and_deeds_have_brought_this_upon_you",
        "this_is_your_doom_bitter_reaching_your_heart"
      ]
    }
  });

  // === 4:19-22 - Jeremiah’s Anguish and God’s Verdict ===
  jeremiah.expressAnguish({
    emotion: [
      "my_heart_is_in_turmoil_within_me",
      "I_cannot_keep_silent"
    ],
    perception: [
      "sound_of_trumpet_heard",
      "alarm_of_war",
      "destruction_upon_destruction_proclaimed",
      "whole_land_laid_waste"
    ],
    imagery: "my_tents_suddenly_destroyed_my_curtains_in_a_moment"
  });

  lord.decree({
    verdict: {
      people: "my_people_are_foolish",
      understanding: [
        "they_do_not_know_me",
        "they_are_stupid_children",
        "they_have_no_understanding"
      ],
      behavior: [
        "skilled_in_doing_evil",
        "do_not_know_how_to_do_good"
      ]
    }
  });

  // === 4:23-28 - Vision of Cosmic Desolation ===
  jeremiah.visionOfJudgment({
    imagery: {
      earth: [
        "formless_and_void",
        "heavens_had_no_light"
      ],
      mountains: "quaking_all_hills_moving_to_and_fro",
      humanity: "no_man_all_birds_of_air_had_fled",
      land: [
        "fruitful_land_became_desert",
        "all_its_cities_laid_in_ruins"
      ]
    },
    cause: "before_the_Lord_before_his_fierce_anger"
  });

  lord.declare({
    judgment: [
      "whole_land_shall_be_a_desolation",
      "yet_I_will_not_make_a_full_end"
    ],
    response: {
      earth: "shall_mourn",
      heavens: "be_black_above"
    },
    resolve: [
      "I_have_spoken_and_will_not_relent",
      "I_have_purposed_and_will_not_turn_back"
    ]
  });

  // === 4:29-31 - Lament for Judah’s Fate ===
  lord.describeFlight({
    context: {
      trigger: "noise_of_horsemen_and_archers"
    },
    reaction: [
      "every_city_flees",
      "they_climb_into_thickets_and_rocky_places",
      "cities_forsaken_no_man_dwells_in_them"
    ]
  });

  lord.addressJudah({
    imagery: {
      personification: "daughter_of_Zion",
      behavior: [
        "you_adorn_yourself_with_ornaments_of_gold",
        "enlarge_your_eyes_with_paint"
      ],
      futility: "in_vain_you_make_yourself_beautiful",
      rejection: "your_lovers_despise_you_they_seek_your_life"
    },
    anguish: {
      metaphor: "like_a_woman_in_labor",
      cry: [
        "woe_is_me",
        "I_am_fainting_before_murderers"
      ]
    }
  });
}
