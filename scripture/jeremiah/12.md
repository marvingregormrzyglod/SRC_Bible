function jeremiah_12_questioningGodsJusticeAndJudgment() {

    context: {
      plea: "righteous_are_you_Lord_when_I_contend_with_you"
    },
    complaint: {
      inquiry: [
        "why_does_way_of_wicked_prosper",
        "why_do_all_who_are_treacherous_thrive"
      ],
      accusation: [
        "you_plant_them_and_they_take_root",

          "you_see_me_and_test_my_heart_toward_you"

          "pull_them_out_like_sheep_for_slaughter",

      },
      lament: [
        "how_long_will_land_mourn",
        "grass_of_every_field_wither",

        "they_say_he_will_not_see_our_latter_end"

    }
  });

    expression: {
      sorrow: [
        "I_have_forsaken_my_house",

      ],
      imagery: [
        "is_my_heritage_to_me_like_speckled_bird_of_prey",
        "other_birds_of_prey_surround_and_attack_her",

      ]
    },
    judgment: {

        "spoilers_come_on_all_bare_heights_in_wilderness",
        "sword_of_Lord_devours_from_one_end_of_land_to_other",
        "no_flesh_has_peace"
      ],
      futility: [
        "they_sowed_wheat_but_reaped_thorns",
        "put_themselves_to_pain_but_profit_nothing",
        "ashamed_of_your_harvest_because_of_fierce_anger_of_Lord"
      ]
    }
  });

  // === 12:14-17 - Judgment and Hope for Neighboring Nations ===
  lord.declareJudgmentAndMercy({
    audience: "evil_neighbors_who_touch_inheritance_of_my_people_Israel",
    judgment: [
      "I_will_pluck_them_up_from_their_land",
      "I_will_pluck_out_house_of_Judah_from_among_them"
    ],
    mercy: {
      condition: [
        "after_I_pluck_them_up",
        "I_will_have_compassion_on_them",
        "bring_them_again_each_to_his_heritage_and_land"
      ],
      promise: [
        "if_they_learn_ways_of_my_people",
        "swear_by_my_name_as_Lord_lives",
        "as_they_taught_my_people_to_swear_by_Baal",
        "then_they_shall_be_built_up_in_midst_of_my_people"
      ],
      warning: [
        "if_any_nation_will_not_listen",
        "I_will_utterly_pluck_up_and_destroy_that_nation"
      ]
    }
  });
}
