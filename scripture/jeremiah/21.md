# Jeremiah 21
```
function jeremiah_21_judgmentOnJudah() {
  // === 21:1-2 - King Zedekiah's Inquiry ===
  jeremiah.receiveInquiry({
    source: "King_Zedekiah",
    messengers: [
      "Pashhur_son_of_Malchiah",
      "Zephaniah_son_of_Maaseiah_the_priest"
    ],
    request: {
      action: "inquire_of_the_LORD_for_us",
      context: "Nebuchadnezzar_king_of_Babylon_is_making_war_against_us",
      hope: [
        "perhaps_LORD_will_deal_with_us_according_to_His_wonderful_deeds",
        "make_him_withdraw_from_us"
      ]
    }
  });

  // === 21:3-7 - Jeremiah's Response to Zedekiah ===
  jeremiah.deliverResponse({
    audience: "Zedekiah",
    message: {
      authority: "LORD_God_of_Israel",
      judgment: [
        "I_Myself_will_turn_back_weapons_of_war_in_your_hands",
        "I_will_strike_inhabitants_of_this_city_both_man_and_beast",
        "they_shall_die_of_great_pestilence"
      ],
      divine_action: [
        "I_will_fight_against_you_with_outstretched_hand_and_strong_arm",
        "in_anger_and_fury_and_great_wrath"
      ],
      fate_of_king: {
        target: [
          "you",
          "your_servants",
          "this_people",
          "those_who_survive_pestilence_sword_and_famine"
        ],
        outcome: [
          "I_will_give_into_hand_of_Nebuchadnezzar_king_of_Babylon",
          "into_hand_of_their_enemies_who_seek_their_life",
          "he_shall_strike_them_with_edge_of_sword",
          "he_shall_not_pity_spare_or_have_compassion"
        ]
      }
    }
  });

  // === 21:8-10 - Choice of Life or Death ===
  jeremiah.proclaimToPeople({
    audience: "this_people",
    message: {
      authority: "LORD",
      options: {
        path_of_life: {
          action: "go_out_and_surrender_to_Chaldeans_who_besiege_you",
          outcome: "your_life_shall_be_to_you_as_spoil"
        },
        path_of_death: {
          action: "remain_in_this_city",
          outcome: [
            "shall_die_by_sword_famine_and_pestilence"
          ]
        }
      },
      divine_intent: [
        "I_have_set_My_face_against_this_city_for_harm_not_for_good",
        "it_shall_be_given_into_hand_of_king_of_Babylon",
        "he_shall_burn_it_with_fire"
      ]
    }
  });

  // === 21:11-14 - Warning to the House of David ===
  jeremiah.addressHouseOfDavid({
    audience: "house_of_king_of_Judah",
    message: {
      authority: "LORD",
      commands: [
        {
          action: "execute_justice_in_the_morning",
          purpose: "deliver_him_who_is_robbed_from_hand_of_oppressor"
        },
        {
          warning: [
            "lest_My_wrath_go_forth_like_fire",
            "burn_with_none_to_quench_it",
            "because_of_your_evil_deeds"
          ]
        }
      ],
      prophecy: {
        target: "inhabitant_of_valley_and_rock_of_plain",
        accusation: [
          "you_who_say_who_shall_come_down_against_us",
          "who_shall_enter_our_habitations"
        ],
        judgment: [
          "I_will_punish_you_according_to_fruit_of_your_deeds",
          "I_will_kindle_a_fire_in_her_forest",
          "it_shall_devour_all_around_her"
        ]
      }
    }
  });
}
```
