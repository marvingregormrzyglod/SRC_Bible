          ability: "according_to_his_ability"
        }
      ]
    },
    narrative: {
      actions: [
        {
          servant_1: {

        }
      ],
      return: {
        event: "after_long_time_master_returned",
        action: "settled_accounts_with_servants"

            }
          }
        },
        {
          servant_2: {
            report: "master_you_delivered_two_talents_here_I_have_made_two_more",
            response: {
              commendation: "well_done_good_and_faithful_servant",
              reward: [
                "you_have_been_faithful_over_little",

              ]
            }
          }
        },
        {
          servant_3: {
            report: [
              "master_I_knew_you_to_be_hard_man",

              "I_was_afraid_and_went_and_hid_your_talent_in_ground",
              "here_you_have_what_is_yours"
            ],
            response: {

                "wicked_and_slothful_servant",
                "you_knew_I_reap_where_I_did_not_sow",

          "I_was_naked_and_you_did_not_clothe_me",

        ],
        response: {
          question: "Lord_when_did_we_see_you_in_these_states_and_not_serve_you",
          answer: [
            "truly_I_say_to_you",
            "as_you_did_not_do_it_to_one_of_least_of_these",

          ]
        },
        outcome: "go_into_eternal_punishment"
      }
    ],
    conclusion: {
      contrast: [
        "righteous_into_eternal_life",
        "others_into_eternal_punishment"
      ]
    }
  });
