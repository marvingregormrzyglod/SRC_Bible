function isaiah_44_proclamationAndPromise() {
  // === 44:1-5 - God's Assurance to Israel ===
  lord.addressIsrael({
    audience: "Jacob_My_servant_Israel_whom_I_have_chosen",
    assurance: {
      declaration: [
        "I_will_pour_My_Spirit_on_your_offspring",
        "My_blessing_on_your_descendants"
      ],
      imagery: [
        "they_will_spring_up_like_grass_in_meadow",
        "like_poplars_by_flowing_streams"
      ],
      identity: [
        "one_will_say_I_belong_to_the_Lord",
        "another_will_call_himself_by_name_of_Jacob",
        "another_will_write_on_his_hand_The_Lords"
      ],
      prophecy: "they_will_take_name_Israel"
    }
  });

  // === 44:6-8 - The Lord as the Only God ===
  lord.declareSovereignty({
    titles: [
      "King_of_Israel",
      "Redeemer",
      "Lord_of_Hosts"
    ],
    claims: [
      "I_am_the_first_and_I_am_the_last",
      "apart_from_Me_there_is_no_God"
    ],
    challenge: {
      question: [
        "who_is_like_Me",
        "who_can_proclaim_from_ancient_times_things_not_yet_done"
      ],
      assertion: "My_counsel_will_stand_and_I_will_accomplish_all_My_purpose"
    },
    reassurance: [
      "do_not_fear_nor_be_afraid",
      "I_have_proclaimed_from_of_old_and_you_are_My_witnesses",
      "is_there_any_God_besides_Me",
      "no_there_is_no_other_Rock"
    ]
  });

  // === 44:9-20 - The Folly of Idolatry ===
  lord.condemnIdolatry({
    target: "makers_of_idols",
    critique: [
      "all_who_make_idols_are_nothing",
      "things_they_treasure_are_worthless",
      "their_witnesses_neither_see_nor_know",
      "they_will_be_put_to_shame"
    ],
    processOfIdolatry: {
      craftsman: {
        actions: [
          "shapes_wood_with_tools",
          "marks_it_with_compass",
          "forms_it_in_human_shape",
          "makes_it_to_dwell_in_shrine"
        ]
      },
      materials: {
        wood: [
          "some_he_burns_in_fire",
          "some_he_uses_to_bake_bread",
          "some_he_makes_into_god_and_worships"
        ],
        irony: [
          "from_same_wood_he_makes_fire_and_idol",
          "he_prays_to_it_saying_deliver_me_for_you_are_my_god"
        ]
      }
    },
    rebuke: [
      "they_know_nothing",
      "they_do_not_consider",
      "their_eyes_are_plastered_over_cannot_see",
      "their_minds_closed_cannot_understand",
      "he_feeds_on_ashes_deluded_heart_misleads_him"
    ]
  });

  // === 44:21-23 - Redemption and Call to Rejoice ===
  lord.callToIsrael({
    exhortation: [
      "remember_these_things_Jacob",
      "you_are_My_servant",
      "I_have_formed_you",
      "you_will_not_be_forgotten_by_Me"
    ],
    redemption: [
      "I_have_swept_away_your_offenses_like_cloud",
      "your_sins_like_morning_mist"
    ],
    invitation: "return_to_Me_for_I_have_redeemed_you",
    creationResponse: {
      command: "sing_for_joy_you_heavens_for_the_Lord_has_done_this",
      participants: [
        "heavens",
        "depths_of_earth",
        "mountains",
        "forests_and_every_tree"
      ],
      reason: "Lord_has_redeemed_Jacob_and_displayed_His_glory_in_Israel"
    }
  });

  // === 44:24-28 - The Lord's Sovereign Power ===
  lord.proclaimAuthority({
    titles: [
      "your_Redeemer",
      "who_formed_you_in_womb"
    ],
    works: [
      "I_am_the_Lord_who_made_all_things",
      "who_alone_stretched_out_heavens",
      "who_spread_out_earth_by_Myself",
      "who_foils_signs_of_false_prophets",
      "who_makes_fools_of_diviners",
      "who_overthrows_wisdom_of_wise",
      "who_confirms_word_of_His_servant"
    ],
    promises: [
      {
        target: "Jerusalem",
        action: "you_will_be_inhabited"
      },
      {
        target: "towns_of_Judah",
        action: "you_will_be_rebuilt_and_I_will_restore_their_ruins"
      },
      {
        target: "deep",
        action: "dry_up_and_make_your_streams_dry"
      },
      {
        target: "Cyrus",
        role: "My_shepherd",
        action: "he_will_accomplish_all_I_please",
        prophecy: [
          "he_will_say_of_Jerusalem_let_it_be_rebuilt",
          "of_temple_your_foundation_will_be_laid"
        ]
      }
    ]
  });
}
