function isaiah_52_proclamationOfRedemption() {

  lord.addressZion({

      audience: "Jerusalem_the_holy_city"
    },
    exhortation: [
      "awake_awake_put_on_your_strength_O_Zion",

    ],
    command: [
      "shake_yourself_from_dust",
      "arise_and_sit_down_O_Jerusalem",

          "my_name_is_blasphemed_continually_all_day_long"
        ],
        resolution: [
          "my_people_shall_know_my_name",
          "in_that_day_they_shall_know_that_it_is_I_who_speak",
          "here_am_I"
        ]
      }
    ]
  });

  // === 52:7-10 - Joyful Proclamation of Salvation ===
  lord.announceGoodNews({
    imagery: {
      messenger: [
        "how_beautiful_upon_mountains_are_feet_of_him_who_brings_good_news",
        "who_publishes_peace",
        "who_brings_good_news_of_happiness",
        "who_publishes_salvation",
        "who_says_to_Zion_your_God_reigns"
      ]
    },

      {
        watchmen: [
          "voice_of_your_watchmen_they_lift_up_their_voice",
          "together_they_sing_for_joy",
          "they_see_eye_to_eye_the_return_of_Lord_to_Zion"
        ]
      },
      {
        Jerusalem: [

        ]
      }
    ],
    revelation: [
      "Lord_has_bared_His_holy_arm_before_eyes_of_all_nations",
      "all_ends_of_earth_shall_see_salvation_of_our_God"
    ]
  });

  // === 52:11-12 - Call to Depart in Purity ===
  lord.commandDeparture({
    audience: "you_who_bear_vessels_of_Lord",
    instruction: [

      "nor_go_by_flight",
      "Lord_will_go_before_you",
      "God_of_Israel_will_be_your_rear_guard"
    ]
  });

  // === 52:13-15 - Introduction to the Suffering Servant ===
  lord.presentServant({
    description: {

      outcome: [
        "he_shall_be_high_and_lifted_up",
        "he_shall_be_exalted"
      ],
      appearance: [
        "his_appearance_was_so_marred_beyond_human_semblance",
        "his_form_beyond_that_of_children_of_mankind"
      ],
      impact: [
        "he_shall_sprinkle_many_nations",
        "kings_shall_shut_their_mouths_because_of_him",
        "what_had_not_been_told_them_they_see",
        "what_they_had_not_heard_they_understand"
      ]
    }
  });
}
