# Isaiah 45
```
function isaiah_45_sovereigntyAndSalvation() {
  // === 45:1-7 - The Lord's Anointing of Cyrus ===
  lord.anointCyrus({
    recipient: "Cyrus_My_anointed",
    purpose: {
      actions: [
        "subdue_nations_before_him",
        "strip_kings_of_their_armor",
        "open_doors_before_him_so_gates_will_not_be_shut"
      ],
      divineSupport: [
        "I_will_go_before_you_and_level_mountains",
        "I_will_break_down_gates_of_bronze",
        "I_will_cut_through_bars_of_iron",
        "I_will_give_you_hidden_treasures_riches_stored_in_secret_places"
      ],
      intent: [
        "so_you_may_know_I_am_the_Lord_God_of_Israel_who_summons_you_by_name",
        "for_sake_of_Jacob_My_servant_Israel_My_chosen"
      ]
    },
    declaration: {
      titles: ["Lord_who_has_no_equal"],
      claims: [
        "I_will_give_you_title_though_you_do_not_acknowledge_Me",
        "I_am_the_Lord_and_there_is_no_other",
        "apart_from_Me_there_is_no_God"
      ],
      sovereignty: [
        "I_form_light_and_create_darkness",
        "I_bring_prosperity_and_create_disaster",
        "I_the_Lord_do_all_these_things"
      ]
    }
  });

  // === 45:8-13 - Call for Righteousness and God's Sovereign Plan ===
  lord.commandCreation({
    imagery: [
      "you_heavens_above_rain_down_righteousness",
      "let_clouds_shower_it_down",
      "let_earth_open_wide",
      "let_salvation_spring_up",
      "let_righteousness_flourish_with_it"
    ],
    assertion: "I_the_Lord_have_created_it"
  });
  lord.rebukeQuestioners({
    target: "those_who_quarrel_with_their_Maker",
    analogy: [
      "woe_to_those_who_strive_with_Him_who_formed_them",
      "does_clay_say_to_potter_what_are_you_making",
      "does_your_work_say_the_potter_has_no_hands"
    ],
    challenge: [
      "woe_to_those_who_say_to_father_what_have_you_begotten",
      "or_to_mother_what_have_you_brought_to_birth"
    ],
    affirmation: {
      titles: ["Lord_of_Hosts", "Holy_One_of_Israel", "your_Maker"],
      works: [
        "I_made_earth_and_created_mankind_on_it",
        "My_own_hands_stretched_out_heavens",
        "I_ordained_all_their_host"
      ],
      plan: [
        "I_will_raise_up_Cyrus_in_My_righteousness",
        "he_will_rebuild_My_city_and_set_My_exiles_free",
        "not_for_price_or_reward"
      ]
    }
  });

  // === 45:14-17 - Nations Will Acknowledge Israel’s God ===
  lord.prophesyToIsrael({
    promise: {
      nations: [
        "Egypt",
        "Cush",
        "Sabeans"
      ],
      actions: [
        "they_will_come_over_to_you_and_be_yours",
        "they_will_bow_down_before_you",
        "they_will_plead_with_you"
      ],
      confession: [
        "surely_God_is_with_you",
        "there_is_no_other_no_god_besides_Him"
      ]
    },
    declaration: {
      titles: ["God_of_Israel", "Savior"],
      attributes: [
        "truly_You_are_a_God_who_hides_Himself",
        "makers_of_idols_will_be_put_to_shame_and_disgraced",
        "Israel_will_be_saved_by_the_Lord_with_everlasting_salvation",
        "you_will_never_be_put_to_shame_or_disgraced"
      ]
    }
  });

  // === 45:18-19 - The Lord’s Open Call ===
  lord.proclaimTruth({
    titles: ["Creator_of_heavens", "God_who_formed_earth"],
    works: [
      "He_did_not_create_it_to_be_empty",
      "He_formed_it_to_be_inhabited"
    ],
    claims: [
      "I_am_the_Lord_and_there_is_no_other",
      "I_have_not_spoken_in_secret_from_somewhere_in_land_of_darkness",
      "I_have_not_said_to_Jacob’s_descendants_seek_Me_in_vain"
    ],
    affirmation: "I_the_Lord_speak_truth_I_declare_what_is_right"
  });

  // === 45:20-25 - Invitation to the Nations and Universal Salvation ===
  lord.inviteNations({
    call: {
      audience: "survivors_of_nations",
      exhortation: [
        "gather_together_and_come",
        "draw_near_you_who_have_escaped"
      ],
      critique: [
        "ignorant_are_those_who_carry_wooden_idols",
        "who_pray_to_gods_that_cannot_save"
      ]
    },
    proclamation: {
      claims: [
        "there_is_no_God_apart_from_Me",
        "a_righteous_God_and_a_Savior",
        "there_is_none_but_Me"
      ],
      invitation: [
        "turn_to_Me_and_be_saved_all_you_ends_of_earth",
        "for_I_am_God_and_there_is_no_other"
      ],
      oath: [
        "by_Myself_I_have_sworn",
        "My_mouth_has_uttered_in_all_integrity_a_word_that_will_not_be_revoked"
      ],
      prophecy: [
        "before_Me_every_knee_will_bow",
        "every_tongue_will_swear_allegiance",
        "they_will_say_of_Me_in_the_Lord_alone_are_deliverance_and_strength"
      ],
      judgment: [
        "all_who_have_raged_against_Him_will_come_to_Him_and_be_put_to_shame",
        "in_the_Lord_all_descendants_of_Israel_will_be_found_righteous_and_will_exult"
      ]
    }
  });
}
```
