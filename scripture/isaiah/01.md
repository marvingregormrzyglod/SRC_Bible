# Isaiah 1
```
function isaiah_1_rebukeAndCallToRepentance() {
  // === 1:1 - Introduction ===
  prophet.introduce({
    name: "Isaiah_son_of_Amoz",
    vision: {
      concerning: "Judah_and_Jerusalem",
      period: [
        "days_of_Uzziah",
        "Jotham",
        "Ahaz",
        "Hezekiah",
        "kings_of_Judah"
      ]
    }
  });

  // === 1:2-9 - God's Accusation Against Judah ===
  lord.declareIndictment({
    audience: "heavens_and_earth",
    charge: {
      rebellion: [
        "children_I_reared_and_brought_up_have_rebelled_against_Me",
        "ox_knows_its_owner_and_donkey_its_master’s_crib",
        "Israel_does_not_know",
        "My_people_do_not_understand"
      ],
      description: [
        "sinful_nation",
        "people_laden_with_iniquity",
        "offspring_of_evildoers",
        "children_who_deal_corruptly"
      ],
      actions: [
        "forsaken_the_Lord",
        "despised_the_Holy_One_of_Israel",
        "utterly_estranged"
      ]
    },
    rhetoricalQuestion: [
      "why_will_you_still_be_struck_down",
      "why_continue_rebellion"
    ],
    condition: {
      metaphor: [
        "whole_head_is_sick",
        "whole_heart_is_faint",
        "from_sole_of_foot_to_head_no_soundness",
        "wounds_bruises_sores_not_pressed_out_or_bound_up_or_softened_with_oil"
      ],
      desolation: [
        "land_desolate",
        "cities_burned_with_fire",
        "fields_stripped_by_foreigners",
        "daughter_Zion_left_like_booth_in_vineyard",
        "like_sieged_city"
      ]
    },
    mercy: {
      statement: "unless_Lord_of_hosts_had_left_us_few_survivors",
      comparison: "we_would_have_been_like_Sodom_and_Gomorrah"
    }
  });

  // === 1:10-17 - Rejection of Hypocritical Worship ===
  lord.addressHypocrisy({
    audience: [
      "rulers_of_Sodom",
      "people_of_Gomorrah"
    ],
    critique: {
      worship: [
        "what_to_Me_is_multitude_of_your_sacrifices",
        "I_have_had_enough_of_burnt_offerings_of_rams",
        "I_do_not_delight_in_blood_of_bulls_lambs_or_goats"
      ],
      assemblies: [
        "trampling_of_My_courts",
        "bring_no_more_vain_offerings",
        "incense_is_abomination_to_Me",
        "new_moons_Sabbaths_appointed_feasts_My_soul_hates",
        "they_have_become_burden_to_Me"
      ],
      prayer: [
        "when_you_spread_out_your_hands_I_hide_My_eyes",
        "when_you_make_many_prayers_I_do_not_listen",
        "your_hands_are_full_of_blood"
      ]
    },
    exhortation: [
      "wash_yourselves_make_yourselves_clean",
      "remove_evil_of_your_deeds_from_before_My_eyes",
      "cease_to_do_evil",
      "learn_to_do_good",
      "seek_justice",
      "correct_oppression",
      "bring_justice_to_fatherless",
      "plead_cause_of_widow"
    ]
  });

  // === 1:18-20 - Invitation to Reconciliation ===
  lord.offerRestoration({
    invitation: "come_now_let_us_reason_together",
    promise: [
      {
        condition: "though_your_sins_are_like_scarlet",
        outcome: "they_shall_be_white_as_snow"
      },
      {
        condition: "though_they_are_red_like_crimson",
        outcome: "they_shall_become_like_wool"
      }
    ],
    condition: [
      {
        obedience: "if_willing_and_obedient",
        reward: "you_shall_eat_good_of_the_land"
      },
      {
        rebellion: "if_you_refuse_and_rebel",
        consequence: "you_shall_be_eaten_by_the_sword"
      }
    ],
    authority: "mouth_of_the_Lord_has_spoken"
  });

  // === 1:21-31 - Lament and Judgment on Jerusalem ===
  lord.lamentAndJudge({
    city: "faithful_city_Zion",
    lament: {
      past: [
        "once_full_of_justice",
        "righteousness_lodged_in_her"
      ],
      present: [
        "now_murderers",
        "silver_has_become_dross",
        "wine_mixed_with_water",
        "princes_are_rebels_and_companions_of_thieves",
        "love_bribes_and_run_after_gifts",
        "do_not_bring_justice_to_fatherless",
        "widow’s_cause_does_not_come_to_them"
      ]
    },
    judgment: {
      declaration: "I_will_turn_My_hand_against_you",
      actions: [
        "purge_away_your_dross",
        "remove_all_your_alloy",
        "restore_your_judges_as_at_first",
        "counselors_as_at_beginning"
      ],
      outcome: "afterward_you_shall_be_called_city_of_righteousness"
    },
    finalJudgment: {
      targets: [
        "transgressors",
        "sinners",
        "those_who_forsake_the_Lord"
      ],
      fate: [
        "they_shall_be_consumed",
        "ashamed_of_oaks_you_desired",
        "confounded_for_gardens_you_chose",
        "you_shall_be_like_oak_whose_leaf_withers",
        "like_garden_without_water"
      ],
      imagery: [
        "strong_shall_become_tinder",
        "their_work_spark",
        "both_burn_together_with_none_to_quench"
      ]
    }
  });
}
```
