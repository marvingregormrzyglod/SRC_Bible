  lord.proclaim({

        "take_the_millstones_and_grind_flour",

        "your_reproach_shall_be_seen"
      ],
      vengeance: {

      }
    }
  });

  // === 47:5-7 - Babylon's Arrogance and Punishment ===
  lord.decree({
    audience: "daughter_of_the_Chaldeans",

        "sit_in_silence_and_go_into_darkness",
        "no_more_will_you_be_called_the_queen_of_kingdoms"

        "I_was_angry_with_My_people",

        "you_showed_them_no_mercy",
        "on_the_aged_you_made_your_yoke_exceedingly_heavy"
      ]
    }
  });

  // === 47:8-11 - Babylon's False Security and Sudden Ruin ===
  lord.rebuke({
    audience: "you_who_are_given_to_pleasures",
    arrogance: {
      claims: [
        "I_am_and_there_is_no_one_besides_me",
        "I_shall_not_sit_as_a_widow",

      ],
      attitude: "dwell_in_security_saying_in_your_heart"
    },
    judgment: {

        "these_two_things_shall_come_to_you_in_a_moment_in_one_day",
        "loss_of_children_and_widowhood"
      ],
      completeness: "they_shall_come_upon_you_in_full_measure",
      cause: [
        "in_spite_of_your_many_sorceries",

          "ruin_shall_come_upon_you_suddenly_of_which_you_know_nothing"
        ]
      }
    }
  });

  // === 47:12-15 - Futility of Babylon's Sorceries ===
  lord.challenge({
    audience: "you_who_have_labored_from_your_youth",
    taunt: {
      sorceries: [
        "stand_fast_in_your_enchantments_and_your_many_sorceries",
        "perhaps_you_may_be_able_to_succeed",
        "perhaps_you_may_inspire_terror"
      ],
      weariness: "you_are_wearied_with_your_many_counsels",
      astrologers: [
        "let_them_stand_forth_and_save_you",
        "those_who_gaze_at_the_stars",
        "who_make_predictions_at_the_new_moons"
      ]
    },
    futility: {
      outcome: [
        "they_shall_be_like_stubble",
        "the_fire_shall_burn_them",
        "they_shall_not_deliver_themselves_from_the_power_of_the_flame"
      ],
      metaphor: [
        "no_coal_for_warming",
        "no_fire_to_sit_before"
      ],
      abandonment: [
        "such_to_you_are_those_with_whom_you_have_labored",
        "your_merchants_from_your_youth",
        "they_wander_every_one_to_his_own_way",
        "there_is_no_one_to_save_you"
      ]
    }
  });
}
