# Isaiah 58
```
function isaiah_58_trueFastingAndSabbath() {
  // === 58:1-5 - Rebuke of Hypocritical Fasting ===
  God.declareToIsrael({
    context: {
      audience: "My_people",
      tone: "loudly_like_trumpet"
    },
    rebuke: {
      sin: "declare_to_them_their_transgression",
      hypocriticalFasting: {
        observation: [
          "they_seek_Me_daily",
          "delight_to_know_My_ways",
          "ask_for_just_decisions",
          "delight_in_nearness_of_God"
        ],
        complaint: [
          "why_have_we_fasted_and_You_see_it_not",
          "why_have_we_humbled_ourselves_and_You_take_no_notice"
        ],
        critique: [
          "in_day_of_your_fast_you_seek_your_own_pleasure",
          "oppress_all_your_workers",
          "fast_for_strife_and_quarreling",
          "strike_with_wicked_fist"
        ],
        unacceptableFasting: [
          "not_fasting_to_make_your_voice_heard_on_high",
          "not_bowing_down_head_like_bulrush",
          "not_spreading_sackcloth_and_ashes"
        ]
      }
    }
  });

  // === 58:6-12 - True Fasting and Its Blessings ===
  God.defineTrueFasting({
    requirements: [
      {
        action: "loose_bonds_of_wickedness",
        detail: "undo_straps_of_yoke"
      },
      {
        action: "let_oppressed_go_free",
        detail: "break_every_yoke"
      },
      {
        action: "share_bread_with_hungry",
        detail: "bring_poor_and_homeless_into_your_house"
      },
      {
        action: "cover_naked_when_you_see_him",
        detail: "not_hide_yourself_from_your_own_flesh"
      }
    ],
    promises: [
      {
        condition: "if_you_do_these",
        blessings: [
          "light_shall_break_forth_like_dawn",
          "healing_shall_spring_up_speedily",
          "righteousness_shall_go_before_you",
          "glory_of_Lord_shall_be_your_rear_guard"
        ]
      },
      {
        condition: "if_you_call_to_Lord",
        response: [
          "Lord_will_answer",
          "when_you_cry_He_will_say_Here_I_am"
        ]
      },
      {
        condition: [
          "remove_yoke_from_your_midst",
          "cease_pointing_finger_and_speaking_wickedness",
          "pour_yourself_out_for_hungry",
          "satisfy_afflicted_soul"
        ],
        blessings: [
          "light_shall_rise_in_darkness",
          "gloom_be_like_noonday",
          "Lord_will_guide_you_continually",
          "satisfy_your_desire_in_scorched_places",
          "make_your_bones_strong",
          "you_shall_be_like_watered_garden",
          "like_spring_of_water_whose_waters_do_not_fail"
        ]
      },
      {
        outcome: "your_people_shall_rebuild_ancient_ruins",
        titles: [
          "Repairer_of_Breach",
          "Restorer_of_Streets_to_Dwell_In"
        ]
      }
    ]
  });

  // === 58:13-14 - Honoring the Sabbath ===
  God.instructSabbathObservance({
    requirements: [
      {
        condition: "if_you_turn_back_your_foot_from_Sabbath",
        actions: [
          "not_doing_your_own_pleasure_on_My_holy_day",
          "call_Sabbath_a_delight",
          "call_holy_day_of_Lord_honorable",
          "honor_it_by_not_going_your_own_way",
          "not_seeking_your_own_pleasure",
          "not_speaking_your_own_words"
        ]
      }
    ],
    promises: [
      {
        blessings: [
          "you_shall_take_delight_in_Lord",
          "I_will_make_you_ride_on_heights_of_earth",
          "I_will_feed_you_with_heritage_of_Jacob_your_father"
        ],
        affirmation: "mouth_of_Lord_has_spoken"
      }
    ]
  });
}
```
