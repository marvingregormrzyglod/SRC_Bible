        "Assyrian_will_be_terror_stricken_at_voice_of_Lord",

        "every_stroke_of_appointed_staff_will_be_with_tambourines_and_lyres"
      ],
      finalOutcome: [
        "Topheth_has_long_been_prepared_for_king",
        "its_pyre_made_deep_and_wide_with_fire_and_wood_in_abundance",
        "breath_of_Lord_like_stream_of_sulfur_kindles_it"
      ]
    }
  });
}
