      ],
      reliance: [
        "call_themselves_after_the_holy_city",
        "lean_on_the_God_of_Israel",
        "whose_name_is_the_Lord_of_hosts"

    }
  });

  // === 48:3-5 - God's Foreknowledge and Israel's Stubbornness ===

    past_acts: {

        "former_things_I_declared_of_old",

    },
    reason: {

      character: [
        "your_neck_is_an_iron_sinew",
        "your_forehead_brass"
      ],
      prevention: [
        "I_declared_them_to_you_from_of_old",
        "before_they_came_to_pass_I_announced_them",
        "lest_you_should_say_My_idol_did_them"
      ]
    }
  });

  // === 48:6-8 - New Things Declared ===
  lord.reveal({
    new_prophecy: {
      announcement: [
        "you_have_heard_now_see_all_this",
        "will_you_not_declare_it",
        "from_this_time_forth_I_announce_to_you_new_things",
        "hidden_things_that_you_have_not_known"
      ],
      uniqueness: [
        "they_are_created_now_not_from_of_old",
        "before_today_you_have_not_heard_them",
        "lest_you_should_say_Behold_I_knew_them"
      ]
    },
    accusation: {
      ignorance: "you_have_not_heard_you_have_not_known",
      rebellion: [
        "from_of_old_your_ear_has_not_been_opened",
        "I_knew_that_you_would_be_utterly_treacherous",
        "you_were_called_a_rebel_from_the_womb"
      ]
    }
  });

  // === 48:9-11 - God's Mercy for His Name's Sake ===
  lord.explain({
    mercy: {
      reason: [
        "for_My_name_s_sake_I_defer_My_anger",
        "for_the_sake_of_My_praise_I_restrain_it_for_you"
      ],
      purpose: "that_I_may_not_cut_you_off"
    },
    refinement: {
      process: [
        "Behold_I_have_refined_you_but_not_as_silver",
        "I_have_tried_you_in_the_furnace_of_affliction"

      motive: [
        "for_My_own_sake_for_My_own_sake_I_do_it",
        "how_should_My_name_be_profaned",
        "I_will_not_give_My_glory_to_another"
      ]
    }
  });

  // === 48:12-16 - God's Sovereignty and Call ===
  lord.proclaim({
    authority: {
      identity: [
        "I_am_He_I_am_the_first_and_I_am_the_last",
        "My_hand_laid_the_foundation_of_the_earth",
        "My_right_hand_spread_out_the_heavens"
      ],
      summons: "when_I_call_to_them_they_stand_forth_together"
    },
    exhortation: {
      audience: ["Jacob", "Israel_whom_I_called"],
      command: [
        "assemble_all_of_you_and_listen",
        "who_among_them_has_declared_these_things"
      ],
      declaration: [
        "the_Lord_loves_him_he_shall_perform_His_purpose_on_Babylon",
        "his_arm_shall_be_against_the_Chaldeans"
      ],
      confirmation: [
        "I_even_I_have_spoken",
        "I_have_called_him_I_have_brought_him",
        "he_shall_make_his_way_successful"
      ]
    },
    revelation: {
      invitation: "come_near_to_Me_hear_this",
      openness: [
        "from_the_beginning_I_have_not_spoken_in_secret",
        "from_the_time_it_came_to_be_I_have_been_there"
      ],
      authority: "now_the_Lord_God_has_sent_Me_and_His_Spirit"

        "your_offspring_would_have_been_like_the_sand",
        "your_descendants_like_its_grains",
        "their_name_would_never_be_cut_off_or_destroyed_from_before_Me"
      ]
    }
  });

  // === 48:20-22 - Call to Flee and No Peace for the Wicked ===

    exhortation: {

        "go_out_from_Babylon",
        "flee_from_Chaldea"
      ],
      proclamation: [
        "declare_this_with_a_shout_of_joy",
        "proclaim_it_send_it_out_to_the_end_of_the_earth",

      ]
    },
    reminder: {
      past_deliverance: [
        "they_did_not_thirst_when_He_led_them_through_the_deserts",
        "He_made_water_flow_for_them_from_the_rock",
        "He_split_the_rock_and_the_water_gushed_out"
      ]
    },
    conclusion: {
      warning: "there_is_no_peace_says_the_Lord_for_the_wicked"
    }
  });
}
