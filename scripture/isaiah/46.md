function isaiah_46_godsSovereigntyAndIdolatry() {
  // === 46:1-2 - Idols' Powerlessness ===
  lord.declare({
    context: {
      idols: ["Bel", "Nebo"]
    },
    description: {
      state: [
        "bow_down",
        "stoop",
        "their_idols_are_on_beasts_and_cattle"
      ],
      burden: [
        "things_you_carry_are_loaded_as_burdens_on_weary_beasts",
        "they_stoop_they_bow_down_together",
        "they_cannot_save_the_burden"
      ],
      outcome: "they_themselves_go_into_captivity"
    }
  });

  // === 46:3-4 - God's Faithful Care ===
  lord.addressIsrael({
    audience: ["house_of_Jacob", "all_remnant_of_house_of_Israel"],
    message: {
      history: [
        "borne_by_Me_from_before_your_birth",
        "carried_from_the_womb"
      ],
      promise: [
        "even_to_your_old_age_I_am_He",
        "to_gray_hairs_I_will_carry_you",
        "I_have_made_and_I_will_bear",
        "I_will_carry_and_will_save"
      ]
    }
  });

  // === 46:5-7 - Folly of Idolatry ===
  lord.challenge({
    comparison: {
      question: "To_whom_will_you_liken_Me_and_make_Me_equal",
      action: "compare_Me_that_we_may_be_alike"
    },
    idolatry: {
      process: [
        "lavish_gold_from_the_purse",
        "weigh_out_silver_in_the_balance",
        "hire_a_goldsmith_and_he_makes_it_into_a_god"
      ],
      behavior: [
        "they_fall_down_and_worship",
        "they_bear_it_on_the_shoulder_they_carry_it",
        "set_it_in_its_place_and_it_stands",
        "from_its_place_it_does_not_move"
      ],
      futility: [
        "though_one_cries_out_to_it_it_does_not_answer",
        "it_cannot_save_him_from_his_trouble"
      ]
    }
  });

  // === 46:8-11 - God's Sovereign Purpose ===
  lord.exhort({
    audience: "transgressors",
    instruction: [
      "remember_this_and_stand_firm",
      "bring_it_to_mind_you_transgressors",
      "remember_the_former_things_of_old"
    ],
    declaration: {
      identity: [
        "I_am_God_and_there_is_no_other",
        "I_am_God_and_there_is_none_like_Me"
      ],
      sovereignty: [
        "declaring_the_end_from_the_beginning",
        "from_ancient_times_things_not_yet_done"
      ],
      resolve: [
        "My_counsel_shall_stand",
        "I_will_accomplish_all_My_purpose"
      ],
      example: [
        "calling_a_bird_of_prey_from_the_east",
        "the_man_of_My_counsel_from_a_far_country"
      ],
      assurance: [
        "I_have_spoken_and_I_will_bring_it_to_pass",
        "I_have_purposed_and_I_will_do_it"
      ]
    }
  });

  // === 46:12-13 - Call to Hear and Promise of Salvation ===
  lord.address({
    audience: ["stubborn_of_heart", "far_from_righteousness"],
    message: {
      promise: [
        "I_bring_near_My_righteousness_it_is_not_far_off",
        "My_salvation_will_not_delay"
      ],
      outcome: [
        "I_will_grant_salvation_in_Zion",
        "My_glory_to_Israel"
      ]
    }
  });
}
