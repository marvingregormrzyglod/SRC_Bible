# Obadiah 1

function obadiah_1_prophecyAgainstEdom() {
  // === 1:1-4 - Vision and Judgment Against Edom ===
  prophet.receiveVision({
    source: "Lord_GOD",
    target: "Edom",
    declaration: {
      summons: "nations_rise_up_for_battle_against_her",
      message: [
        "I_will_make_you_small_among_nations",
        "you_will_be_utterly_despised"
      ],
      pride: {
        critique: [
          "pride_of_your_heart_has_deceived_you",
          "you_dwell_in_clefts_of_rock",
          "say_in_heart_who_will_bring_me_down_to_ground"
        ],
        response: [
          "though_you_soar_like_eagle",
          "though_your_nest_is_set_among_stars",
          "from_there_I_will_bring_you_down",
          "declares_Lord"
        ]
      }
    }
  });

  // === 1:5-7 - Plunder and Betrayal of Edom ===
  prophet.describeJudgment({
    imagery: [
      {
        comparison: "if_thieves_came_to_you",
        detail: [
          "if_robbers_by_night",
          "would_they_not_steal_only_what_they_wanted"
        ]
      },
      {
        comparison: "if_grape_gatherers_came_to_you",
        detail: [
          "would_they_not_leave_gleanings"
        ]
      }
    ],
    reality: [
      "how_you_have_been_destroyed",
      "Esau_searched_out",
      "your_hidden_treasures_discovered"
    ],
    betrayal: {
      allies: [
        "men_at_peace_with_you",
        "those_who_eat_your_bread"
      ],
      actions: [
        "driven_you_to_border",
        "deceived_and_overpowered_you",
        "set_trap_for_you"
      ],
      outcome: "you_are_not_aware"
    }
  });

  // === 1:8-14 - Edom’s Arrogance and Crimes ===
  prophet.pronounceDoom({
    time: "that_day",
    actions: [
      {
        target: "wise_men_of_Edom",
        judgment: "I_will_destroy",
        location: "mountain_of_Esau"
      },
      {
        target: "warriors_of_Teman",
        judgment: "dismayed",
        purpose: "everyone_cut_off_from_mountain_of_Esau_by_slaughter"
      }
    ],
    cause: {
      reason: "because_of_violence_to_your_brother_Jacob",
      consequences: [
        "shame_shall_cover_you",
        "you_shall_be_cut_off_forever"
      ]
    },
    crimes: [
      {
        act: "stood_aloof_on_day_strangers_carried_off_Jacob’s_wealth",
        detail: "foreigners_entered_his_gates_and_cast_lots_for_Jerusalem",
        role: "you_were_like_one_of_them"
      },
      {
        act: "did_not_gloat_over_your_brother_in_day_of_his_misfortune",
        prohibition: [
          "nor_rejoice_over_people_of_Judah_in_day_of_their_ruin",
          "nor_boast_in_day_of_distress"
        ]
      },
      {
        act: "did_not_enter_gate_of_My_people_in_day_of_their_calamity",
        prohibition: [
          "nor_gloat_over_their_affliction",
          "nor_loot_their_wealth_in_day_of_their_calamity"
        ]
      },
      {
        act: "did_not_stand_at_crossway_to_cut_off_Jacob’s_fugitives",
        prohibition: "nor_hand_over_survivors_in_day_of_distress"
      }
    ]
  });

  // === 1:15-16 - The Day of the Lord ===
  prophet.announceDayOfLord({
    scope: "near_upon_all_nations",
    principle: [
      "as_you_have_done_it_will_be_done_to_you",
      "your_deeds_shall_return_on_your_own_head"
    ],
    judgment: {
      metaphor: "as_you_drank_on_My_holy_mountain",
      reality: [
        "all_nations_shall_drink_continually",
        "they_shall_drink_and_swallow",
        "they_shall_be_as_though_they_had_never_been"
      ]
    }
  });

  // === 1:17-21 - Restoration of Israel and Judgment of Edom ===
  prophet.declareRestoration({
    location: "Mount_Zion",
    promises: [
      {
        outcome: "there_shall_be_deliverance",
        detail: "it_shall_be_holy"
      },
      {
        outcome: "house_of_Jacob_shall_possess_their_own_possessions",
        detail: [
          "house_of_Jacob_shall_be_fire",
          "house_of_Joseph_a_flame",
          "house_of_Esau_stubble"
        ],
        judgment: [
          "they_shall_burn_them_and_consume_them",
          "there_shall_be_no_survivor_for_house_of_Esau",
          "declares_Lord"
        ]
      }
    ],
    territorialRestoration: [
      {
        people: "house_of_Jacob",
        possession: [
          "mountain_of_Esau",
          "land_of_Philistines",
          "fields_of_Ephraim_and_Samaria"
        ]
      },
      {
        people: "Benjamin",
        possession: "Gilead"
      },
      {
        people: "exiles_of_Israelites",
        possession: [
          "land_of_Canaanites_as_far_as_Zarephath"
        ]
      },
      {
        people: "exiles_of_Jerusalem_in_Sepharad",
        possession: "cities_of_Negeb"
      }
    ],
    finalJudgment: {
      action: "saviors_shall_go_up_to_Mount_Zion_to_rule_mountain_of_Esau",
      outcome: "kingdom_shall_be_Lord’s"
    }
  });
}
