# Luke 21
```
function luke_21_teachingAndProphecy() {
  // === 21:1-4 - The Widow’s Offering ===
  jesus.observeOffering({
    context: {
      location: "temple",
      activity: "watching_people_put_money_into_offering_box"
    },
    observation: {
      rich: "put_in_their_gifts",
      widow: {
        status: "poor",
        action: "put_in_two_small_copper_coins"
      }
    },
    teaching: {
      audience: "disciples",
      statement: [
        "this_poor_widow_has_put_in_more_than_all",
        "others_gave_out_of_their_abundance",
        "she_out_of_her_poverty_put_in_all_she_had_to_live_on"
      ]
    }
  });

  // === 21:5-6 - Prediction of Temple Destruction ===
  jesus.predictDestruction({
    context: {
      trigger: "some_spoke_of_temple_adorned_with_noble_stones_and_offerings",
      location: "temple_area"
    },
    prophecy: {
      declaration: [
        "these_things_you_see",
        "days_will_come_when_not_one_stone_will_be_left_upon_another",
        "all_will_be_thrown_down"
      ]
    }
  });

  // === 21:7-19 - Signs and Persecutions Before the End ===
  jesus.teachEndTimes({
    audience: "disciples",
    question: {
      inquiry: [
        "Teacher_when_will_these_things_be",
        "what_will_be_the_sign_when_these_things_are_about_to_take_place"
      ]
    },
    response: {
      warnings: [
        {
          deception: [
            "see_that_you_are_not_led_astray",
            "many_will_come_in_my_name_saying_I_am_he_and_time_is_at_hand",
            "do_not_go_after_them"
          ]
        },
        {
          disturbances: [
            "when_you_hear_of_wars_and_tumults_do_not_be_terrified",
            "these_things_must_first_take_place_but_end_will_not_be_at_once"
          ]
        }
      ],
      signs: [
        {
          global: [
            "nation_will_rise_against_nation",
            "kingdom_against_kingdom",
            "great_earthquakes_in_various_places",
            "famines_and_pestilences",
            "terrors_and_great_signs_from_heaven"
          ]
        },
        {
          personal: [
            "before_all_this_they_will_lay_hands_on_you",
            "persecute_you_delivering_you_to_synagogues_and_prisons",
            "bring_you_before_kings_and_governors_for_my_name’s_sake"
          ],
          purpose: "opportunity_to_bear_witness",
          instruction: [
            "settle_it_in_your_hearts_not_to_meditate_beforehand_how_to_answer",
            "I_will_give_you_mouth_and_wisdom_which_none_of_your_adversaries_will_be_able_to_withstand_or_contradict"
          ],
          trials: [
            "betrayed_by_parents_brothers_relatives_and_friends",
            "some_of_you_they_will_put_to_death",
            "you_will_be_hated_by_all_for_my_name’s_sake"
          ],
          assurance: [
            "not_a_hair_of_your_head_will_perish",
            "by_your_endurance_you_will_gain_your_lives"
          ]
        }
      ]
    }
  });

  // === 21:20-24 - Destruction of Jerusalem Foretold ===
  jesus.prophesyJerusalem({
    context: {
      continuation: "further_teaching_on_signs"
    },
    prophecy: {
      sign: "when_you_see_Jerusalem_surrounded_by_armies_know_its_desolation_is_near",
      instructions: [
        {
          for_locals: [
            "those_in_Judea_flee_to_mountains",
            "those_in_city_depart",
            "those_in_country_not_enter_city"
          ]
        },
        {
          reason: [
            "days_of_vengeance_to_fulfill_all_that_is_written",
            "woe_to_pregnant_women_and_nursing_mothers",
            "great_distress_on_land_and_wrath_against_this_people"
          ]
        }
      ],
      outcome: [
        "they_will_fall_by_edge_of_sword",
        "be_led_captive_among_all_nations",
        "Jerusalem_will_be_trampled_by_Gentiles_until_times_of_Gentiles_are_fulfilled"
      ]
    }
  });

  // === 21:25-28 - Coming of the Son of Man ===
  jesus.describeReturn({
    signs: [
      {
        cosmic: [
          "signs_in_sun_moon_and_stars",
          "on_earth_distress_of_nations_in_perplexity_because_of_roaring_of_sea_and_waves",
          "people_fainting_with_fear_and_foreboding_of_what_is_coming_on_world"
        ],
        reason: "powers_of_heavens_will_be_shaken"
      },
      {
        revelation: [
          "they_will_see_Son_of_Man_coming_in_cloud_with_power_and_great_glory"
        ]
      }
    ],
    exhortation: [
      "when_these_things_begin_to_take_place",
      "straighten_up_and_raise_your_heads",
      "because_your_redemption_is_drawing_near"
    ]
  });

  // === 21:29-33 - Parable of the Fig Tree ===
  jesus.teachParable({
    parable: {
      illustration: [
        "look_at_fig_tree_and_all_trees",
        "when_they_put_out_leaves_you_see_and_know_summer_is_near"
      ],
      application: [
        "when_you_see_these_things_taking_place",
        "know_that_kingdom_of_God_is_near"
      ]
    },
    assurance: [
      "truly_I_say_to_you_this_generation_will_not_pass_away_until_all_has_taken_place",
      "heaven_and_earth_will_pass_away_but_my_words_will_not_pass_away"
    ]
  });

  // === 21:34-36 - Exhortation to Watchfulness ===
  jesus.exhortVigilance({
    audience: "disciples",
    warning: [
      "watch_yourselves_lest_your_hearts_be_weighed_down",
      "with_dissipation_and_drunkenness_and_cares_of_life",
      "that_day_come_upon_you_suddenly_like_trap"
    ],
    description: [
      "it_will_come_upon_all_who_dwell_on_face_of_whole_earth"
    ],
    instruction: [
      "stay_awake_at_all_times",
      "praying_that_you_may_have_strength_to_escape_all_these_things",
      "to_stand_before_Son_of_Man"
    ]
  });

  // === 21:37-38 - Jesus’ Routine ===
  jesus.dailyActivity({
    daytime: {
      location: "temple",
      action: "teaching"
    },
    nighttime: {
      location: "Mount_of_Olives",
      action: "lodging"
    },
    response: {
      people: [
        "came_early_in_morning_to_hear_him",
        "in_temple"
      ]
    }
  });
}
```
