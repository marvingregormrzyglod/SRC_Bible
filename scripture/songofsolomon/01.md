# Song of Solomon 1
```
function songOfSolomon_1_loveAndLonging() {
  // === 1:1 - Title and Attribution ===
  poem.setTitle({
    name: "The_Song_of_Songs",
    author: "which_is_Solomon's"
  });

  // === 1:2-4 - The Bride's Longing for Her Beloved ===
  bride.expressDesire({
    speaker: "the_bride",
    to: "her_beloved",
    longing: [
      {
        request: "let_him_kiss_me_with_the_kisses_of_his_mouth",
        reason: "for_your_love_is_better_than_wine"
      },
      {
        praise: "your_anointing_oils_are_fragrant",
        declaration: "your_name_is_oil_poured_out",
        effect: "therefore_virgins_love_you"
      },
      {
        plea: [
          "draw_me_after_you_let_us_run",
          "the_king_has_brought_me_into_his_chambers"
        ]
      }
    ],
    chorus: {
      speakers: "daughters_of_Jerusalem",
      affirmation: [
        "we_will_exult_and_rejoice_in_you",
        "we_will_extol_your_love_more_than_wine",
        "rightly_do_they_love_you"
      ]
    }
  });

  // === 1:5-6 - The Bride's Self-Description and Defense ===
  bride.describeSelf({
    appearance: [
      "I_am_very_dark_but_lovely",
      "like_the_tents_of_Kedar",
      "like_the_curtains_of_Solomon"
    ],
    explanation: [
      {
        command: "do_not_gaze_at_me_because_I_am_dark",
        reason: "the_sun_has_looked_upon_me"
      },
      {
        circumstance: [
          "my_mother's_sons_were_angry_with_me",
          "they_made_me_keeper_of_the_vineyards",
          "but_my_own_vineyard_I_have_not_kept"
        ]
      }
    ]
  });

  // === 1:7-8 - The Bride's Inquiry and the Beloved's Response ===
  bride.inquire({
    to: "her_beloved",
    question: [
      "tell_me_you_whom_my_soul_loves",
      "where_you_pasture_your_flock",
      "where_you_make_it_lie_down_at_noon"
    ],
    concern: "why_should_I_be_like_one_who_veils_herself_beside_the_flocks_of_your_companions"
  });

  beloved.respond({
    to: "the_bride",
    address: "if_you_do_not_know_O_most_beautiful_among_women",
    instruction: [
      "follow_the_tracks_of_the_flock",
      "pasture_your_young_goats_beside_the_shepherds'_tents"
    ]
  });

  // === 1:9-11 - The Beloved's Praise of the Bride ===
  beloved.praiseBride({
    comparison: [
      "I_compare_you_my_love_to_a_mare_among_Pharaoh's_chariots"
    ],
    description: [
      "your_cheeks_are_lovely_with_ornaments",
      "your_neck_with_strings_of_jewels"
    ],
    intention: [
      "we_will_make_you_ornaments_of_gold",
      "studded_with_silver"
    ]
  });

  // === 1:12-14 - The Bride's Admiration of the Beloved ===
  bride.expressAdmiration({
    setting: "while_the_king_was_on_his_couch",
    imagery: [
      "my_nard_gave_forth_its_fragrance",
      "my_beloved_is_to_me_a_sachet_of_myrrh",
      "that_lies_between_my_breasts",
      "my_beloved_is_to_me_a_cluster_of_henna_blossoms",
      "in_the_vineyards_of_Engedi"
    ]
  });

  // === 1:15-17 - Mutual Adoration ===
  beloved.praiseBride({
    affirmation: [
      "behold_you_are_beautiful_my_love",
      "behold_you_are_beautiful",
      "your_eyes_are_doves"
    ]
  });

  bride.respond({
    affirmation: [
      "behold_you_are_beautiful_my_beloved",
      "truly_delightful"
    ],
    setting: [
      "our_couch_is_green",
      "the_beams_of_our_house_are_cedar",
      "our_rafters_are_pine"
    ]
  });
}
```
