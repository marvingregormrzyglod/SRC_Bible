# Tobit 8
```
function tobit_8_marriageAndDeliverance() {
  // === 8:1-3 - Tobias and Sarah’s Wedding Night ===
  tobiasAndSarah.beginWeddingNight({
    context: {
      setting: "After_dinner_in_prepared_room",
      participants: ["Tobias", "Sarah"]
    },
    actions: [
      tobias.recallRaphaelInstructions({
        items: [
          "Take_liver_and_heart_of_fish",
          "Place_on_embers_of_incense"
        ]
      }),
      tobias.performRitual({
        ritual: "Burn_liver_and_heart_on_incense",
        effect: "Odor_drives_demon_away",
        outcome: {
          demon: "Asmodeus",
          action: "Flees_to_upper_Egypt",
          pursuit: "Raphael_binds_demon"
        }
      })
    ]
  });

  // === 8:4-8 - Prayer of Tobias and Sarah ===
  tobiasAndSarah.prayTogether({
    context: {
      setting: "Before_consummating_marriage",
      posture: "Rise_from_bed_to_pray"
    },
    prayer: {
      tobias: {
        address: "God_of_our_ancestors",
        content: [
          "Blessed_be_Your_name_forever",
          "You_made_Adam_and_gave_him_Eve_as_helper",
          "You_said_it_is_not_good_for_man_to_be_alone",
          "I_take_Sarah_not_out_of_lust_but_with_sincerity"
        ],
        request: [
          "Grant_mercy_to_me_and_Sarah",
          "Allow_us_to_grow_old_together"
        ]
      },
      sarah: {
        response: "Amen_Amen",
        agreement: "Joins_in_prayer"
      }
    },
    outcome: "Both_return_to_bed_and_sleep_safely"
  });

  // === 8:9-14 - Raguel’s Anxiety and Discovery ===
  raguel.manageAnxiety({
    context: {
      setting: "During_wedding_night",
      concern: "Fear_of_Tobias’_death_like_previous_husbands"
    },
    actions: [
      raguel.instruct({
        to: "Servants",
        task: "Dig_grave_in_secret",
        rationale: "To_avoid_shame_if_Tobias_dies"
      }),
      raguel.sendMaid({
        task: "Check_on_Tobias_and_Sarah",
        report: [
          "They_are_alive_and_asleep",
          "All_is_well"
        ]
      }),
      raguel.rejoice({
        response: [
          "Blessed_be_God_for_His_mercy",
          "Order_grave_to_be_filled_before_dawn"
        ]
      })
    ]
  });

  // === 8:15-18 - Raguel’s Thanksgiving and Celebration ===
  raguel.offerThanksgiving({
    prayer: {
      address: "Blessed_are_You_God_of_Israel",
      content: [
        "You_have_shown_mercy_to_two_only_children",
        "Grant_them_Lord_mercy_and_safety",
        "Let_them_complete_their_lives_in_happiness_and_mercy"
      ]
    },
    actions: [
      raguel.instruct({
        to: "Servants",
        tasks: [
          "Prepare_second_room_for_continued_celebration",
          "Extend_wedding_feast_for_fourteen_days"
        ]
      })
    ]
  });

  // === 8:19-21 - Preparations for Tobias’ Departure ===
  raguel.prepareTobiasForJourney({
    actions: [
      raguel.offerWealth({
        to: "Tobias",
        gifts: [
          "Half_of_all_my_property",
          "Male_and_female_slaves",
          "Cattle_and_sheep",
          "Donkeys_and_camels",
          "Clothing_money_and_household_goods"
        ]
      }),
      raguel.instruct({
        message: [
          "Stay_with_me_for_fourteen_days_of_celebration",
          "I_will_send_messengers_to_inform_Tobit_of_your_safety",
          "Afterward_take_Sarah_and_return_to_your_father"
        ]
      })
    ]
  });
}
```
