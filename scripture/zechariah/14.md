      {

        comparison: "flee_as_from_earthquake_in_days_of_Uzziah_king_of_Judah",
        culmination: "Lord_my_God_comes_and_all_holy_ones_with_Him"
      }
    ]
  });

  // === 14:6-7 - Cosmic Changes ===
  lord.alterCreation({
    context: "that_day",
    phenomena: [
      "no_cold_or_frost",
      "one_day_known_to_Lord",
      "neither_day_nor_night",
      "at_evening_time_there_will_be_light"
    ]
  });

  // === 14:8-11 - Living Waters and Jerusalem's Exaltation ===

      image: "living_waters_flow_out_from_Jerusalem",

        "half_to_eastern_sea",

    security: {
      status: "dwell_in_safety",
      condition: "no_longer_devoted_to_destruction"
    }
  });

  lord.strikeEnemies({
    target: "all_peoples_who_wage_war_against_Jerusalem",
    judgment: {
      plague: [
        "flesh_rots_while_they_stand_on_their_feet",

      ],
      panic: "great_panic_from_Lord_upon_them",
      conflict: "each_seize_hand_of_another_and_hand_rise_against_hand_of_neighbor",

      livestockPlague: "plague_strike_horses_mules_camels_donkeys_and_all_animals_in_those_camps"
    }
  });

  // === 14:16-19 - Worship of the Lord by Survivors ===
  lord.establishWorship({
    target: "all_who_survive_of_nations_that_came_against_Jerusalem",

      {

      },
      {
        exclusion: "no_trader_in_house_of_Lord_of_hosts_on_that_day"
      }
    ]
  });
