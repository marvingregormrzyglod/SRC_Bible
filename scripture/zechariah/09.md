          ],
          judgment: [
            "Lord_will_strip_her_wealth",
            "cast_it_into_the_sea",

          "he_too_shall_be_a_remnant_for_our_God",

          "Ekron_shall_be_like_the_Jebusite"
        ]
      },
      {

        }
      }
    ]
  });

  // === 9:9-10 - The Coming King of Zion ===
  zechariah.announceMessianicKing({
    audience: "daughter_of_Zion",
    exhortation: [
      "Rejoice_greatly_O_daughter_of_Zion",
      "Shout_aloud_O_daughter_of_Jerusalem"
    ],

        attributes: [
          "just",

          "humble",
          "riding_on_a_donkey_on_a_colt_the_foal_of_a_donkey"

          {
            disarmament: "I_will_cut_off_the_chariot_from_Ephraim_and_the_war_horse_from_Jerusalem",

          },
          {
            proclamation: "He_shall_speak_peace_to_the_nations",

        ]
      }
    }
  });

  zechariah.declareRestoration({
    source: "As_for_you_also",

      {

          ]
        }

          ],
          prosperity: [
            "How_great_is_his_goodness",
            "How_great_is_his_beauty",
            "Grain_shall_make_the_young_men_flourish",
            "New_wine_the_young_women"
          ]
        }
      }
    ]
  });
}
