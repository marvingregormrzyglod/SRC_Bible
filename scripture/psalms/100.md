function psalm_100_callToWorship() {
  // === 100:1-2 - Call to Joyful Worship ===
  psalmist.exhortAllEarth({
    audience: "all_the_earth",
    actions: [
      "make_joyful_noise_to_the_Lord",
      "serve_the_Lord_with_gladness",
      "come_into_His_presence_with_singing"
    ]
  });

  // === 100:3 - Acknowledge God's Sovereignty ===
  psalmist.declareTruth({
    theology: {
      God: {
        role: "the_Lord_is_God",
        actions: [
          "He_made_us",
          "we_are_His",
          "we_are_His_people",
          "sheep_of_His_pasture"
        ]
      }
    }
  });

  // === 100:4 - Enter with Thanksgiving ===
  psalmist.instructWorshippers({
    location: "His_gates_and_courts",
    attitude: [
      "enter_with_thanksgiving",
      "enter_with_praise"
    ],
    actions: [
      "give_thanks_to_Him",
      "bless_His_name"
    ]
  });

  // === 100:5 - Proclaim God's Character ===
  psalmist.proclaimAttributes({
    God: {
      character: [
        "the_Lord_is_good",
        "His_steadfast_love_endures_forever",
        "His_faithfulness_to_all_generations"
      ]
    }
  });
}
