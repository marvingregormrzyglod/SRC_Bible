function cry_out_in_distress_and_trust() {
    // Verse 1-2
    lament = {
        target: "O_Lord",
        questions: [
            "how_long_will_You_forget_me_forever",
            "how_long_will_You_hide_Your_face_from_me",
            "how_long_must_I_take_counsel_in_my_soul",
            "how_long_shall_my_enemy_be_exalted_over_me"
        ],
        state: "having_sorrow_in_my_heart_all_the_day"
    }

    // Verse 3-4
    plea = {
        request: "consider_and_answer_me_O_Lord_my_God",
        need: "light_up_my_eyes",
        fear: "lest_I_sleep_the_sleep_of_death",
        consequence: "lest_my_enemy_say_I_have_prevailed_over_him",
        outcome: "lest_my_foes_rejoice_because_I_am_shaken"
    }

    // Verse 5-6
    trust = {
        declaration: "but_I_have_trusted_in_Your_steadfast_love",
        hope: "my_heart_shall_rejoice_in_Your_salvation",
        response: "I_will_sing_to_the_Lord",
        reason: "because_He_has_dealt_bountifully_with_me"
    }

    return faith.in_Gods_goodness()
}
