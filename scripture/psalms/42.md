function express_longing_for_God() {
    // Verse 1-2
    soul = {
        state: "thirsts",
        target: "living_God"
    }
    deer = analogy("pants_for_water")
    question = "when_can_I_meet_God?"

    // Verse 3
    tears = {
        frequency: "day_and_night",
        role: "my_food"
    }
    taunts = others.say("where_is_your_God?")

    // Verse 4
    memory = recall({
        event: "joining_the_throng",
        action: "worship_at_God’s_house",
        emotion: "joy_and_praise"
    })
    current_state = "pour_out_soul"

    // Verse 5
    if (soul.is_downcast()) {
        hope = place_in("God")
        praise = offer_to("my_Savior_and_God")
    }

    // Verse 6-7
    if (soul.is_disturbed()) {
        remembrance = "God’s_presence"
        turmoil = likened_to("waves_and_waterfalls")
        location = "land_of_Jordan"
    }

    // Verse 8
    day = God.directs("love")
    night = God.gives("song_and_prayer")
    source = "God_my_Rock"

    // Verse 9-10
    plea = ask_God({
        question: "why_forgotten?",
        issue: "oppressed_by_enemies",
        pain: "bones_suffer_mortal_agony"
    })
    taunts = enemies.say("where_is_your_God?")

    // Verse 11
    if (soul.is_downcast_and_disturbed()) {
        hope = place_in("God")
        praise = offer_to("my_Savior_and_God")
    }

    return yearning.for_God()
}
