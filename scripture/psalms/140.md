```
function psalm_140_prayerForDeliverance() {
  // === 140:1-5 - Plea for Protection from Evil Men ===
  psalmist.seekDeliverance({
    context: {
      speaker: "David",
      audience: "LORD"
    },
    plea: [
      "deliver_me_O_LORD_from_evil_men",
      "preserve_me_from_violent_men"
    ],
    accusation: [
      "who_plan_evil_things_in_their_heart",
      "continually_stir_up_wars",
      "sharpen_their_tongues_like_a_serpent",
      "venom_of_asps_is_under_their_lips"
    ],
    specificThreats: [
      "wicked_have_set_snares_for_me",
      "spread_a_net_by_the_wayside",
      "set_traps_for_me"
    ]
  });

  // === 140:6-8 - Appeal to God's Strength ===
  psalmist.affirmTrust({
    declaration: [
      "I_say_to_the_LORD_You_are_my_God",
      "give_ear_to_the_voice_of_my_pleas_for_mercy_O_LORD"
    ],
    divineAttributes: [
      "O_LORD_my_strong_deliverer",
      "You_cover_my_head_in_the_day_of_battle"
    ],
    request: [
      "grant_not_O_LORD_the_desires_of_the_wicked",
      "do_not_further_their_evil_plot_lest_they_be_exalted"
    ]
  });

  // === 140:9-11 - Imprecation Against Enemies ===
  psalmist.prayAgainstFoes({
    targets: "those_who_surround_me",
    curses: [
      "let_the_mischief_of_their_lips_overwhelm_them",
      "let_burning_coals_fall_upon_them",
      "let_them_be_cast_into_fire",
      "let_them_be_thrown_into_miry_pits_no_more_to_rise"
    ],
    furtherJudgment: [
      "let_not_a_slanderer_be_established_in_the_land",
      "let_evil_hunt_down_the_violent_man_speedily"
    ]
  });

  // === 140:12-13 - Confidence in God's Justice ===
  psalmist.expressConfidence({
    assurance: [
      "I_know_that_the_LORD_will_maintain_the_cause_of_the_afflicted",
      "will_execute_justice_for_the_needy"
    ],
    outcome: [
      "surely_the_righteous_shall_give_thanks_to_Your_name",
      "the_upright_shall_dwell_in_Your_presence"
    ]
  });
}
```
