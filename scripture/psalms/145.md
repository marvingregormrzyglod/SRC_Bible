function psalm_145_praiseToGod() {
  // === 145:1-2 - Commitment to Praise ===
  david.declarePraise({
    speaker: "David",
    intent: {
      action: "extol_You_my_God_and_King",
      duration: "forever_and_ever",
      frequency: "every_day_I_will_bless_You",
      target: "Your_name"
    }
  });

  // === 145:3-7 - Greatness of God ===
  david.proclaimGodsGreatness({
    attributes: {
      greatness: "unsearchable",
      description: "great_is_the_Lord_and_greatly_to_be_praised"
    },
    response: [
      {
        group: "one_generation",
        action: "commend_Your_works_to_another",
        focus: "Your_mighty_acts"
      },
      {
        group: "people",
        action: "speak_of_splendor_of_Your_majesty",
        focus: [
          "Your_wondrous_works",
          "abundance_of_Your_greatness",
          "Your_mighty_acts"
        ]
      },
      {
        group: "men",
        action: "utter_memory_of_Your_great_goodness",
        expression: "sing_aloud_of_Your_righteousness"
      }
    ]
  });

  // === 145:8-9 - Character of God ===
  david.describeGodsCharacter({
    qualities: [
      "gracious_and_merciful",
      "slow_to_anger",
      "abounding_in_steadfast_love",
      "good_to_all",
      "mercies_over_all_His_works"
    ]
  });

  // === 145:10-13 - Universal Praise ===
  david.callForUniversalPraise({
    participants: [
      {
        group: "all_Your_works",
        action: "give_thanks_to_You_O_Lord"
      },
      {
        group: "Your_saints",
        action: "bless_You",
        focus: [
          "speak_of_glory_of_Your_kingdom",
          "tell_of_Your_power",
          "make_known_to_sons_of_men_Your_mighty_deeds",
          "glorious_splendor_of_Your_kingdom"
        ]
      }
    ],
    kingdomAttributes: {
      nature: "everlasting_kingdom",
      duration: "endures_throughout_all_generations"
    }
  });

  // === 145:14-16 - Godâ€™s Provision and Care ===
  david.praiseGodsProvision({
    actions: [
      {
        target: "all_who_fall",
        action: "upholds"
      },
      {
        target: "all_who_are_bowed_down",
        action: "raises_up"
      },
      {
        target: "all",
        action: "gives_them_their_food_in_due_season",
        mechanism: [
          "open_Your_hand",
          "satisfy_desire_of_every_living_thing"
        ]
      }
    ],
    context: {
      expectation: "eyes_of_all_look_to_You"
    }
  });

  // === 145:17-20 - Righteousness and Nearness of God ===
  david.affirmGodsRighteousness({
    qualities: [
      "righteous_in_all_His_ways",
      "kind_in_all_His_works",
      "near_to_all_who_call_on_Him_in_truth"
    ],
    responses: [
      {
        condition: "those_who_call_on_Him",
        action: "fulfills_their_desire"
      },
      {
        condition: "those_who_fear_Him",
        action: [
          "hears_their_cry",
          "saves_them"
        ]
      },
      {
        condition: "all_who_love_Him",
        action: "preserves"
      },
      {
        condition: "all_the_wicked",
        action: "destroys"
      }
    ]
  });

  // === 145:21 - Closing Commitment to Praise ===
  david.concludePraise({
    personalCommitment: {
      action: "speak_praise_of_the_Lord",
      duration: "forever_and_ever"
    },
    universalCall: {
      target: "all_flesh",
      action: "bless_His_holy_name"
    }
  });
}
