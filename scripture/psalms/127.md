# The Psalms

## Psalm 127
```javascript
function describe_the_lords_blessing() {
    building = {
        effort: "human",
        success: "only_if_lord_builds"
    }

    guarding = {
        effort: "human",
        safety: "only_if_lord_protects"
    }

    if (lord.is_absent()) {
        work.outcome = "useless"
        sleep.lost = "in_vain"
    }

    rest = lord.gives_to("those_he_loves")
    children = {
        status: "gift_from_lord",
        value: "reward",
        strength: "like_arrows_in_hand"
    }

    if (warrior.has_children()) {
        shame.felt = false
        victory.claimed = "at_the_gate"
    }

    return blessing.depends_on_lord()
}
```
