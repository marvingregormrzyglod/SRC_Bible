function psalm_51_prayerForMercyAndRestoration() {
  // === 51:1-2 - Plea for Mercy ===
  psalmist.begForMercy({
    context: {
      superscript: "For_the_choir_director_A_Psalm_of_David",
      occasion: "when_Nathan_the_prophet_went_to_him_after_he_had_gone_in_to_Bathsheba"
    },
    address: "O_God",
    plea: [
      {
        request: "have_mercy_on_me",
        basis: [
          "according_to_Your_steadfast_love",
          "according_to_Your_abundant_mercy"
        ]
      },
      {
        request: [
          "blot_out_my_transgressions",
          "wash_me_thoroughly_from_my_iniquity",
          "cleanse_me_from_my_sin"
        ]
      }
    ]
  });

  // === 51:3-6 - Confession of Sin ===
  psalmist.confessSin({
    acknowledgment: [
      "I_know_my_transgressions",
      "my_sin_is_ever_before_me"
    ],
    confession: {
      recipient: "against_You_You_only_have_I_sinned",
      action: "done_what_is_evil_in_Your_sight",
      consequence: [
        "so_that_You_may_be_justified_in_Your_words",
        "blameless_in_Your_judgment"
      ]
    },
    reflection: [
      "behold_I_was_brought_forth_in_iniquity",
      "in_sin_did_my_mother_conceive_me",
      "behold_You_delight_in_truth_in_inward_being",
      "You_teach_me_wisdom_in_secret_heart"
    ]
  });

  // === 51:7-12 - Prayer for Cleansing and Renewal ===
  psalmist.seekRestoration({
    requests: [
      {
        action: "purge_me_with_hyssop",
        outcome: "I_shall_be_clean"
      },
      {
        action: "wash_me",
        outcome: "I_shall_be_whiter_than_snow"
      },
      {
        action: "let_me_hear_joy_and_gladness",
        outcome: "let_bones_You_have_broken_rejoice"
      },
      {
        action: [
          "hide_Your_face_from_my_sins",
          "blot_out_all_my_iniquities"
        ]
      },
      {
        action: [
          "create_in_me_a_clean_heart_O_God",
          "renew_a_right_spirit_within_me"
        ]
      },
      {
        action: [
          "cast_me_not_away_from_Your_presence",
          "take_not_Your_Holy_Spirit_from_me"
        ]
      },
      {
        action: "restore_to_me_joy_of_Your_salvation",
        outcome: "uphold_me_with_a_willing_spirit"
      }
    ]
  });

  // === 51:13-17 - Commitment to Worship and Testimony ===
  psalmist.vowResponse({
    intentions: [
      {
        action: "teach_transgressors_Your_ways",
        outcome: "sinners_will_return_to_You"
      },
      {
        action: "deliver_me_from_bloodguiltiness_O_God_God_of_my_salvation",
        outcome: "my_tongue_will_sing_aloud_of_Your_righteousness"
      },
      {
        action: "O_Lord_open_my_lips",
        outcome: "my_mouth_will_declare_Your_praise"
      }
    ],
    clarification: {
      rejection: [
        "You_will_not_delight_in_sacrifice_or_I_would_give_it",
        "You_will_not_be_pleased_with_burnt_offering"
      ],
      acceptance: [
        "sacrifices_of_God_are_broken_spirit",
        "broken_and_contrite_heart_O_God_You_will_not_despise"
      ]
    }
  });

  // === 51:18-19 - Prayer for Zion ===
  psalmist.prayForZion({
    requests: [
      {
        action: "do_good_to_Zion_in_Your_good_pleasure",
        outcome: "build_up_walls_of_Jerusalem"
      }
    ],
    outcome: {
      worship: [
        "then_will_You_delight_in_right_sacrifices",
        "in_burnt_offerings_and_whole_burnt_offerings",
        "then_bulls_will_be_offered_on_Your_altar"
      ]
    }
  });
}
