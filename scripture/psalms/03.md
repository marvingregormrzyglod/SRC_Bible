function call_for_help_in_distress() {
    // Verse 1
    prayer = {
        target: "Lord",
        plea: "how_many_are_my_foes"
    }

    // Verse 2
    enemies = {
        count: "many",
        taunts: "no_help_for_him_in_God"
    }

    // Verse 3
    shield = Lord.provides_around_me()
    glory = Lord.bestows_upon_me()
    confidence = Lord.lifts_my_head()

    // Verse 4
    voice = I.cry_aloud_to("Lord")
    answer = Lord.responds_from("holy_hill")

    // Verse 5
    sleep = Lord.sustains_me_through()
    awakening = Lord.grants_me_safely()

    // Verse 6
    if (enemies.number > "tens_of_thousands") {
        fear.is_present = false
        reason = Lord.protects_me()
    }

    // Verse 7
    deliverance = I.request_from("Lord")
    action = Lord.strikes("enemies_on_the_cheek")
    defeat = Lord.shatters("teeth_of_the_wicked")

    // Verse 8
    salvation = {
        source: "Lord",
        scope: "belongs_to_His_people"
    }

    return faith.in_Lord()
}
