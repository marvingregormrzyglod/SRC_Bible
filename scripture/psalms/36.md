# The Psalms

## Psalm 36
```pseudo
function describe_the_lords_love_and_wickedness() {
    // Verses 1-2
    wickedness = {
        heart: "sinful",
        fear_of_God: "none"
    }
    if (wickedness.looks_at_self()) {
        flaws.seen = false
    }

    // Verses 3-4
    wickedness.actions = {
        speech: "falsehood",
        deeds: "evil",
        path: "wrong"
    }

    // Verses 5-6
    lords_love = {
        faithfulness: "reaches_the_skies",
        righteousness: "like_mighty_mountains",
        justice: "like_deep_ocean"
    }

    // Verses 7-9
    refuge = lords_love.offers_to("all_people")
    abundance = lords_love.provides("feast_and_river")
    life = lords_love.gives_source_of("light")
    sight = lords_love.grants("to_see_light")

    // Verses 10-12
    prayer = {
        for_righteous: "keep_loving_them",
        for_wicked: "let_evil_fall",
        outcome: "wicked_defeated"
    }

    return lords_love.prevails()
}
```
