function describe_the_blessed_life() {
    // Verse 1
    blessed_person = {
        avoids: [
            "walking_in_wicked_counsel",
            "standing_in_sinners_path",
            "sitting_with_mockers"
        ]
    }

    // Verse 2
    joy = blessed_person.delights_in("law_of_the_Lord")
    meditation = blessed_person.reflects_on("day_and_night")

    // Verse 3
    life = blessed_person.becomes_like("tree_by_streams") {
        yields: "fruit_in_season",
        leaves: "never_wither",
        prospers: "in_all_they_do"
    }

    // Verse 4
    wicked = {
        state: "not_like_the_blessed",
        likeness: "chaff_in_the_wind"
    }

    // Verse 5
    if (judgment.day_arrives()) {
        wicked.stand = false
        sinners.belong_in_righteous_assembly = false
    }

    // Verse 6
    oversight = {
        Lord: "watches_over_righteous_path",
        wicked: "path_leads_to_destruction"
    }

    return blessed_person.thrives()
}
