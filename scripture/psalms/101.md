        "walk_with_integrity_of_heart_within_my_house"

        stance: "shall_be_far_from_me",

  });

      actions: [
        "destroy_all_the_wicked_in_the_land",
        "cut_off_all_evildoers_from_city_of_the_Lord"
      ]
    }
  });
}
