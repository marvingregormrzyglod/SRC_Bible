function psalm_55_prayerAmidBetrayalAndDistress() {

      {
        action: "attend_to_me_and_answer_me"
      }
    ],
    state: [
      "I_am_restless_in_my_complaint",
      "I_moan"
    ]
  });

  // === 55:3-8 - Description of Distress and Desire for Escape ===
  psalmist.expressAnguish({
    causes: [
      "because_of_voice_of_enemy",

    ],
    longing: {

        "I_would_lodge_in_wilderness",
        "I_would_hurry_to_find_shelter_from_raging_wind_and_tempest"
      ]
    },
    refrain: "Selah"
  });

  // === 55:9-11 - Prayer Against the Wicked ===
  psalmist.seekDivineIntervention({
    requests: [
      {
        action: [
          "destroy_O_Lord",
          "confuse_their_tongues"
        ],
        reason: [
          "I_see_violence_and_strife_in_city",
          "day_and_night_they_go_around_it_on_its_walls",
          "iniquity_and_trouble_are_within_it",
          "ruin_is_in_its_midst",
          "oppression_and_fraud_do_not_depart_from_its_marketplace"
        ]
      }
    ]
  });

  psalmist.mournBetrayal({
    source: {
      notEnemy: [
        "it_is_not_enemy_who_taunts_me",

        "I_could_hide_from_him"
      ],
      betrayer: [
        "you_a_man_my_equal",
        "my_companion_my_familiar_friend",

      ]
    },
    imprecation: [
      {
        action: [
          "let_death_steal_over_them",
          "let_them_go_down_to_Sheol_alive"
        ],
        reason: "evil_is_in_their_dwelling_place_and_heart"
      }
    ]
  });

  // === 55:16-19 - Confidence in God’s Deliverance ===
  psalmist.expressTrust({
    action: [
      "I_call_to_God",
      "the_Lord_will_save_me"
    ],
    practice: [
      "evening_morning_and_noon_I_utter_my_complaint_and_moan",
      "He_hears_my_voice"
    ],
    assurance: [
      {
        action: "He_redeems_my_soul_in_safety_from_battle_against_me",
        reason: "many_are_arrayed_against_me"
      },
      {

        basis: [
          "He_who_is_enthroned_from_of_old",
          "because_they_do_not_change_and_do_not_fear_God"
        ]
      }
    ],
    refrain: "Selah"
  });

  // === 55:20-21 - Further Condemnation of the Betrayer ===
  psalmist.denounceBetrayer({
    actions: [
      "he_has_stretched_out_his_hand_against_those_at_peace_with_him",
      "he_has_violated_his_covenant"
    ],

      "his_words_were_softer_than_oil_yet_they_were_drawn_swords"
    ]
  });

  // === 55:22-23 - Final Trust and Warning ===
  psalmist.concludeWithHope({
    exhortation: [
      {
        action: "cast_your_burden_on_the_Lord",

    ],

        subjects: [
          "men_of_blood_and_treachery_shall_not_live_out_half_their_days"
        ]
      }
    ],
    affirmation: "I_will_trust_in_You"
  });
}
