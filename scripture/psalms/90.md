function psalm_90_prayerOfMoses() {
  // === 90:1-2 - God's Eternity ===
  moses.declareGodsEternity({

      "You_have_been_our_dwelling_place_in_all_generations",

        beforeCreation: [

    ]
  });

        "You_sweep_them_away_as_with_a_flood",
        "They_are_like_a_dream",
        "Like_grass_that_is_renewed_in_the_morning",
        "In_the_morning_it_flourishes_and_is_renewed",
        "In_the_evening_it_fades_and_withers"
      ]
    }
  });

  // === 90:7-11 - Human Sin and Divine Wrath ===
  moses.reflectOnSinAndJudgment({
    cause: [
      "We_are_brought_to_an_end_by_Your_anger",
      "By_Your_wrath_we_are_dismayed"
    ],
    divineKnowledge: [
      "You_have_set_our_iniquities_before_You",
      "Our_secret_sins_in_the_light_of_Your_presence"
    ],
    consequence: [
      "All_our_days_pass_away_under_Your_wrath",
      "We_bring_our_years_to_an_end_like_a_sigh"

      duration: "Years_of_our_life_are_seventy_or_even_by_reason_of_strength_eighty",

        "Their_span_is_but_toil_and_trouble",
        "They_are_soon_gone_and_we_fly_away"
      ]
    },
    rhetoricalQuestion: {
      divineWrath: "Who_considers_the_power_of_Your_anger",
      fearDue: "Your_wrath_according_to_the_fear_due_You"
    }
  });

  // === 90:12-17 - Prayer for Wisdom and Mercy ===
  moses.prayForDivineFavor({
    requests: [
      {
        wisdom: "Teach_us_to_number_our_days",
        purpose: "That_we_may_get_a_heart_of_wisdom"
      },
      {

        plea: "Have_pity_on_Your_servants"
      },
      {

        outcome: "That_we_may_rejoice_and_be_glad_all_our_days"
      },
      {
        joy: "Make_us_glad_for_as_many_days_as_You_have_afflicted_us",
        balance: "For_as_many_years_as_we_have_seen_evil"
      },
      {
        revelation: "Let_Your_work_be_shown_to_Your_servants",
        glory: "Your_glorious_power_to_their_children"
      },
      {
        favor: "Let_the_favor_of_the_Lord_our_God_be_upon_us",
        work: "Establish_the_work_of_our_hands_upon_us",
        repetition: "Yes_establish_the_work_of_our_hands"
      }
    ]
  });
}
