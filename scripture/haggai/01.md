function haggai_1_callToRebuild() {
  // === 1:1-2 - Introduction and Context ===
  prophet.receiveWordOfLord({
    context: {
      date: "second_year_of_Darius_king",
      day: "first_day_of_sixth_month",
      recipient: "Haggai_prophet",
      audience: [
        "Zerubbabel_son_of_Shealtiel_governor_of_Judah",
        "Joshua_son_of_Jehozadak_high_priest"
      ]
    },
    message: {
      observation: "people_say_time_has_not_yet_come_to_rebuild_house_of_Lord"
    }
  });

  // === 1:3-6 - Rebuke for Neglecting Temple ===
  lord.speakThroughHaggai({
    audience: "this_people",
    rebuke: {
      question: "is_it_time_for_you_to_dwell_in_your_paneled_houses_while_this_house_lies_in_ruins",
      consequences: [
        "you_have_sown_much_but_harvested_little",
        "you_eat_but_never_have_enough",
        "you_drink_but_never_are_filled",
        "you_clothe_yourselves_but_no_one_is_w foolishnessarm",
        "you_earn_wages_to_put_into_bag_with_holes"
      ]
    }
  });

  // === 1:7-11 - Call to Consider and Act ===
  lordOfHosts.issueCommand({
    instruction: [
      "consider_your_ways",
      "go_up_to_hills_bring_wood_and_build_house",
      "that_I_may_take_pleasure_in_it_and_be_glorified"
    ],
    explanation: {
      currentState: [
        "you_expected_much_but_it_came_to_little",
        "when_you_brought_it_home_I_blew_it_away"
      ],
      reason: "because_my_house_lies_in_ruins_while_each_of_you_busies_himself_with_his_own_house",
      divineAction: [
        "heavens_above_you_withheld_dew",
        "earth_withheld_its_produce",
        "I_called_for_drought_on_land_hills_grain_new_wine_oil_and_all_ground_produces"
      ],
      affected: [
        "man_and_beast",
        "all_labor_of_your_hands"
      ]
    }
  });

  // === 1:12-15 - Response and Obedience ===
  people.respondToCall({
    leaders: [
      "Zerubbabel_son_of_Shealtiel",
      "Joshua_son_of_Jehozadak_high_priest"
    ],
    action: {
      obedience: "obeyed_voice_of_Lord_their_God_and_words_of_Haggai_prophet",
      attitude: "people_feared_Lord"
    }
  });

  lord.encourageThroughHaggai({
    message: "I_am_with_you_declares_Lord",
    result: {
      divineAction: "Lord_stirred_up_spirit_of_Zerubbabel_Joshua_and_remnant_of_people",
      workBegun: {
        date: "twenty_fourth_day_of_sixth_month",
        task: "began_work_on_house_of_Lord_of_hosts_their_God"
      }
    }
  });
}
