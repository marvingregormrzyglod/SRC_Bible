function haggai_2_encouragementAndPromise() {
  // === 2:1-9 - Encouragement for the Temple Work ===
  prophet.receiveWordOfLord({
    context: {
      date: "second_year_of_Darius_king",
      day: "twenty_first_day_of_seventh_month",
      recipient: "Haggai_prophet",
      audience: [
        "Zerubbabel_son_of_Shealtiel_governor_of_Judah",
        "Joshua_son_of_Jehozadak_high_priest",
        "remnant_of_the_people"
      ]
    },
    message: {
      question: [
        "who_among_you_saw_this_house_in_its_former_glory",
        "how_do_you_see_it_now",
        "does_it_not_seem_like_nothing_in_your_eyes"
      ],
      encouragement: [
        "be_strong_Zerubbabel",
        "be_strong_Joshua_son_of_Jehozadak_high_priest",
        "be_strong_all_you_people_of_the_land",
        "work_for_I_am_with_you_declares_Lord_of_hosts"
      ],
      covenant: "according_to_word_I_covenanted_with_you_when_you_came_out_of_Egypt",
      assurance: [
        "my_Spirit_remains_in_your_midst",
        "fear_not"
      ],
      prophecy: {
        futureAction: [
          "I_will_once_more_shake_heavens_and_earth_sea_and_dry_land",
          "I_will_shake_all_nations",
          "desired_of_all_nations_shall_come"
        ],
        promise: [
          "I_will_fill_this_house_with_glory",
          "glory_of_this_house_shall_be_greater_than_former",
          "in_this_place_I_will_give_peace_declares_Lord_of_hosts"
        ]
      }
    }
  });

  // === 2:10-19 - Holiness and Blessing ===
  lord.speakThroughHaggai({
    context: {
      date: "twenty_fourth_day_of_ninth_month",
      audience: "priests"
    },
    inquiry: [
      {
        question: "if_someone_carries_holy_meat_in_fold_of_garment_and_touches_bread_stew_wine_oil_or_any_food_does_it_become_holy",
        answer: "no"
      },
      {
        question: "if_someone_defiled_by_contact_with_dead_body_touches_any_of_these_does_it_become_unclean",
        answer: "yes"
      }
    ],
    application: {
      declaration: [
        "so_it_is_with_this_people_and_nation_before_me",
        "every_work_of_their_hands_and_what_they_offer_there_is_unclean"
      ],
      callToConsider: [
        "consider_from_this_day_onward",
        "before_stone_was_placed_on_stone_in_temple_of_Lord"
      ],
      pastConsequences: [
        "when_one_came_to_heap_of_twenty_there_were_but_ten",
        "when_one_came_to_wine_vat_to_draw_fifty_there_were_but_twenty",
        "I_struck_you_and_all_works_of_your_hands_with_blight_mildew_and_hail",
        "yet_you_did_not_return_to_me"
      ],
      promise: [
        "from_this_day_onward_I_will_bless_you",
        "consider_from_day_temple_foundation_was_laid"
      ]
    }
  });

  // === 2:20-23 - Promise to Zerubbabel ===
  lord.deliverMessage({
    context: {
      date: "twenty_fourth_day_of_month",
      recipient: "Haggai_prophet",
      audience: "Zerubbabel_son_of_Shealtiel_governor_of_Judah"
    },
    prophecy: {
      divineAction: [
        "I_will_shake_heavens_and_earth",
        "I_will_overthrow_thrones_of_kingdoms",
        "I_will_destroy_strength_of_kingdoms_of_nations",
        "I_will_overthrow_chariots_and_their_riders",
        "horses_and_riders_shall_go_down_each_by_sword_of_his_brother"
      ],
      promiseToZerubbabel: [
        "on_that_day_I_will_take_you_Zerubbabel_my_servant",
        "I_will_make_you_like_a_signet_ring",
        "for_I_have_chosen_you_declares_Lord_of_hosts"
      ]
    }
  });
}
