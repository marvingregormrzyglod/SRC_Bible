# Hosea 3
```
function hosea_3_redemptionAndRestoration() {
  // === 3:1-3 - Command to Redeem and Symbolic Act ===
  lord.commandHosea({
    instruction: {
      action: "go_again_love_a_woman_who_is_loved_by_another_man_and_is_an_adulteress",
      analogy: "even_as_the_Lord_loves_the_children_of_Israel_though_they_turn_to_other_gods_and_love_cakes_of_raisins"
    }
  });

  hosea.obeyCommand({
    action: {
      redemption: {
        woman: "adulteress",
        price: [
          "fifteen_shekels_of_silver",
          "a_homer_and_a_lethech_of_barley"
        ]
      },
      instructionToWoman: [
        "you_must_dwell_with_me_many_days",
        "you_shall_not_play_the_whore_or_belong_to_another_man",
        "so_will_I_also_be_to_you"
      ]
    }
  });

  // === 3:4-5 - Prophecy of Israel's Discipline and Restoration ===
  lord.declareProphecy({
    recipients: "children_of_Israel",
    discipline: {
      duration: "many_days",
      condition: [
        "without_king_or_prince",
        "without_sacrifice_or_pillar",
        "without_ephod_or_household_gods"
      ]
    },
    restoration: {
      timing: "afterward",
      actions: [
        "children_of_Israel_shall_return",
        "seek_the_Lord_their_God_and_David_their_king"
      ],
      outcome: [
        "they_shall_come_in_fear_to_the_Lord_and_to_His_goodness",
        "in_the_latter_days"
      ]
    }
  });
}
```
