# Hosea 14
```
function hosea_14_callToRepentanceAndRestoration() {
  // === 14:1-3 - Call to Return to the Lord ===
  prophet.urgeRepentance({
    audience: "Israel",
    exhortation: {
      action: "return_to_the_Lord_your_God",
      reason: "you_have_stumbled_because_of_your_iniquity"
    },
    instruction: {
      prayer: [
        "take_with_you_words",
        "return_to_the_Lord"
      ],
      plea: [
        "say_to_Him_take_away_all_iniquity",
        "accept_what_is_good",
        "we_will_render_offerings_of_our_lips"
      ],
      commitment: [
        "Assyria_shall_not_save_us",
        "we_will_not_ride_on_horses",
        "we_will_no_more_say_Our_God_to_work_of_our_hands",
        "for_in_You_the_orphan_finds_mercy"
      ]
    }
  });

  // === 14:4-7 - God's Promise of Healing and Restoration ===
  lord.declareRestoration({
    promise: {
      healing: "I_will_heal_their_apostasy",
      love: "I_will_love_them_freely",
      condition: "My_anger_has_turned_from_them"
    },
    imagery: {
      blessing: [
        "I_will_be_like_dew_to_Israel",
        "he_shall_blossom_like_lily",
        "cast_out_his_roots_like_Lebanon"
      ],
      flourishing: [
        "his_shoots_shall_spread_out",
        "his_beauty_shall_be_like_olive_tree",
        "his_fragrance_like_Lebanon"
      ],
      prosperity: [
        "they_shall_return_and_dwell_beneath_My_shadow",
        "they_shall_flourish_like_grain",
        "they_shall_blossom_like_vine",
        "their_fame_shall_be_like_wine_of_Lebanon"
      ]
    }
  });

  // === 14:8 - God's Appeal to Abandon Idolatry ===
  lord.addressEphraim({
    rhetoricalQuestion: "O_Ephraim_what_have_I_to_do_with_idols",
    declaration: [
      "it_is_I_who_answer_and_look_after_you",
      "I_am_like_an_evergreen_cypress",
      "from_Me_comes_your_fruit"
    ]
  });

  // === 14:9 - Conclusion and Call to Wisdom ===
  prophet.concludeWithWisdom({
    exhortation: {
      audience: "whoever_is_wise",
      instruction: [
        "let_him_understand_these_things",
        "let_him_know_them"
      ]
    },
    truth: [
      "the_ways_of_the_Lord_are_right",
      "the_righteous_walk_in_them",
      "transgressors_stumble_in_them"
    ]
  });
}
```
