# Habakkuk 1

function habakkuk_1_oracleAndComplaint() {
  // === 1:1 - Introduction ===
  prophet.receiveOracle({
    recipient: "Habakkuk_the_prophet",
    method: "vision"
  });

  // === 1:2-4 - Habakkuk's First Complaint ===
  habakkuk.complainToGod({
    context: {
      audience: "LORD"
    },
    grievances: [
      {
        question: "how_long_shall_I_cry_for_help_and_you_will_not_hear",
        issue: "violence",
        divineResponse: "no_answer"
      },
      {
        question: "why_do_you_make_me_see_iniquity_and_why_do_you_idly_look_at_wrong",
        observations: [
          "destruction_and_violence_are_before_me",
          "strife_and_contention_arise"
        ]
      },
      {
        issue: "law_is_paralyzed",
        consequences: [
          "justice_never_goes_forth",
          "wicked_surround_the_righteous",
          "justice_goes_forth_perverted"
        ]
      }
    ]
  });

  // === 1:5-11 - God's Response ===
  god.respondToHabakkuk({
    audience: "Habakkuk_and_nations",
    instruction: "look_among_nations_and_see_wonder_and_be_amazed",
    action: {
      work: "I_am_raising_up_the_Chaldeans",
      description: {
        nature: "bitter_and_hasty_nation",
        actions: [
          "march_through_breadth_of_earth",
          "seize_dwellings_not_their_own"
        ],
        attributes: [
          "dreadful_and_fearsome",
          "justice_and_dignity_go_forth_from_themselves"
        ],
        military: {
          horses: "swifter_than_leopards_fiercer_than_wolves_at_dusk",
          horsemen: "press_on_from_afar_fly_like_eagle_swift_to_devour",
          approach: "come_for_violence_faces_forward_to_gather_captives_like_sand"
        },
        behavior: [
          "scoff_at_kings",
          "laugh_at_fortresses",
          "heap_up_earth_and_capture_them"
        ],
        power: {
          speed: "sweep_by_like_wind",
          guilt: "their_own_might_is_their_god"
        }
      }
    }
  });

  // === 1:12-17 - Habakkuk's Second Complaint ===
  habakkuk.questionGod({
    context: {
      divineAttributes: [
        "are_you_not_from_everlasting_O_LORD_my_God_my_Holy_One",
        "we_shall_not_die",
        "you_O_Rock_have_ordained_them_for_judgment",
        "established_them_for_reproof"
      ]
    },
    objections: [
      {
        question: "why_do_you_idly_look_at_traitors_and_remain_silent",
        issue: "wicked_swallows_up_one_more_righteous_than_he"
      },
      {
        metaphor: "you_make_mankind_like_fish_of_sea_like_crawling_things_with_no_ruler",
        detail: [
          "he_brings_all_of_them_up_with_hook",
          "catches_them_in_his_net",
          "gathers_them_in_his_dragnet",
          "rejoices_and_is_glad"
        ]
      },
      {
        issue: "he_sacrifices_to_his_net_and_makes_offerings_to_his_dragnet",
        reason: "by_them_his_portion_is_fat_and_his_food_plentiful"
      },
      {
        question: "shall_he_keep_on_emptying_his_net_and_mercilessly_killing_nations_forever"
      }
    ]
  });
}
