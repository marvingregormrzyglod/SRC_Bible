# Ezekiel 12
```
function ezekiel_12_rebukeAndSigns() {
  // === 12:1-16 - Sign of the Exile's Baggage ===
  lord.commandEzekiel({
    context: {
      audience: "rebellious_house",
      description: [
        "they_have_eyes_to_see_but_do_not_see",
        "they_have_ears_to_hear_but_do_not_hear"
      ]
    },
    symbolicAct: {
      instructions: [
        "prepare_baggage_for_exile_by_day_in_their_sight",
        "go_out_at_evening_in_their_sight_as_those_going_into_exile",
        "dig_through_wall_in_their_sight_and_go_out_through_it",
        "carry_baggage_on_shoulder_in_their_sight_at_twilight",
        "cover_your_face_so_you_cannot_see_the_land"
      ],
      purpose: "I_have_made_you_a_sign_to_house_of_Israel"
    },
    execution: {
      ezekielAction: [
        "did_as_I_was_commanded",
        "brought_out_baggage_by_day_as_for_exile",
        "dug_through_wall_by_evening_with_my_hands",
        "went_out_in_twilight_carrying_baggage_on_shoulder_in_their_sight"
      ]
    },
    explanation: {
      inquiry: "Lord_asked_what_are_you_doing",
      messageToIsrael: {
        prophecy: [
          "this_burden_concerns_prince_in_Jerusalem_and_all_house_of_Israel",
          "I_will_scatter_them_among_nations_and_dispeerse_them_through_countries",
          "a_few_will_escape_sword_famine_and_pestilence_to_declare_their_abominations_among_nations"
        ],
        purpose: "they_shall_know_I_am_the_Lord"
      }
    }
  });

  // === 12:17-20 - Sign of Trembling and Anxiety ===
  lord.commandEzekiel({
    symbolicAct: {
      instructions: [
        "eat_your_bread_with_trembling",
        "drink_your_water_with_quaking_and_anxiety"
      ],
      messageToPeople: [
        "say_to_people_of_land",
        "inhabitants_of_Jerusalem_and_land_of_Israel_shall_eat_bread_with_anxiety",
        "drink_water_with_dismay",
        "land_will_be_stripped_of_all_it_contains_because_of_violence_of_those_who_dwell_in_it",
        "inhabited_cities_will_be_laid_waste_and_land_will_become_desolate"
      ],
      purpose: "they_shall_know_I_am_the_Lord"
    }
  });

  // === 12:21-28 - Rebuke of False Proverbs and Delayed Prophecies ===
  lord.addressFalseSayings({
    firstSaying: {
      proverb: "days_are_prolonged_and_every_vision_fails",
      lordResponse: [
        "I_will_make_this_proverb_cease",
        "they_shall_no_longer_use_it_in_Israel",
        "say_to_them_the_days_are_near_and_fulfillment_of_every_vision",
        "no_vision_shall_be_vain_nor_divination_flattering_within_house_of_Israel",
        "I_am_the_Lord_I_speak_and_word_I_speak_will_be_performed_without_delay"
      ]
    },
    secondSaying: {
      proverb: "vision_he_sees_is_for_many_days_from_now_and_he_prophesies_of_times_far_off",
      lordResponse: [
        "none_of_My_words_will_be_delayed_any_longer",
        "word_I_speak_will_be_performed",
        "says_the_Lord_God"
      ]
    }
  });
}
```
