# Ezekiel 22
```
function ezekiel_22_judgmentOnJerusalem() {
  // === 22:1-16 - Indictment of Jerusalem’s Sins ===
  lord.instructEzekiel({
    audience: "son_of_man",
    inquiry: "will_you_judge_bloody_city",
    action: "make_known_to_her_all_her_abominations",
    indictment: {
      sins: [
        "sheds_blood_in_her_midst",
        "makes_idols_to_defile_herself",
        "fathers_and_mothers_dishonored",
        "oppress_sojourner_without_justice",
        "wrong_orphan_and_widow",
        "despise_my_holy_things",
        "profane_my_sabbaths",
        "slanderers_uncover_fathers_nakedness",
        "violate_women_during_impurity",
        "commit_abominations_with_neighbor’s_wife",
        "bribe_to_shed_blood",
        "take_interest_and_profit_unjustly"
      ],
      consequence: [
        "you_have_become_guilty_by_blood_shed",
        "defiled_by_idols_you_have_made",
        "brought_your_days_near_and_years_to_end"
      ],
      judgment: {
        action: [
          "I_will_scatter_you_among_nations",
          "disperse_you_through_countries",
          "put_an_end_to_your_uncleanliness"
        ],
        shame: [
          "you_will_be_profaned_in_sight_of_nations",
          "you_shall_know_I_am_Lord"
        ]
      }
    }
  });

  // === 22:17-22 - Refining Furnace for Israel ===
  lord.declareRefining({
    metaphor: {
      houseOfIsrael: "dross_to_me",
      materials: ["silver", "bronze", "iron", "lead", "tin"],
      process: [
        "I_will_gather_you_into_midst_of_Jerusalem",
        "as_one_g Celebrateathers_metal_to_melt_in_furnace",
        "I_will_blow_fire_to_melt_you"
      ],
      purpose: [
        "your_dross_will_be_melted",
        "you_shall_know_I_Lord_have_poured_out_my_wrath"
      ]
    }
  });

  // === 22:23-31 - Corruption of Leaders and People ===
  lord.addressEzekiel({
    prophecy: {
      land: "not_cleansed_or_rained_upon_in_day_of_indignation",
      leaders: [
        {
          role: "prophets",
          sins: [
            "conspire_like_roaring_lion_tearing_prey",
            "devour_human_lives",
            "take_treasure_and_precious_things",
            "make_many_widows"
          ]
        },
        {
          role: "priests",
          sins: [
            "do_violence_to_my_law",
            "profane_my_holy_things",
            "make_no_distinction_between_holy_and_common",
            "hide_eyes_from_my_sabbaths"
          ]
        },
        {
          role: "princes",
          sins: [
            "like_wolves_tearing_prey",
            "shed_blood_and_destroy_lives_for_dishonest_gain"
          ]
        },
        {
          role: "prophets",
          sins: [
            "whitewash_sins_with_false_visions",
            "divine_lies_saying_Lord_speaks_when_I_have_not_spoken"
          ]
        },
        {
          role: "people_of_land",
          sins: [
            "practice_extortion",
            "commit_robbery",
            "oppress_poor_and_needy",
            "wrong_sojourner_without_justice"
          ]
        }
      ],
      divineResponse: {
        search: "I_sought_man_to_stand_in_breach_before_me",
        result: "I_found_none",
        judgment: [
          "I_have_poured_out_my_indignation",
          "consumed_them_with_fire_of_my_wrath",
          "their_own_way_I_have_brought_upon_their_heads"
        ]
      }
    }
  });
}
```
