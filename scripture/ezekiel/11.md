# Ezekiel 11
```
function ezekiel_11_rebukeAndRestoration() {
  // === 11:1-13 - Judgment on Wicked Leaders ===
  lord.visionToEzekiel({
    context: {
      location: "east_gate_of_Lords_house",
      audience: "twenty_five_men_including_Jaazaniah_son_of_Azzur_and_Pelatiah_son_of_Benaiah"
    },
    accusation: {
      leaders: {
        role: "princes_of_the_people",
        sins: [
          "devise_iniquity",
          "give_wicked_counsel_in_this_city",
          "say_it_is_not_near_to_build_houses",
          "call_city_the_cauldron_and_themselves_the_meat"
        ]
      }
    },
    judgment: {
      prophecy: [
        "Lord_will_judge_you_at_border_of_Israel",
        "you_shall_know_I_am_the_Lord",
        "you_have_not_walked_in_My_statutes_nor_executed_My_judgments",
        "you_have_followed_judgments_of_nations_around_you"
      ],
      execution: {
        event: "Pelatiah_son_of_Benaiah_died",
        ezekielResponse: "fell_on_face_and_cried_Ah_Lord_God_will_You_make_full_end_of_remnant_of_Israel"
      }
    }
  });

  // === 11:14-21 - Promise of Restoration ===
  lord.promiseToRemnant({
    recipients: "those_exiled_from_Jerusalem",
    message: {
      reassurance: [
        "though_I_removed_them_far_off_among_nations",
        "though_I_scattered_them_among_countries",
        "I_have_been_a_sanctuary_to_them_for_a_while_in_countries_where_they_have_gone"
      ],
      restoration: [
        "I_will_gather_you_from_peoples",
        "assemble_you_out_of_countries_where_you_have_been_scattered",
        "I_will_give_you_the_land_of_Israel",
        "they_shall_come_there_and_remove_all_its_detestable_things_and_abominations"
      ],
      transformation: [
        "I_will_give_them_one_heart",
        "I_will_put_new_spirit_within_them",
        "I_will_take_heart_of_stone_out_of_their_flesh",
        "I_will_give_them_heart_of_flesh",
        "that_they_may_walk_in_My_statutes_and_keep_My_ordinances"
      ],
      covenant: [
        "they_shall_be_My_people",
        "I_will_be_their_God"
      ],
      warning: [
        "those_whose_heart_goes_after_detestable_things_and_abominations",
        "I_will_bring_their_deeds_upon_their_heads"
      ]
    }
  });

  // === 11:22-25 - Departure of God's Glory ===
  lord.departFromJerusalem({
    vision: {
      description: [
        "cherubim_lifted_up_wings",
        "glory_of_God_above_them",
        "glory_of_Lord_went_up_from_midst_of_city",
        "stood_on_mountain_east_of_city"
      ],
      transport: [
        "Spirit_lifted_me_up",
        "brought_me_in_vision_by_Spirit_of_God_into_Chaldea_to_exiles"
      ]
    },
    ezekielResponse: {
      action: "vision_I_had_seen_went_up_from_me",
      report: "I_told_exiles_all_things_Lord_had_shown_me"
    }
  });
}
```
