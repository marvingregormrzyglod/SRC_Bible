# Ezekiel 39

function ezekiel_39_judgmentAndRestoration() {
  // === 39:1-16 - Continued Judgment on Gog ===
  ezekiel.receiveOracle({
    target: "Gog_chief_prince_of_Meshech_and_Tubal",
    divineCommand: {
      instruction: "prophesy_against_Gog",
      message: {
        declaration: [
          "I_am_against_you_Gog",
          "I_will_turn_you_about_and_drive_you_forward",
          "bring_you_up_from_uttermost_parts_of_north",
          "lead_you_against_mountains_of_Israel"
        ],
        judgment: [
          "I_will_strike_your_bow_from_your_left_hand",
          "make_your_arrows_drop_out_of_your_right_hand",
          "you_shall_fall_on_mountains_of_Israel_you_and_all_your_hordes_and_peoples_with_you",
          "I_will_give_you_to_birds_of_prey_of_every_sort_and_beasts_of_field_to_be_devoured"
        ],
        purpose: [
          "you_shall_fall_in_open_field_for_I_have_spoken",
          "I_will_send_fire_on_Magog_and_on_those_who_dwell_securely_in_coastlands",
          "they_shall_know_that_I_am_the_Lord"
        ]
      }
    },
    divineIntent: {
      revelation: [
        "I_will_make_my_holy_name_known_in_midst_of_my_people_Israel",
        "I_will_not_let_my_holy_name_be_profaned_anymore"
      ],
      outcome: [
        "nations_shall_know_that_I_am_the_Lord_the_Holy_One_in_Israel",
        "it_is_coming_and_it_shall_be_done"
      ]
    },
    aftermath: {
      scene: {
        timing: "in_that_day",
        location: "mountains_of_Israel",
        event: [
          "Gog_and_all_his_hordes_fall",
          "I_will_give_Gog_a_place_for_burial_in_Israel",
          "Valley_of_Travelers_east_of_sea"
        ],
        burial: {
          process: [
            "it_will_block_travelers",
            "there_Gog_and_all_his_multitude_will_be_buried",
            "called_Valley_of_Hamon-gog"
          ],
          duration: "seven_months_house_of_Israel_will_be_burying_them",
          purpose: "to_cleanse_the_land"
        },
        participants: [
          "all_people_of_land_will_bury_them",
          "it_will_be_a_day_of_renown_for_me"
        ]
      },
      cleanup: {
        process: [
          "after_seven_months_they_will_make_search",
          "pass_through_land_and_if_anyone_sees_human_bone_set_up_sign_by_it",
          "until_gravediggers_bury_it_in_Valley_of_Hamon-gog"
        ],
        location: "city_called_Hamonah",
        result: "thus_shall_they_cleanse_the_land"
      }
    }
  });

  // === 39:17-20 - Sacrificial Feast ===
  ezekiel.deliverMessage({
    recipient: "son_of_man",
    command: {
      instruction: "speak_to_birds_of_every_sort_and_all_beasts_of_field",
      message: {
        invitation: [
          "assemble_and_come",
          "gather_from_all_around_to_my_sacrificial_feast",
          "I_am_preparing_for_you_a_great_sacrificial_feast_on_mountains_of_Israel"
        ],
        menu: [
          "eat_flesh_and_drink_blood",
          "eat_flesh_of_mighty_and_drink_blood_of_princes_of_earth",
          "rams_lambs_goats_bulls_all_fatlings_of_Bashan",
          "eat_fat_till_you_are_filled_and_drink_blood_till_you_are_drunk"
        ],
        outcome: "you_shall_be_filled_at_my_table_with_horses_and_charioteers_mighty_men_and_all_kinds_of_warriors"
      }
    }
  });

  // === 39:21-29 - Restoration of Israel ===
  ezekiel.proclaimRestoration({
    divinePurpose: {
      glory: [
        "I_will_set_my_glory_among_nations",
        "all_nations_shall_see_my_judgment_that_I_have_executed",
        "my_hand_that_I_have_laid_on_them"
      ],
      recognition: [
        "house_of_Israel_shall_know_that_I_am_the_Lord_their_God_from_that_day_forward",

        "because_they_dealt_treacherously_with_me"
      ]
    },
    pastAction: [
      "I_hid_my_face_from_them",
      "gave_them_into_hand_of_their_adversaries",
      "they_fell_by_sword_according_to_their_uncleanliness_and_transgressions"
    ],
    restoration: {
      timing: "now",
      actions: [
        "I_will_restore_fortunes_of_Jacob",
        "have_mercy_on_whole_house_of_Israel",
        "I_will_be_jealous_for_my_holy_name"
      ],
      outcome: [
        "they_shall_forget_their_shame_and_all_treachery_they_practiced_against_me",
        "when_they_dwell_securely_in_their_land_with_none_to_make_them_afraid",
        "when_I_have_brought_them_back_from_peoples_and_gathered_them_from_enemies_lands"
      ]
    },
    finalPromise: {
      sanctification: [
        "I_will_vindicate_my_holiness_through_them_in_sight_of_many_nations",
        "they_shall_know_that_I_am_the_Lord_their_God"
      ],
      return: [
        "I_sent_them_into_exile_among_nations",
        "then_gathered_them_into_their_own_land"
      ],
      presence: [
        "I_will_leave_none_of_them_remaining_among_nations_anymore",
        "I_will_not_hide_my_face_anymore_from_them",
        "I_will_pour_out_my_Spirit_upon_house_of_Israel"
      ]
    }
  });
}
