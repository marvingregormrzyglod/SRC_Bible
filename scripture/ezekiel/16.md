# Ezekiel 16
```
function ezekiel_16_jerusalemAllegory() {
  // === 16:1-14 - Jerusalem’s Origins and God’s Grace ===
  lord.addressEzekiel({
    context: {
      recipient: "son_of_man",
      purpose: "make_Jerusalem_know_her_abominations"
    },
    allegory: {
      origins: {
        description: [
          "your_origin_and_birth_from_land_of_Canaanites",
          "father_was_Amorite_mother_Hittite",
          "on_day_you_were_born_your_cord_not_cut",
          "not_washed_with_water_not_rubbed_with_salt_not_wrapped_in_cloths",
          "no_eye_pitied_you_no_one_had_compassion",
          "cast_out_in_open_field_despised"
        ]
      },
      divineIntervention: {
        action: [
          "I_said_to_you_in_your_blood_Live",
          "made_you_flourish_like_plant_of_field",
          "you_grew_and_developed_became_most_beautiful",
          "breasts_formed_hair_grown_but_naked_and_bare"
        ],
        covenant: [
          "I_passed_by_saw_you_at_age_for_love",
          "spread_my_cloak_over_you_covered_your_nakedness",
          "swore_to_you_entered_covenant_you_became_mine"
        ],
        adornment: [
          "washed_you_with_water_cleansed_blood_anointed_with_oil",
          "clothed_you_with_embroidered_cloth_sandals_of_fine_leather",
          "wrapped_you_in_fine_linen_covered_with_silk",
          "adorned_with_ornaments_bracelets_rings_necklace_crown",
          "ate_fine_flour_honey_oil",
          "became_exceedingly_beautiful_fit_to_be_queen",
          "fame_spread_among_nations_because_of_your_beauty",
          "it_was_perfect_through_my_splendor"
        ]
      }
    }
  });

  // === 16:15-34 - Jerusalem’s Unfaithfulness ===
  lord.accuseJerusalem({
    betrayal: {
      idolatry: [
        "trusted_in_your_beauty_acted_like_prostitute",
        "lavished_your_favors_on_any_passerby",
        "used_your_garments_to_make_high_places_decorated_with_colors",
        "committed_prostitution_on_them",
        "took_my_jewelry_gold_silver_made_male_idols",
        "committed_prostitution_with_them",
        "offered_my_oil_incense_fine_flour_honey_as_sweet_savor"
      ],
      childSacrifice: [
        "slaughtered_my_children",
        "offered_them_up_to_idols_by_burning"
      ],
      ingratitude: [
        "did_not_remember_days_of_your_youth_when_naked_and_bare",
        "wallowing_in_your_blood"
      ]
    },
    comparison: {
      prostitute: {
        behavior: [
          "unlike_other_women_you_were_not_hired",
          "gave_gifts_to_lovers_bribed_them_to_come_to_you",
          "lavished_prostitution_on_all_sons_of_Egypt_Philistines_Assyria_Chaldea"
        ],
        uniqueness: "in_your_prostitution_you_were_different",
        outcome: "no_one_solicited_you_for_hire"
      }
    }
  });

  // === 16:35-43 - Judgment on Jerusalem ===
  lord.pronounceJudgment({
    accusation: "O_prostitute_hear_word_of_the_Lord",
    consequences: [
      {
        action: [
          "I_will_gather_all_your_lovers_with_whom_you_took_pleasure",
          "gather_them_against_you_from_every_side",
          "uncover_your_nakedness_to_them",
          "they_shall_see_all_your_nakedness"
        ],
        punishment: [
          "judge_you_as_women_who_commit_adultery_shed_blood",
          "bring_blood_upon_you_in_wrath_and_jealousy",
          "give_you_into_their_hands",
          "they_shall_tear_down_your_high_places_strip_your_clothes_take_jewelry",
          "leave_you_naked_and_bare",
          "bring_up_a_mob_against_you",
          "stone_you_cut_you_to_pieces_with_swords",
          "burn_your_houses_with_fire"
        ]
      },
      purpose: [
        "execute_judgments_in_sight_of_many_women",
        "make_you_stop_playing_prostitute",
        "no_longer_pay_your_lovers",
        "satisfy_my_anger_jealousy_turn_away_from_you"
      ]
    }
  });

  // === 16:44-58 - Comparison with Samaria and Sodom ===
  lord.compareJerusalem({
    proverb: "like_mother_like_daughter",
    family: {
      mother: "Hittite_who_despised_husband_and_children",
      sisters: [
        "elder_sister_Samaria_with_her_daughters",
        "younger_sister_Sodom_with_her_daughters"
      ],
      behavior: [
        "you_did_not_walk_in_their_ways_or_act_according_to_their_abominations",
        "but_soon_became_more_corrupt_than_them",
        "your_sister_Sodom_and_her_daughters_not_as_wicked_as_you",
        "Samaria_did_not_commit_half_your_sins",
        "you_multiplied_abominations_more_than_they"
      ]
    },
    sodomSins: [
      "pride_excess_food_easeful_prosperity",
      "did_not_aid_poor_and_needy",
      "haughty_did_abominable_things_before_me",
      "I_removed_them_when_I_saw_it"
    ],
    jerusalemSins: [
      "made_sisters_appear_righteous_by_your_abominations",
      "bear_your_own_shame_for_your_sins",
      "more_abominable_than_theirs"
    ],
    judgment: [
      "bear_your_disgrace",
      "made_judgment_favorable_to_your_sisters",
      "Sodom_and_her_daughters_restored_to_former_state",
      "Samaria_and_her_daughters_restored_to_former_state",
      "you_and_your_daughters_restored_to_former_state",
      "bear_your_disgrace_and_be_ashamed"
    ]
  });

  // === 16:59-63 - Covenant Restoration ===
  lord.declareRestoration({
    judgment: [
      "deal_with_you_as_you_have_done",
      "despised_oath_by_breaking_covenant"
    ],
    promise: [
      "I_will_remember_my_covenant_with_you_in_days_of_your_youth",
      "establish_everlasting_covenant_with_you",
      "give_your_sisters_to_you_as_daughters_not_by_your_covenant",
      "establish_my_covenant_with_you",
      "you_shall_know_that_I_am_the_Lord"
    ],
    purpose: [
      "you_may_remember_and_be_confounded",
      "never_open_your_mouth_because_of_your_shame",
      "when_I_atone_for_all_you_have_done"
    ]
  });
}
```
