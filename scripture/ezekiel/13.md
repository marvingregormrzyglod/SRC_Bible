# Ezekiel 13
```
function ezekiel_13_rebukeFalseProphets() {
  // === 13:1-16 - Judgment on False Prophets ===
  lord.commandEzekiel({
    target: "prophets_of_Israel_who_prophesy_from_their_own_hearts",
    accusation: {
      behavior: [
        "prophesy_out_of_their_own_minds",
        "follow_their_own_spirit",
        "have_seen_nothing"
      ],
      metaphor: [
        "like_jackals_among_ruins",
        "have_not_gone_up_into_breaches",
        "have_not_built_wall_for_house_of_Israel_to_stand_in_battle_in_day_of_Lord"
      ],
      falseClaims: [
        "say_Lord_declares_it_when_Lord_has_not_spoken",
        "prophesy_vain_visions_and_lying_divinations",
        "say_Lord_says_though_I_have_not_spoken"
      ]
    },
    judgment: {
      declaration: [
        "My_hand_will_be_against_prophets_who_see_false_visions_and_utter_lying_divinations",
        "they_shall_not_be_in_council_of_My_people",
        "they_shall_not_be_written_in_register_of_house_of_Israel",
        "they_shall_not_enter_land_of_Israel"
      ],
      purpose: "you_shall_know_I_am_the_Lord_God",
      reason: [
        "they_have_misled_My_people",
        "say_peace_when_there_is_no_peace",
        "people_build_wall_and_prophets_plaster_it_with_whitewash"
      ],
      destruction: [
        "I_will_break_down_wall_you_have_plastered_with_whitewash",
        "bring_it_down_to_ground_so_its_foundation_is_laid_bare",
        "it_will_fall_and_you_will_perish_in_its_midst"
      ],
      naturalJudgment: [
        "I_will_spend_My_wrath_upon_wall_and_those_who_plastered_it_with_whitewash",
        "I_will_send_torrential_rain_hailstones_and_stormy_wind_to_break_it_down"
      ],
      outcome: [
        "wall_will_be_no_more_nor_those_who_plastered_it",
        "I_will_say_to_you_where_is_the_whitewash_you_plastered"
      ],
      purpose: "thus_I_will_spend_My_wrath_and_you_shall_know_I_am_the_Lord"
    }
  });

  // === 13:17-23 - Judgment on False Prophetesses ===
  lord.commandEzekiel({
    target: "daughters_of_your_people_who_prophesy_from_their_own_hearts",
    accusation: {
      behavior: [
        "sew_magic_bands_on_wrists",
        "make_veils_for_heads_of_persons_of_every_stature",
        "hunt_souls_of_My_people",
        "keep_souls_alive_for_your_profit"
      ],
      consequences: [
        "slay_souls_that_should_not_die",
        "keep_alive_souls_that_should_not_live",
        "lie_to_My_people_who_listened_to_lies",
        "dishonored_Me_among_My_people_for_handfuls_of_barley_and_pieces_of_bread"
      ]
    },
    judgment: {
      declaration: [
        "I_am_against_your_magic_bands_with_which_you_hunt_souls_like_birds",
        "I_will_tear_them_from_your_arms",
        "I_will_let_souls_go_free_like_birds_you_hunt"
      ],
      actions: [
        "I_will_tear_your_veils_and_deliver_My_people_out_of_your_hand",
        "they_shall_no_longer_be_prey_in_your_hand"
      ],
      purpose: "you_shall_know_I_am_the_Lord",
      reason: [
        "you_have_made_hearts_of_righteous_sad_whom_I_have_not_made_sad",
        "you_have_strengthened_hands_of_wicked_so_he_does_not_turn_from_his_wicked_way"
      ],
      outcome: [
        "you_shall_no_longer_see_vain_visions_nor_practice_divination",
        "I_will_deliver_My_people_out_of_your_hand"
      ],
      purpose: "you_shall_know_I_am_the_Lord"
    }
  });
}
```
