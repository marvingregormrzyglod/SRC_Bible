# Ezekiel 18
```
function ezekiel_18_individualResponsibility() {
  // === 18:1-4 - Refuting the Proverb of Sour Grapes ===
  lord.addressIsrael({
    audience: "house_of_Israel",
    issue: {
      proverb: "fathers_have_eaten_sour_grapes_and_childrens_teeth_are_set_on_edge",
      critique: "you_shall_not_use_this_proverb_anymore_in_Israel"
    },
    declaration: {
      speaker: "Lord_God",
      principle: [
        "all_souls_are_mine",
        "soul_of_father_and_soul_of_son_are_mine",
        "soul_who_sins_shall_die"
      ]
    }
  });

  // === 18:5-20 - Principles of Righteousness and Sin ===
  lord.defineJustice({
    cases: [
      {
        case: "righteous_man",
        actions: [
          "does_what_is_just_and_right",
          "does_not_eat_at_mountain_shrines",
          "does_not_lift_eyes_to_idols_of_Israel",
          "does_not_defile_neighbors_wife",
          "does_not_approach_woman_in_her_impurity",
          "does_not_oppress_anyone",
          "restores_debtors_pledge",
          "does_not_commit_robbery",
          "gives_bread_to_hungry",
          "covers_naked_with_clothing",
          "does_not_lend_at_interest_or_take_profit",
          "withholds_hand_from_injustice",
          "executes_true_justice_between_man_and_man",
          "walks_in_my_statutes_and_keeps_my_rules"
        ],
        outcome: "he_is_righteous_he_shall_surely_live"
      },
      {
        case: "son_of_righteous_man",
        condition: "violent_son_who_does_not_do_like_father",
        actions: [
          "eats_at_mountain_shrines",
          "defiles_neighbors_wife",
          "oppresses_poor_and_needy",
          "commits_robbery",
          "does_not_restore_pledge",
          "lifts_eyes_to_idols",
          "commits_abomination",
          "lends_at_interest_and_takes_profit"
        ],
        question: "shall_he_live",
        outcome: [
          "he_shall_not_live",
          "he_has_done_all_these_abominations",
          "he_shall_surely_die",
          "his_blood_shall_be_upon_himself"
        ]
      },
      {
        case: "son_of_wicked_man",
        condition: "does_not_do_like_father",
        actions: [
          "sees_fathers_sins_but_does_not_do_likewise",
          "does_not_eat_at_mountain_shrines",
          "does_not_defile_neighbors_wife",
          "does_not_oppress_anyone",
          "does_not_take_pledge",
          "does_not_commit_robbery",
          "gives_bread_to_hungry",
          "covers_naked_with_clothing",
          "withholds_hand_from_iniquity",
          "does_not_take_interest_or_profit",
          "executes_my_rules",
          "walks_in_my_statutes"
        ],
        outcome: [
          "he_shall_not_die_for_fathers_iniquity",
          "he_shall_surely_live"
        ]
      },
      {
        case: "father_of_righteous_son",
        actions: [
          "does_what_is_not_good",
          "all_his_transgressions_remain_upon_him"
        ],
        outcome: "he_shall_die_in_his_iniquity"
      }
    ],
    principle: [
      "son_shall_not_suffer_for_fathers_iniquity",
      "father_shall_not_suffer_for_sons_iniquity",
      "righteousness_of_righteous_shall_be_upon_himself",
      "wickedness_of_wicked_shall_be_upon_himself"
    ]
  });

  // === 18:21-29 - Repentance and Judgment ===
  lord.declareRepentance({
    scenarios: [
      {
        case: "wicked_man_repents",
        actions: [
          "turns_away_from_all_his_sins",
          "keeps_my_statutes",
          "does_what_is_just_and_right"
        ],
        outcome: [
          "he_shall_surely_live",
          "he_shall_not_die",
          "none_of_his_transgressions_shall_be_remembered_against_him",
          "in_his_righteousness_he_shall_live"
        ]
      },
      {
        case: "righteous_man_turns_to_wickedness",
        actions: [
          "turns_away_from_his_righteousness",
          "does_injustice",
          "does_same_abominations_as_wicked"
        ],
        question: "shall_he_live",
        outcome: [
          "none_of_his_righteous_deeds_shall_be_remembered",
          "in_his_treachery_and_sin_he_shall_die"
        ]
      }
    ],
    responseToObjection: {
      objection: "way_of_Lord_is_not_just",
      rebuttal: [
        "are_my_ways_not_just",
        "are_not_your_ways_unjust",
        "I_judge_each_one_according_to_his_ways"
      ]
    }
  });

  // === 18:30-32 - Call to Repentance ===
  lord.callToRepentance({
    audience: "house_of_Israel",
    exhortation: [
      "I_will_judge_each_by_his_ways",
      "repent_and_turn_from_all_your_transgressions",
      "let_iniquity_not_be_your_ruin",
      "cast_away_all_transgressions",
      "make_yourselves_a_new_heart_and_new_spirit"
    ],
    question: "why_will_you_die_house_of_Israel",
    desire: {
      speaker: "Lord_God",
      statement: [
        "I_have_no_pleasure_in_death_of_anyone",
        "turn_and_live"
      ]
    }
  });
}
```
