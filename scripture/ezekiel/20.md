# Ezekiel 20
```
function ezekiel_20_rebellionAndRestoration() {
  // === 20:1-4 - Introduction and Inquiry ===
  context.setHistoricalStage({
    date: "seventh_year_fifth_month_tenth_day",
    setting: "elders_of_Israel_come_to_inquire_of_Lord",
    location: "exile_in_Babylon"
  });

  lord.addressEzekiel({
    audience: "son_of_man",
    inquiryResponse: {
      question: "will_you_judge_them",
      instruction: "make_known_to_them_abominations_of_their_fathers"
    }
  });

  // === 20:5-9 - Rebellion in Egypt ===
  lord.recallCovenant({
    time: "day_I_chose_Israel",
    actions: [
      "swore_to_seed_of_house_of_Jacob",
      "made_myself_known_in_land_of_Egypt",
      "swore_I_am_Lord_your_God"
    ],
    command: [
      "cast_away_detestable_things_of_their_eyes",
      "do_not_defile_yourselves_with_idols_of_Egypt"
    ],
    rebellion: {
      response: "they_did_not_cast_away_detestable_things",
      consequence: "I_said_I_would_pour_out_my_wrath_in_Egypt",
      restraint: "acted_for_sake_of_my_name_not_to_be_profaned_among_nations"
    }
  });

  // === 20:10-17 - Rebellion in the Wilderness (First Generation) ===
  lord.recallWilderness({
    actions: [
      "brought_them_out_of_Egypt",
      "led_them_into_wilderness",
      "gave_them_my_statutes_and_rules_by_which_man_shall_live"
    ],
    gift: {
      sabbaths: "sign_between_me_and_them_to_know_I_am_Lord_who_sanctifies"
    },
    rebellion: {
      response: [
        "despised_my_statutes",
        "did_not_walk_in_my_rules",
        "profaned_my_sabbaths"
      ],
      consequence: "I_said_I_would_pour_out_my_wrath_in_wilderness_to_consume_them",
      restraint: "acted_for_sake_of_my_name_not_to_be_profaned_among_nations"
    }
  });

  // === 20:18-26 - Rebellion in the Wilderness (Second Generation) ===
  lord.addressSecondGeneration({
    command: [
      "do_not_walk_in_statutes_of_your_fathers",
      "keep_my_rules_and_statutes",
      "hallow_my_sabbaths"
    ],
    rebellion: {
      response: [
        "rebelled_against_me",
        "did_not_walk_in_my_statutes",
        "profaned_my_sabbaths"
      ],
      consequence: [
        "I_said_I_would_pour_out_my_wrath",
        "I_gave_them_statutes_not_good_and_rules_by_which_they_could_not_live",
        "defiled_them_through_their_gifts_in_offering_all_their_firstborn"
      ],
      purpose: "that_they_might_know_I_am_Lord"
    }
  });

  // === 20:27-29 - Rebellion in the Promised Land ===
  lord.recallPromisedLand({
    rebellion: [
      "blasphemed_me_by_acting_treacherously",
      "offered_gifts_on_every_high_hill_and_under_every_green_tree",
      "provoked_me_with_high_places"
    ]
  });

  // === 20:30-38 - Present Rebellion and Future Judgment ===
  lord.addressExiles({
    accusation: [
      "defile_yourselves_after_manner_of_your_fathers",
      "whore_after_their_detestable_things",
      "pollute_yourselves_with_idols_to_this_day"
    ],
    inquiryResponse: {
      refusal: "I_will_not_be_inquired_of_by_you"
    },
    judgment: {
      declaration: "I_will_rule_over_you_with_strong_hand_and_outstretched_arm",
      process: [
        "bring_you_out_from_peoples_and_gather_you_from_countries",
        "bring_you_into_wilderness_of_peoples",
        "enter_into_judgment_with_you_face_to_face"
      ],
      outcome: [
        "purge_out_rebels_and_transgressors",
        "bring_you_into_bond_of_covenant",
        "you_shall_know_I_am_Lord"
      ]
    }
  });

  // === 20:39-44 - Future Restoration ===
  lord.declareRestoration({
    command: [
      "go_serve_your_idols_now",
      "but_afterward_you_shall_listen_to_me"
    ],
    prophecy: {
      worship: [
        "all_house_of_Israel_shall_serve_me_on_my_holy_mountain",
        "accept_your_sacrifices_and_firstfruits"
      ],
      restoration: [
        "gather_you_from_countries_where_you_are_scattered",
        "deal_with_you_for_my_name_sake_not_according_to_your_evil_ways"
      ],
      realization: [
        "you_shall_know_I_am_Lord_when_I_deal_with_you_for_my_name_sake",
        "you_shall_loathe_yourselves_for_all_your_evils"
      ]
    }
  });

  // === 20:45-49 - Parable of the Southern Forest Fire ===
  lord.instructEzekiel({
    audience: "son_of_man",
    action: "set_your_face_toward_southland",
    prophecy: {
      parable: {
        subject: "forest_land_in_south",
        judgment: [
          "I_will_kindle_fire_in_you",
          "it_shall_devour_every_green_tree_and_every_dry_tree",
          "blazing_flame_shall_not_be_quenched"
        ],
        scope: "all_faces_from_south_to_north_shall_be_scorched"
      },
      purpose: "all_may_know_I_Lord_have_kindled_it"
    }
  });

  elders.respond({
    complaint: "does_he_not_speak_in_parables"
  });
}
```
